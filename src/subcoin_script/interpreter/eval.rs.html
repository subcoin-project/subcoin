23pub fn eval_script<SC: SignatureChecker>(
24    stack: &mut Stack,
25    script: &Script,
26    flags: &VerifyFlags,
27    checker: &mut SC,
28    sig_version: SigVersion,
29    exec_data: &mut ScriptExecutionData,
30) -> Result<bool, Error> {
31    use crate::opcode::Opcode::*;
32
33    if matches!(sig_version, SigVersion::Taproot) {
34        return Err(Error::NoScriptExecution);
35    }
36
37    let mut alt_stack = Stack::with_flags(flags);
38
39    // Create a vector of conditional execution states
40    let mut exec_stack: Vec<bool> = Vec::new();
41
42    let mut begincode = 0;
43    let mut op_count = 0;
44    let mut opcode_pos = 0;
45
46    let step = |instruction: &Instruction| match instruction {
47        Instruction::PushBytes(b) => b.as_bytes().len() + 1,
48        Instruction::Op(_) => 1,
49    };
50
51    let instructions = if flags.intersects(VerifyFlags::MINIMALDATA) {
52        script.instruction_indices_minimal()
53    } else {
54        script.instruction_indices()
55    };
56
57    let mut pc = 0;
58
59    for instruction in instructions {
60        let (_pos, instruction) = instruction.map_err(Error::ReadInstruction)?;
61
62        pc += step(&instruction);
63
64        let executing = exec_stack.iter().all(|x| *x);
65
66        let should_handle_instruction = executing
67            || instruction
68                .opcode()
69                .map(|opcode| {
70                    (CONST_OP_IF.to_u8()..=CONST_OP_ENDIF.to_u8()).contains(&opcode.to_u8())
71                })
72                .unwrap_or(false);
73
74        if !should_handle_instruction {
75            tracing::debug!("Ignoring instruction: {instruction:?}");
76            continue;
77        }
78
79        match instruction {
80            Instruction::PushBytes(p) => {
81                if p.len() > MAX_SCRIPT_ELEMENT_SIZE {
82                    return Err(Error::PushSize);
83                }
84                // TODO: avoid allocation?
85                stack.push(p.as_bytes().to_vec());
86            }
87            Instruction::Op(op) => {
88                if matches!(sig_version, SigVersion::Base | SigVersion::WitnessV0) {
89                    // Note how OP_RESERVED does not count towards the opcode limit.
90                    if op.to_u8() > CONST_OP_PUSHNUM_16.to_u8() {
91                        op_count += 1;
92
93                        if op_count > MAX_OPS_PER_SCRIPT {
94                            return Err(Error::OpCount);
95                        }
96                    }
97                }
98
99                let opcode =
100                    crate::opcode::Opcode::from_u8(op.to_u8()).ok_or(Error::UnknownOpcode(op))?;
101
102                match opcode {
103                    OP_0 | OP_PUSHBYTES_1 | OP_PUSHBYTES_2 | OP_PUSHBYTES_3 | OP_PUSHBYTES_4
104                    | OP_PUSHBYTES_5 | OP_PUSHBYTES_6 | OP_PUSHBYTES_7 | OP_PUSHBYTES_8
105                    | OP_PUSHBYTES_9 | OP_PUSHBYTES_10 | OP_PUSHBYTES_11 | OP_PUSHBYTES_12
106                    | OP_PUSHBYTES_13 | OP_PUSHBYTES_14 | OP_PUSHBYTES_15 | OP_PUSHBYTES_16
107                    | OP_PUSHBYTES_17 | OP_PUSHBYTES_18 | OP_PUSHBYTES_19 | OP_PUSHBYTES_20
108                    | OP_PUSHBYTES_21 | OP_PUSHBYTES_22 | OP_PUSHBYTES_23 | OP_PUSHBYTES_24
109                    | OP_PUSHBYTES_25 | OP_PUSHBYTES_26 | OP_PUSHBYTES_27 | OP_PUSHBYTES_28
110                    | OP_PUSHBYTES_29 | OP_PUSHBYTES_30 | OP_PUSHBYTES_31 | OP_PUSHBYTES_32
111                    | OP_PUSHBYTES_33 | OP_PUSHBYTES_34 | OP_PUSHBYTES_35 | OP_PUSHBYTES_36
112                    | OP_PUSHBYTES_37 | OP_PUSHBYTES_38 | OP_PUSHBYTES_39 | OP_PUSHBYTES_40
113                    | OP_PUSHBYTES_41 | OP_PUSHBYTES_42 | OP_PUSHBYTES_43 | OP_PUSHBYTES_44
114                    | OP_PUSHBYTES_45 | OP_PUSHBYTES_46 | OP_PUSHBYTES_47 | OP_PUSHBYTES_48
115                    | OP_PUSHBYTES_49 | OP_PUSHBYTES_50 | OP_PUSHBYTES_51 | OP_PUSHBYTES_52
116                    | OP_PUSHBYTES_53 | OP_PUSHBYTES_54 | OP_PUSHBYTES_55 | OP_PUSHBYTES_56
117                    | OP_PUSHBYTES_57 | OP_PUSHBYTES_58 | OP_PUSHBYTES_59 | OP_PUSHBYTES_60
118                    | OP_PUSHBYTES_61 | OP_PUSHBYTES_62 | OP_PUSHBYTES_63 | OP_PUSHBYTES_64
119                    | OP_PUSHBYTES_65 | OP_PUSHBYTES_66 | OP_PUSHBYTES_67 | OP_PUSHBYTES_68
120                    | OP_PUSHBYTES_69 | OP_PUSHBYTES_70 | OP_PUSHBYTES_71 | OP_PUSHBYTES_72
121                    | OP_PUSHBYTES_73 | OP_PUSHBYTES_74 | OP_PUSHBYTES_75 | OP_PUSHDATA1
122                    | OP_PUSHDATA2 | OP_PUSHDATA4 => {
123                        unreachable!("Instruction::Op(opcode) contains non-push opcode only");
124                    }
125
126                    // Constants
127                    OP_1NEGATE | OP_1 | OP_2 | OP_3 | OP_4 | OP_5 | OP_6 | OP_7 | OP_8 | OP_9
128                    | OP_10 | OP_11 | OP_12 | OP_13 | OP_14 | OP_15 | OP_16 => {
129                        let value = (opcode as u8 as i32).wrapping_sub(OP_1 as u8 as i32 - 1);
130                        stack.push_num(value as i64);
131                    }
132
133                    // Flow control
134                    OP_NOP => {}
135                    OP_IF | OP_NOTIF => {
136                        let mut value = false;
137
138                        if executing {
139                            let top = stack.pop()?;
140
141                            // Tapscript requires minimal IF/NOTIF inputs as a consensus rule.
142                            if matches!(sig_version, SigVersion::Tapscript) {
143                                // The input argument to the OP_IF and OP_NOTIF opcodes must be either
144                                // exactly 0 (the empty vector) or exactly 1 (the one-byte vector with value 1).
145                                if top.len() > 1 || (top.len() == 1 && top[0] != 1) {
146                                    return Err(Error::TaprootMinimalif);
147                                }
148                            }
149
150                            // Under witness v0 rules it is only a policy rule, enabled through SCRIPT_VERIFY_MINIMALIF.
151                            if matches!(sig_version, SigVersion::WitnessV0)
152                                && flags.intersects(VerifyFlags::MINIMALIF)
153                            {
154                                #[allow(clippy::collapsible_if)]
155                                if top.len() > 1 || (top.len() == 1 && top[0] != 1) {
156                                    return Err(Error::Minimalif);
157                                }
158                            }
159
160                            value = crate::stack::cast_to_bool(&top);
161
162                            if opcode == OP_NOTIF {
163                                value = !value;
164                            }
165                        }
166
167                        exec_stack.push(value);
168                    }
169                    OP_ELSE => {
170                        // Toggle top.
171                        if let Some(last) = exec_stack.last_mut() {
172                            *last = !*last;
173                        } else {
174                            return Err(Error::UnbalancedConditional);
175                        }
176                    }
177                    OP_ENDIF => {
178                        if exec_stack.is_empty() {
179                            return Err(Error::UnbalancedConditional);
180                        }
181                        exec_stack.pop();
182                    }
183                    OP_VERIFY => {
184                        let exec_value = stack.pop_bool()?;
185                        if !exec_value {
186                            return Err(Error::Verify(op));
187                        }
188                    }
189                    OP_RETURN => return Err(Error::OpReturn),
190
191                    // Stack
192                    OP_TOALTSTACK => {
193                        alt_stack.push(stack.pop()?);
194                    }
195                    OP_FROMALTSTACK => {
196                        let v = alt_stack
197                            .pop()
198                            .map_err(|_| Error::InvalidAltStackOperation)?;
199                        stack.push(v);
200                    }
201                    OP_2DROP => stack.drop(2)?,
202                    OP_2DUP => stack.dup(2)?,
203                    OP_3DUP => stack.dup(3)?,
204                    OP_2OVER => stack.over(2)?,
205                    OP_2ROT => stack.rot(2)?,
206                    OP_2SWAP => stack.swap(2)?,
207                    OP_IFDUP => {
208                        if stack.peek_bool()? {
209                            stack.dup(1)?;
210                        }
211                    }
212                    OP_DEPTH => {
213                        // Push the current number of stack items onto the stack.
214                        stack.push_num(stack.len() as i64);
215                    }
216                    OP_DROP => stack.drop(1)?,
217                    OP_DUP => stack.dup(1)?,
218                    OP_NIP => stack.nip()?,
219                    OP_OVER => stack.over(1)?,
220                    OP_PICK | OP_ROLL => {
221                        // Pop the top stack element as N.
222                        let n = stack.pop_num()?.value();
223                        if n < 0 || n >= stack.len() as i64 {
224                            return Err(StackError::InvalidOperation.into());
225                        }
226                        let v = if opcode == OP_PICK {
227                            // Copy the Nth stack element to the top.
228                            stack.top(n as usize)?.clone()
229                        } else {
230                            // Move the Nth stack element to the top.
231                            stack.remove(n as usize)?
232                        };
233                        stack.push(v);
234                    }
235                    OP_ROT => stack.rot(1)?,
236                    OP_SWAP => stack.swap(1)?,
237                    OP_TUCK => stack.tuck()?,
238
239                    // Splice
240                    OP_CAT | OP_SUBSTR | OP_LEFT | OP_RIGHT => {
241                        return Err(Error::DisabledOpcode(op));
242                    }
243                    OP_SIZE => {
244                        stack.push_num(stack.last()?.len() as i64);
245                    }
246
247                    // Bitwise logic
248                    OP_INVERT | OP_AND | OP_OR | OP_XOR => return Err(Error::DisabledOpcode(op)),
249                    OP_EQUAL => {
250                        let equal = stack.pop()? == stack.pop()?;
251                        stack.push_bool(equal);
252                    }
253                    OP_EQUALVERIFY => {
254                        let equal = stack.pop()? == stack.pop()?;
255                        if !equal {
256                            return Err(Error::Verify(op));
257                        }
258                    }
259
260                    // Arithmetic
261                    OP_2MUL | OP_2DIV | OP_MUL | OP_DIV | OP_MOD | OP_LSHIFT | OP_RSHIFT => {
262                        return Err(Error::DisabledOpcode(op));
263                    }
264                    OP_1ADD => {
265                        let n = stack.pop_num()?.add(1.into())?;
266                        stack.push_num(n);
267                    }
268                    OP_1SUB => {
269                        let n = stack.pop_num()?.sub(1.into())?;
270                        stack.push_num(n);
271                    }
272                    OP_NEGATE => {
273                        let n = stack.pop_num()?.neg()?;
274                        stack.push_num(n);
275                    }
276                    OP_ABS => {
277                        let n = stack.pop_num()?.abs();
278                        stack.push_num(n);
279                    }
280                    OP_NOT => {
281                        let n = ScriptNum::from(stack.pop_num()?.is_zero());
282                        stack.push_num(n);
283                    }
284                    OP_0NOTEQUAL => {
285                        let n = ScriptNum::from(!stack.pop_num()?.is_zero());
286                        stack.push_num(n);
287                    }
288                    OP_ADD => {
289                        let v1 = stack.pop_num()?;
290                        let v2 = stack.pop_num()?;
291                        stack.push_num((v1 + v2)?);
292                    }
293                    OP_SUB => {
294                        let v1 = stack.pop_num()?;
295                        let v2 = stack.pop_num()?;
296                        stack.push_num((v2 - v1)?);
297                    }
298                    OP_BOOLAND => {
299                        let v1 = !stack.pop_num()?.is_zero();
300                        let v2 = !stack.pop_num()?.is_zero();
301                        stack.push_num(v1 && v2);
302                    }
303                    OP_BOOLOR => {
304                        let v1 = !stack.pop_num()?.is_zero();
305                        let v2 = !stack.pop_num()?.is_zero();
306                        stack.push_num(v1 || v2);
307                    }
308                    OP_NUMEQUAL => {
309                        let v1 = stack.pop_num()?;
310                        let v2 = stack.pop_num()?;
311                        stack.push_num(v1 == v2);
312                    }
313                    OP_NUMEQUALVERIFY => {
314                        let v1 = stack.pop_num()?;
315                        let v2 = stack.pop_num()?;
316                        if v1 != v2 {
317                            return Err(Error::Verify(op));
318                        }
319                    }
320                    OP_NUMNOTEQUAL => {
321                        let v1 = stack.pop_num()?;
322                        let v2 = stack.pop_num()?;
323                        stack.push_num(v1 != v2);
324                    }
325                    OP_LESSTHAN => {
326                        let v1 = stack.pop_num()?;
327                        let v2 = stack.pop_num()?;
328                        stack.push_num(v2 < v1);
329                    }
330                    OP_GREATERTHAN => {
331                        let v1 = stack.pop_num()?;
332                        let v2 = stack.pop_num()?;
333                        stack.push_num(v2 > v1);
334                    }
335                    OP_LESSTHANOREQUAL => {
336                        let v1 = stack.pop_num()?;
337                        let v2 = stack.pop_num()?;
338                        stack.push_num(v2 <= v1);
339                    }
340                    OP_GREATERTHANOREQUAL => {
341                        let v1 = stack.pop_num()?;
342                        let v2 = stack.pop_num()?;
343                        stack.push_num(v2 >= v1);
344                    }
345                    OP_MIN => {
346                        let v1 = stack.pop_num()?;
347                        let v2 = stack.pop_num()?;
348                        stack.push_num(v1.min(v2));
349                    }
350                    OP_MAX => {
351                        let v1 = stack.pop_num()?;
352                        let v2 = stack.pop_num()?;
353                        stack.push_num(v1.max(v2));
354                    }
355                    OP_WITHIN => {
356                        // [v3 v2 v1] = [x min max]
357                        let v1 = stack.pop_num()?;
358                        let v2 = stack.pop_num()?;
359                        let v3 = stack.pop_num()?;
360                        stack.push_bool((v2..v1).contains(&v3));
361                    }
362
363                    // Crypto
364                    OP_RIPEMD160 => {
365                        let v = ripemd160::Hash::hash(&stack.pop()?);
366                        stack.push(v.to_byte_array().to_vec());
367                    }
368                    OP_SHA1 => {
369                        let v = sha1::Hash::hash(&stack.pop()?);
370                        stack.push(v.to_byte_array().to_vec());
371                    }
372                    OP_SHA256 => {
373                        let v = sha256::Hash::hash(&stack.pop()?);
374                        stack.push(v.to_byte_array().to_vec());
375                    }
376                    OP_HASH160 => {
377                        let v = hash160::Hash::hash(&stack.pop()?);
378                        stack.push(v.to_byte_array().to_vec());
379                    }
380                    OP_HASH256 => {
381                        let v = sha256d::Hash::hash(&stack.pop()?);
382                        stack.push(v.to_byte_array().to_vec());
383                    }
384                    OP_CODESEPARATOR => {
385                        // Use of OP_CODESEPARATOR is rejected in pre-segwit script.
386                        if matches!(sig_version, SigVersion::Base)
387                            && flags.intersects(VerifyFlags::CONST_SCRIPTCODE)
388                        {
389                            return Err(Error::OpCodeSeparator);
390                        }
391
392                        begincode = pc;
393                        exec_data.codeseparator_pos = opcode_pos;
394                    }
395                    OP_CHECKSIG | OP_CHECKSIGVERIFY => {
396                        // [sig pubkey] -> bool
397                        let pubkey = stack.pop()?;
398                        let sig = stack.pop()?;
399
400                        let success = sig::handle_checksig(
401                            &sig,
402                            &pubkey,
403                            script,
404                            begincode,
405                            exec_data,
406                            flags,
407                            checker,
408                            sig_version,
409                        )?;
410
411                        match opcode {
412                            OP_CHECKSIG => {
413                                stack.push_bool(success);
414                            }
415                            OP_CHECKSIGVERIFY if !success => {
416                                return Err(Error::Verify(op));
417                            }
418                            _ => {}
419                        }
420                    }
421                    OP_CHECKMULTISIG | OP_CHECKMULTISIGVERIFY => {
422                        let multisig_op = if opcode == OP_CHECKMULTISIG {
423                            multisig::MultiSigOp::CheckMultiSig
424                        } else {
425                            multisig::MultiSigOp::CheckMultiSigVerify
426                        };
427
428                        multisig::handle_checkmultisig(
429                            stack,
430                            flags,
431                            begincode,
432                            script,
433                            sig_version,
434                            checker,
435                            multisig_op,
436                            &mut op_count,
437                        )?;
438                    }
439                    OP_CHECKSIGADD => {
440                        // (sig num pubkey -- num)
441                        let pubkey = stack.pop()?;
442                        let num = stack.pop_num()?;
443                        let sig = stack.pop()?;
444
445                        let success = sig::handle_checksig(
446                            &sig,
447                            &pubkey,
448                            script,
449                            begincode,
450                            exec_data,
451                            flags,
452                            checker,
453                            sig_version,
454                        )
455                        .map_err(Error::CheckSig)?;
456
457                        let v = num.value() + if success { 1 } else { 0 };
458                        stack.push_num(v);
459                    }
460
461                    // Locktime, CLTV
462                    OP_CHECKLOCKTIMEVERIFY => {
463                        // not enabled; treat as a NOP3
464                        if !flags.intersects(VerifyFlags::CHECKLOCKTIMEVERIFY) {
465                            continue;
466                        }
467
468                        // Note that elsewhere numeric opcodes are limited to
469                        // operands in the range -2**31+1 to 2**31-1, however it is
470                        // legal for opcodes to produce results exceeding that
471                        // range. This limitation is implemented by CScriptNum's
472                        // default 4-byte limit.
473                        //
474                        // If we kept to that limit we'd have a year 2038 problem,
475                        // even though the nLockTime field in transactions
476                        // themselves is uint32 which only becomes meaningless
477                        // after the year 2106.
478                        //
479                        // Thus as a special case we tell CScriptNum to accept up
480                        // to 5-byte bignums, which are good until 2**39-1, well
481                        // beyond the 2**32-1 limit of the nLockTime field itself.
482                        let lock_time = ScriptNum::from_bytes(
483                            stack.last()?,
484                            flags.intersects(VerifyFlags::MINIMALDATA),
485                            Some(5),
486                        )?;
487
488                        // In the rare event that the argument may be < 0 due to
489                        // some arithmetic being done first, you can always use
490                        // 0 MAX CHECKLOCKTIMEVERIFY.
491                        if lock_time.is_negative() {
492                            return Err(Error::NegativeLocktime);
493                        }
494
495                        if !checker.check_lock_time(lock_time) {
496                            return Err(Error::UnsatisfiedLocktime);
497                        }
498                    }
499                    // CSV
500                    OP_CHECKSEQUENCEVERIFY => {
501                        // not enabled; treat as a NOP3
502                        if !flags.intersects(VerifyFlags::CHECKSEQUENCEVERIFY) {
503                            continue;
504                        }
505
506                        // opcodeCheckSequenceVerify compares the top item on the data stack to the
507                        // LockTime field of the transaction containing the script signature
508                        // validating if the transaction outputs are spendable yet.  If flag
509                        // ScriptVerifyCheckSequenceVerify is not set, the code continues as if OP_NOP3
510                        // were executed.
511                        let sequence = ScriptNum::from_bytes(
512                            stack.last()?,
513                            flags.intersects(VerifyFlags::MINIMALDATA),
514                            Some(5),
515                        )?;
516
517                        if sequence.is_negative() {
518                            return Err(Error::NegativeLocktime);
519                        }
520
521                        if (sequence.value() & SEQUENCE_LOCKTIME_DISABLE_FLAG as i64) == 0
522                            && !checker.check_sequence(sequence)
523                        {
524                            return Err(Error::UnsatisfiedLocktime);
525                        }
526                    }
527
528                    // Reserved words
529                    OP_RESERVED | OP_VER | OP_RESERVED1 | OP_RESERVED2 => {
530                        if executing {
531                            return Err(Error::DisabledOpcode(op));
532                        }
533                    }
534                    OP_VERIF | OP_VERNOTIF => {
535                        return Err(Error::DisabledOpcode(op));
536                    }
537                    OP_NOP1 | OP_NOP4 | OP_NOP5 | OP_NOP6 | OP_NOP7 | OP_NOP8 | OP_NOP9
538                    | OP_NOP10 => {
539                        if flags.intersects(VerifyFlags::DISCOURAGE_UPGRADABLE_NOPS) {
540                            return Err(Error::DiscourageUpgradableNops);
541                        }
542                    }
543                }
544            }
545        }
546
547        opcode_pos += 1;
548
549        if stack.len() + alt_stack.len() > MAX_STACK_SIZE {
550            return Err(Error::StackSize);
551        }
552    }
553
554    if !exec_stack.is_empty() {
555        return Err(Error::UnbalancedConditional);
556    }
557
558    let success = !stack.is_empty() && stack.peek_bool()?;
559
560    Ok(success)
561}
