9/// Multisig error type.
10#[derive(Debug, Eq, PartialEq, thiserror::Error)]
11pub enum CheckMultiSigError {
12    #[error("Invalid number of pubkeys, expected in the range of [0, {MAX_PUBKEYS_PER_MULTISIG}]")]
13    InvalidPubkeyCount,
14    #[error("Exceeded max OPS limit {MAX_OPS_PER_SCRIPT}")]
15    TooManyOps,
16    #[error("Invalid number of signatures, expected in the range of [0, {0}]")]
17    InvalidSignatureCount(usize),
18    #[error("Multisig dummy argument has length {0} instead of 0")]
19    SignatureNullDummy(usize),
20    #[error("OP_CHECKMULTISIG and OP_CHECKMULTISIGVERIFY are disabled during tapscript execution")]
21    TapscriptCheckMultiSig,
22    #[error("CHECK_MULTISIGVERIFY failed")]
23    CheckMultiSigVerify,
24    #[error(transparent)]
25    CheckSig(#[from] CheckSigError),
26    #[error(transparent)]
27    InvalidSignature(#[from] SignatureError),
28    #[error(transparent)]
29    Stack(#[from] StackError),
30}

32pub enum MultiSigOp {
33    CheckMultiSig,
34    CheckMultiSigVerify,

37/// Handles the OP_CHECKMULTISIG opcode.
38///
39/// This opcode validates a multi-signature script against the provided public keys and signatures.
40#[allow(clippy::too_many_arguments)]
41pub(super) fn handle_checkmultisig(
42    stack: &mut Stack,
43    flags: &VerifyFlags,
44    begincode: usize,
45    script: &Script,
46    sig_version: SigVersion,
47    checker: &mut impl SignatureChecker,
48    multisig_op: MultiSigOp,
49    op_count: &mut usize,
50) -> Result<(), CheckMultiSigError> {
51    let success = eval_checkmultisig(
52        stack,
53        flags,
54        begincode,
55        script,
56        sig_version,
57        checker,
58        op_count,
59    )?;
60
61    match multisig_op {
62        MultiSigOp::CheckMultiSig => {
63            stack.push_bool(success);
64        }
65        MultiSigOp::CheckMultiSigVerify if !success => {
66            return Err(CheckMultiSigError::CheckMultiSigVerify);
67        }
68        _ => {}
69    }
70
71    Ok(())
72}

74fn eval_checkmultisig(
75    stack: &mut Stack,
76    flags: &VerifyFlags,
77    begincode: usize,
78    script: &Script,
79    sig_version: SigVersion,
80    checker: &mut impl SignatureChecker,
81    op_count: &mut usize,
82) -> Result<bool, CheckMultiSigError> {
83    if matches!(sig_version, SigVersion::Tapscript) {
84        return Err(CheckMultiSigError::TapscriptCheckMultiSig);
85    }
86
87    // ([sig ...] num_of_signatures [pubkey ...] num_of_pubkeys -- bool)
88
89    let keys_count = stack.pop_num()?;
90    if keys_count < 0.into() || keys_count > MAX_PUBKEYS_PER_MULTISIG.into() {
91        return Err(CheckMultiSigError::InvalidPubkeyCount);
92    }
93
94    let keys_count = keys_count.value() as usize;
95
96    *op_count += keys_count;
97
98    if *op_count > MAX_OPS_PER_SCRIPT {
99        return Err(CheckMultiSigError::TooManyOps);
100    }
101
102    let mut keys = Vec::with_capacity(keys_count);
103    for _ in 0..keys_count {
104        keys.push(stack.pop()?);
105    }
106
107    let sigs_count = stack.pop_num()?.value();
108    if sigs_count < 0 || sigs_count as usize > keys_count {
109        return Err(CheckMultiSigError::InvalidSignatureCount(keys_count));
110    }
111
112    let sigs_count = sigs_count as usize;
113    let mut sigs = Vec::with_capacity(sigs_count);
114    for _ in 0..sigs_count {
115        sigs.push(stack.pop()?);
116    }
117
118    // A bug in the original Satoshi client implementation means one more
119    // stack value than should be used must be popped.  Unfortunately, this
120    // buggy behavior is now part of the consensus and a hard fork would be
121    // required to fix it.
122    let dummy = stack.pop()?;
123
124    // Since the dummy argument is otherwise not checked, it could be any
125    // value which unfortunately provides a source of malleability.  Thus,
126    // there is a script flag to force an error when the value is NOT 0.
127    if flags.intersects(VerifyFlags::NULLDUMMY) && !dummy.is_empty() {
128        return Err(CheckMultiSigError::SignatureNullDummy(dummy.len()));
129    }
130
131    // Drop the signature in pre-segwit scripts but not segwit scripts
132    let mut subscript = script.as_bytes()[begincode..].to_vec();
133
134    for signature in &sigs {
135        if matches!(sig_version, SigVersion::Base) {
136            let mut push_buf = PushBytesBuf::with_capacity(signature.len());
137            push_buf
138                .extend_from_slice(signature)
139                .expect("Signature length must be within length limits; qed");
140            let sig_script = Builder::default().push_slice(push_buf).into_script();
141            let found = find_and_delete(&mut subscript, sig_script.as_bytes());
142            if found > 0 && flags.intersects(VerifyFlags::CONST_SCRIPTCODE) {
143                return Err(CheckSigError::FindAndDelete.into());
144            }
145        }
146    }
147
148    let subscript = Script::from_bytes(&subscript);
149
150    // Verify signatures against public keys.
151    let mut success = true;
152    let mut checked_keys_count = 0;
153    let mut satisfied_sigs_count = 0;
154
155    while satisfied_sigs_count < sigs.len() && success {
156        let key = &keys[checked_keys_count];
157        let sig = &sigs[satisfied_sigs_count];
158
159        check_signature_encoding(sig, flags)?;
160        check_pubkey_encoding(key, flags, sig_version)?;
161
162        let signature_is_correct = check_ecdsa_signature(sig, key, checker, subscript, sig_version);
163
164        if signature_is_correct {
165            satisfied_sigs_count += 1;
166        }
167
168        checked_keys_count += 1;
169
170        // Early exit if remaining keys can't satisfy remaining signatures.
171        success = keys.len() - checked_keys_count >= sigs.len() - satisfied_sigs_count;
172    }
173
174    Ok(success)
175}
