5/// Script number error type.
6#[derive(Debug, Eq, PartialEq, thiserror::Error)]
7pub enum NumError {
8    #[error("Script number overflow")]
9    Overflow,
10    #[error("Non-minimally encoded script number")]
11    NotMinimallyEncoded,
12}
13
14/// A numeric type used in Bitcoin Script operations.
15#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
16pub struct ScriptNum {
17    value: i64,
18}
19
20impl<T: Into<i64>> From<T> for ScriptNum {
21    fn from(value: T) -> Self {
22        Self {
23            value: value.into(),
24        }
25    }
26}

28impl ScriptNum {
29    /// Maximum script number length in bytes.
30    pub const MAX_NUM_SIZE: usize = 4;
31
32    /// Construct a [`ScriptNum`] with size validation.
33    pub fn from_bytes(
34        data: &[u8],
35        require_minimal: bool,
36        max_size: Option<usize>,
37    ) -> Result<Self, NumError> {
38        let max_size = max_size.unwrap_or(Self::MAX_NUM_SIZE);
39
40        if data.len() > max_size {
41            return Err(NumError::Overflow);
42        }
43
44        if data.is_empty() {
45            return Ok(Self { value: 0 });
46        }
47
48        // Check for minimal encoding if required
49        if require_minimal && !Self::is_minimally_encoded(data) {
50            return Err(NumError::NotMinimallyEncoded);
51        }
52
53        let mut result = 0i64;
54
55        // Parse bytes in little-endian order.
56        for (i, &byte) in data.iter().enumerate() {
57            result |= i64::from(byte).wrapping_shl(8 * i as u32);
58        }
59
60        // Check sign bit and extend if necessary.
61        if data
62            .last()
63            .expect("Data must not be empty as checked above; qed")
64            & 0x80
65            != 0
66        {
67            let value = -(result & !(0x80i64.wrapping_shl((8 * (data.len() - 1)) as u32)));
68            Ok(Self { value })
69        } else {
70            Ok(Self { value: result })
71        }
72    }
73
74    /// Convert the number to a minimally encoded byte vector.
75    pub fn as_bytes(&self) -> Vec<u8> {
76        if self.value == 0 {
77            return vec![];
78        }
79
80        let mut result = Vec::new();
81
82        let mut abs_value = self.value.abs();
83
84        while abs_value != 0 {
85            result.push((abs_value & 0xff) as u8);
86            abs_value >>= 8;
87        }
88
89        let negative = self.value < 0;
90
91        // Handle sign bit
92        if result.last().unwrap() & 0x80 != 0 {
93            result.push(if negative { 0x80 } else { 0 });
94        } else if negative {
95            let len = result.len();
96            result[len - 1] |= 0x80;
97        }
98
99        result
100    }
101
102    /// Check if the byte array is minimally encoded.
103    fn is_minimally_encoded(data: &[u8]) -> bool {
104        if data.is_empty() {
105            return true;
106        }
107
108        // Check leading zeros.
109        if data[data.len() - 1] & 0x7f == 0
110            && (data.len() <= 1 || (data[data.len() - 2] & 0x80) == 0)
111        {
112            return false;
113        }
114
115        true
116    }
117
118    /// Get the underlying value.
119    pub fn value(&self) -> i64 {
120        self.value
121    }
122
123    pub fn is_zero(&self) -> bool {
124        self.value == 0
125    }
126
127    pub fn is_negative(&self) -> bool {
128        self.value.is_negative()
129    }
130
131    pub fn abs(&self) -> Self {
132        self.value.abs().into()
133    }
134}

136impl Add for ScriptNum {
137    type Output = Result<Self, NumError>;
138
139    fn add(self, other: Self) -> Result<Self, NumError> {
140        self.value
141            .checked_add(other.value)
142            .map(|value| Self { value })
143            .ok_or(NumError::Overflow)
144    }
145}
146
147impl Sub for ScriptNum {
148    type Output = Result<Self, NumError>;
149
150    fn sub(self, other: Self) -> Result<Self, NumError> {
151        self.value
152            .checked_sub(other.value)
153            .map(|value| Self { value })
154            .ok_or(NumError::Overflow)
155    }
156}

162        self.value

165            .ok_or(NumError::Overflow)
166    }
