8// A function to generate the keystream from the test vectors
9pub fn chacha20_block(key: &[u8; 32], nonce: &[u8; 12], counter: u32) -> Vec<u8> {
10    let mut cipher = ChaCha20::new(key.into(), nonce.into());
11
12    // Set the counter (as the starting position in the keystream)
13    cipher.seek(counter as u64 * 64); // Each block is 64 bytes
14
15    let mut keystream = vec![0u8; 64];
16    cipher.apply_keystream(&mut keystream);
17
18    keystream
19}
