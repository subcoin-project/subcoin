12/// The state of syncing between a Peer and ourselves.
13#[derive(Clone, Eq, PartialEq, Ord, PartialOrd, Serialize, Deserialize, Hash)]
14#[serde(rename_all = "camelCase")]
15pub enum SyncState {
16    Available,
17    Deprioritized,
18    DownloadingNew,
19}
20
21/// Overview of peers in chain sync.
22#[derive(Clone, Serialize, Deserialize)]
23#[serde(rename_all = "camelCase")]
24pub struct SyncPeers {
25    /// A map containing the count of peers in each sync state (e.g., syncing, idle).
26    peer_counts: BTreeMap<SyncState, usize>,
27    /// The highest block height known across all peers in the network.
28    best_known_block: Option<u32>,
29    /// Detailed synchronization information for each peer.
30    peer_sync_details: Vec<PeerSync>,
31}
32
33#[rpc(client, server)]
34pub trait NetworkApi {
35    /// Get the sync peers.
36    #[method(name = "network_syncPeers")]
37    async fn network_sync_peers(&self) -> Result<SyncPeers, Error>;
38
39    /// Get overall network status.
40    #[method(name = "network_status")]
41    async fn network_status(&self) -> Result<Option<NetworkStatus>, Error>;
42
43    /// Trigger the block sync manually.
44    ///
45    /// This API is unsafe and primarily for the local development.
46    #[method(name = "network_startBlockSync", with_extensions)]
47    fn network_start_block_sync(&self) -> Result<(), Error>;
48}

79    async fn network_sync_peers(&self) -> Result<SyncPeers, Error> {
80        let mut sync_peers = self.network_api.sync_peers().await;
81
82        let mut available = 0;
83        let mut deprioritized = 0;
84        let mut downloading_new = 0;
85
86        let mut best_known_block = 0;
87
88        for peer in &sync_peers {
89            match peer.state {
90                PeerSyncState::Available => available += 1,
91                PeerSyncState::Deprioritized { .. } => deprioritized += 1,
92                PeerSyncState::DownloadingNew { .. } => downloading_new += 1,
93            }
94
95            if peer.best_number > best_known_block {
96                best_known_block = peer.best_number;
97            }
98        }
99
100        sync_peers.sort_by_key(|x| x.latency);
101
102        Ok(SyncPeers {
103            peer_counts: BTreeMap::from([
104                (SyncState::Available, available),
105                (SyncState::Deprioritized, deprioritized),
106                (SyncState::DownloadingNew, downloading_new),
107            ]),
108            best_known_block: (best_known_block > 0).then_some(best_known_block),
109            peer_sync_details: sync_peers,
110        })
111    }
112
113    async fn network_status(&self) -> Result<Option<NetworkStatus>, Error> {
114        Ok(self.network_api.status().await)
115    }
116
117    fn network_start_block_sync(&self, ext: &Extensions) -> Result<(), Error> {
118        sc_rpc_api::check_if_safe(ext)?;
119
120        self.network_api.start_block_sync();
121
122        Ok(())
123    }
