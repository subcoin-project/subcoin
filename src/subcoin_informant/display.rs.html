31impl Display for DisplayBlockHash {
32    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
33        const BYTES: usize = 32;
34        let bytes = self.0.as_byte_array();
35        for i in bytes[BYTES - 2..BYTES].iter().rev() {
36            write!(f, "{i:02x}")?;
37        }
38        write!(f, "…")?;
39        for i in bytes[0..3].iter().rev() {
40            write!(f, "{i:02x}")?;
41        }
42        Ok(())
43    }
44}

46/// State of the informant display system.
47///
48/// This is the system that handles the line that gets regularly printed and that looks something
49/// like:
50///
51/// > Syncing  5.4 bps, target=#531028 (4 peers), best: #90683 (0x4ca8…51b8),
52/// > finalized #360 (0x6f24…a38b), ⬇ 5.5kiB/s ⬆ 0.9kiB/s
53///
54/// # Usage
55///
56/// Call `InformantDisplay::new` to initialize the state, then regularly call `display` with the
57/// information to display.
58pub struct InformantDisplay<B: BlockT> {
59    /// Head of chain block number from the last time `display` has been called.
60    /// `None` if `display` has never been called.
61    last_number: Option<NumberFor<B>>,
62    /// The last time `display` or `new` has been called.
63    last_update: Instant,
64    /// The last seen total of bytes received.
65    last_total_bytes_inbound: u64,
66    /// The last seen total of bytes sent.
67    last_total_bytes_outbound: u64,
68}
69
70impl<B: BlockT> InformantDisplay<B> {
71    /// Builds a new informant display system.
72    pub fn new() -> InformantDisplay<B> {
73        InformantDisplay {
74            last_number: None,
75            last_update: Instant::now(),
76            last_total_bytes_inbound: 0,
77            last_total_bytes_outbound: 0,
78        }
79    }
80
81    /// Displays the informant by calling `info!`.
82    pub fn display(&mut self, info: ClientInfoExt<B>, net_status: NetworkStatus) {
83        let best_number = info.chain.best_number;
84        let best_hash = info.chain.best_hash;
85        let best_bitcoin_hash = DisplayBlockHash(info.best_bitcoin_hash);
86        let finalized_number = info.chain.finalized_number;
87        let finalized_bitcoin_hash = DisplayBlockHash(info.finalized_bitcoin_hash);
88        let speed = speed::<B>(best_number, self.last_number, self.last_update);
89        let num_connected_peers = net_status.num_connected_peers;
90        let total_bytes_inbound = net_status.total_bytes_inbound;
91        let total_bytes_outbound = net_status.total_bytes_outbound;
92
93        let now = Instant::now();
94        let elapsed = (now - self.last_update).as_secs();
95        self.last_update = now;
96        self.last_number = Some(best_number);
97
98        let diff_bytes_inbound = total_bytes_inbound - self.last_total_bytes_inbound;
99        let diff_bytes_outbound = total_bytes_outbound - self.last_total_bytes_outbound;
100        let (avg_bytes_per_sec_inbound, avg_bytes_per_sec_outbound) = if elapsed > 0 {
101            self.last_total_bytes_inbound = total_bytes_inbound;
102            self.last_total_bytes_outbound = total_bytes_outbound;
103            (diff_bytes_inbound / elapsed, diff_bytes_outbound / elapsed)
104        } else {
105            (diff_bytes_inbound, diff_bytes_outbound)
106        };
107
108        let (level, status, target) = match net_status.sync_status {
109            SyncStatus::Idle => ("💤", "Idle".into(), "".into()),
110            SyncStatus::Downloading { target, .. } => {
111                let progress = best_number.saturated_into::<u32>() as f64 * 100.0 / target as f64;
112                (
113                    "⚙️ ",
114                    format!("Syncing{speed}"),
115                    format!(", target=#{target} ({progress:.2}%)"),
116                )
117            }
118            SyncStatus::Importing { target, .. } => {
119                let progress = best_number.saturated_into::<u32>() as f64 * 100.0 / target as f64;
120                (
121                    "⚙️ ",
122                    format!("Preparing{speed}"),
123                    format!(", target=#{target} ({progress:.2}%)"),
124                )
125            }
126        };
127
128        let finalized_hash = info.chain.finalized_hash;
129
130        tracing::info!(
131            target: "subcoin",
132            "{level} {}{target} ({} peers), best: #{} ({best_bitcoin_hash},{best_hash}), finalized #{} ({finalized_bitcoin_hash},{finalized_hash}), ⬇ {} ⬆ {}",
133            style(status).white().bold(),
134            style(num_connected_peers).white().bold(),
135            style(best_number).white().bold(),
136            style(finalized_number).white().bold(),
137            style(TransferRateFormat(avg_bytes_per_sec_inbound)).green(),
138            style(TransferRateFormat(avg_bytes_per_sec_outbound)).red(),
139        )
140    }
141}

143/// Calculates `(best_number - last_number) / (now - last_update)` and returns a `String`
144/// representing the speed of import.
145fn speed<B: BlockT>(
146    best_number: NumberFor<B>,
147    last_number: Option<NumberFor<B>>,
148    last_update: Instant,
149) -> String {
150    // Number of milliseconds elapsed since last time.
151    let elapsed_ms = {
152        let elapsed = last_update.elapsed();
153        let since_last_millis = elapsed.as_secs() * 1000;
154        let since_last_subsec_millis = elapsed.subsec_millis() as u64;
155        since_last_millis + since_last_subsec_millis
156    };
157
158    // Number of blocks that have been imported since last time.
159    let diff = match last_number {
160        None => return String::new(),
161        Some(n) => best_number.saturating_sub(n),
162    };
163
164    if let Ok(diff) = TryInto::<u128>::try_into(diff) {
165        // If the number of blocks can be converted to a regular integer, then it's easy: just
166        // do the math and turn it into a `f64`.
167        let speed = diff
168            .saturating_mul(10_000)
169            .checked_div(u128::from(elapsed_ms))
170            .map_or(0.0, |s| s as f64)
171            / 10.0;
172        format!(" {speed:4.1} bps")
173    } else {
174        // If the number of blocks can't be converted to a regular integer, then we need a more
175        // algebraic approach and we stay within the realm of integers.
176        let one_thousand = NumberFor::<B>::from(1_000u32);
177        let elapsed =
178            NumberFor::<B>::from(<u32 as TryFrom<_>>::try_from(elapsed_ms).unwrap_or(u32::MAX));
179
180        let speed = diff
181            .saturating_mul(one_thousand)
182            .checked_div(&elapsed)
183            .unwrap_or_else(Zero::zero);
184        format!(" {speed} bps")
185    }
186}

188/// Contains a number of bytes per second. Implements `fmt::Display` and shows this number of bytes
189/// per second in a nice way.
190struct TransferRateFormat(u64);
191impl fmt::Display for TransferRateFormat {
192    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
193        // Special case 0.
194        if self.0 == 0 {
195            return write!(f, "0");
196        }
197
198        // Under 0.1 kiB, display plain bytes.
199        if self.0 < 100 {
200            return write!(f, "{} B/s", self.0);
201        }
202
203        // Under 1.0 MiB/sec, display the value in kiB/sec.
204        if self.0 < 1024 * 1024 {
205            return write!(f, "{:.1}kiB/s", self.0 as f64 / 1024.0);
206        }
207
208        write!(f, "{:.1}MiB/s", self.0 as f64 / (1024.0 * 1024.0))
209    }
210}

212#[test]
213fn test_display_block_hash() {
214    let genesis_hash: BlockHash =
215        "000000000019d6689c085ae165831e934ff763ae46a2a6c172b3f1b60a8ce26f"
216            .parse()
217            .unwrap();
218    assert_eq!(
219        format!("{}", DisplayBlockHash(genesis_hash)),
220        "0000…8ce26f".to_string()
221    );
