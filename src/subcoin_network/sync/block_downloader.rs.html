29/// A writer that only counts the number of bytes written, without storing them.
30/// This is used for efficient size calculation without memory allocation.

33}

56/// Manages queued blocks and their memory usage.
57#[derive(Default, Debug, Clone)]
58pub(crate) struct QueuedBlocks {
59    hash2number: HashMap<BlockHash, u32>,
60    number2hash: HashMap<u32, BlockHash>,
61    /// Tracks memory usage for each block hash.
62    block_memory: HashMap<BlockHash, usize>,
63}
64
65impl QueuedBlocks {
66    pub(crate) fn block_hash(&self, number: u32) -> Option<BlockHash> {
67        self.number2hash.get(&number).copied()
68    }
69
70    pub(crate) fn block_number(&self, hash: BlockHash) -> Option<u32> {
71        self.hash2number.get(&hash).copied()
72    }
73
74    /// Gets the cached memory usage for a block.
75    pub(crate) fn block_memory_usage(&self, hash: BlockHash) -> Option<usize> {
76        self.block_memory.get(&hash).copied()
77    }
78
79    fn insert(&mut self, number: u32, hash: BlockHash, memory_usage: usize) {
80        self.hash2number.insert(hash, number);
81        self.number2hash.insert(number, hash);
82        self.block_memory.insert(hash, memory_usage);
83    }
84
85    fn remove(&mut self, hash: &BlockHash) {
86        if let Some(number) = self.hash2number.remove(hash) {
87            self.number2hash.remove(&number);
88        }
89        self.block_memory.remove(hash);
90    }
91
92    #[cfg(test)]
93    fn clear(&mut self) {
94        self.hash2number.clear();
95        self.number2hash.clear();
96        self.block_memory.clear();
97    }

101pub(crate) enum ImportQueueStatus {
102    /// Queue is ready to accept more blocks for import.
103    Ready,
104    /// The import system is saturated and cannot accept more blocks at the moment.
105    ///
106    /// This can occur due to either:
107    /// - Too many blocks pending in the import queue (block count limit)
108    /// - Memory pressure from downloaded blocks exceeding configured limits
109    Saturated,
110}

112impl ImportQueueStatus {
113    pub(crate) fn is_saturated(&self) -> bool {
114        matches!(self, Self::Saturated)
115    }
116
117    pub(crate) fn is_ready(&self) -> bool {
118        matches!(self, Self::Ready)
119    }

122/// Responsible for downloading blocks from a specific peer.
123///
124/// This struct keeps track of:
125/// - Blocks that have been requested from the network but not yet received.
126/// - Blocks that have been downloaded but not yet sent to the import queue.
127/// - Blocks that are currently in the import queue awaiting processing.
128/// - The highest block number that is either in the queue or pending to be queued.
129///
130/// [`BlockDownloader`] is designed to be used in both Blocks-First and
131/// Headers-First sync strategies, providing a common component for managing
132/// the state of blocks during the sync process.
133#[derive(Clone)]
134pub(crate) struct BlockDownloader {
135    pub(super) peer_id: PeerId,
136    /// Blocks awaiting to be downloaded.
137    pub(super) missing_blocks: Vec<BlockHash>,
138    /// A set of block hashes that have been requested from the network.
139    ///
140    /// This helps tracking which blocks are pending download.
141    pub(super) requested_blocks: HashSet<BlockHash>,
142    /// A vector of blocks that have been downloaded from the network.
143    ///
144    /// These blocks are ready to be sent to the import queue.
145    pub(super) downloaded_blocks: Vec<BitcoinBlock>,
146    /// A map of block hashes to their respective heights the import queue.
147    /// This helps tracking which blocks are currently being processed in the import queue.
148    pub(super) blocks_in_queue: HashMap<BlockHash, u32>,
149    /// Pending blocks that either the queue or waiting to be queued.
150    pub(super) queued_blocks: QueuedBlocks,
151    /// The highest block number that is either the queue or waiting to be queued.
152    pub(super) best_queued_number: u32,
153    /// Orphan blocks
154    pub(super) orphan_blocks_pool: OrphanBlocksPool,
155    /// Last time at which the block was received or imported.
156    ///
157    /// This is updated whenever a block is received from the network or
158    /// when the results of processed blocks are notified. It helps track
159    /// the most recent activity related to block processing.
160    pub(super) last_progress_time: Instant,
161    /// Import queue status.
162    pub(super) queue_status: ImportQueueStatus,
163    /// Last time the log of saturated queue was printed.
164    pub(super) last_saturated_queue_log_time: Option<Instant>,
165    /// Peer store.
166    pub(super) peer_store: Arc<dyn PeerStore>,
167    /// Tracks the number of blocks downloaded from the current sync peer.
168    pub(super) downloaded_blocks_count: usize,
169    /// Memory management configuration.
170    pub(super) memory_config: MemoryConfig,
171    /// Current memory usage for downloaded blocks in bytes.
172    pub(super) downloaded_blocks_memory: usize,
173    /// Size of the last requested batch (for pipelining).
174    ///
175    /// Used to determine when to request the next batch before all blocks
176    /// in the current batch are received.
177    last_batch_size: usize,
178}

242    /// Checks if we should request the next batch of blocks (pipelining).
243    ///
244    /// Returns `true` when we've received enough blocks from the current batch
245    /// to justify requesting the next batch. This reduces idle time waiting
246    /// for the last few blocks in a batch.
247    ///
248    /// Pipelining is triggered when:
249    /// - There are more blocks to download (`missing_blocks` not empty)
250    /// - We've received at least 50% of the current batch
251    /// - We haven't already started pipelining (requested_blocks not empty)
252    pub(crate) fn should_pipeline(&self) -> bool {
253        // Only pipeline if there are more blocks to download
254        if self.missing_blocks.is_empty() {
255            return false;
256        }
257
258        // Don't pipeline if we've already sent another request (requested_blocks would be full again)
259        // This is detected by checking if requested_blocks is less than half the original batch
260        if self.last_batch_size == 0 {
261            return false;
262        }
263
264        let received_count = self
265            .last_batch_size
266            .saturating_sub(self.requested_blocks.len());
267        let pipeline_threshold = self.last_batch_size / 2;
268
269        // Pipeline when we've received at least 50% of the batch
270        received_count >= pipeline_threshold && !self.requested_blocks.is_empty()
271    }

290    /// Determine if the downloader is stalled based on the time elapsed since the last progress
291    /// update.
292    ///
293    /// The stall detection is influenced by the size of the blockchain and the time since the last
294    /// successful block processing. As the chain grows, the following factors contribute to the need
295    /// for an extended timeout:
296    ///
297    /// - **Increased Local Block Execution Time**: Larger chain state lead to longer execution times
298    ///   when processing blocks (primarily due to the state root computation).
299    /// - **Higher Average Block Size**: As the blockchain grows, the average size of blocks typically
300    ///   increases, resulting in longer network response times for block retrieval.
301    ///
302    /// The timeout values are configurated arbitrarily.
303    pub(crate) fn has_stalled(&self) -> Option<PeerId> {
304        let stall_timeout = match self.best_queued_number {
305            0..300_000 => 60,        // Standard timeout, 1 minute
306            300_000..600_000 => 120, // Extended timeout, 2 minutes
307            600_000..800_000 => 180, // Extended timeout, 3 minutes
308            _ => 300,
309        };
310
311        let stalled = self.last_progress_time.elapsed().as_secs() > stall_timeout;
312
313        if stalled {
314            self.peer_store.record_failure(self.peer_id);
315            Some(self.peer_id)
316        } else {
317            None
318        }
319    }

321    /// Evaluates the import queue status and updates the internal state.
322    ///
323    /// The queue is considered saturated if either:
324    /// - Too many blocks are pending in the import queue (exceeds block count limits)
325    /// - Memory pressure from downloaded blocks exceeds configured limits
326    pub(super) fn evaluate_queue_status(&mut self, best_number: u32) -> ImportQueueStatus {
327        // Maximum number of pending blocks in the import queue.
328        let max_queued_blocks = match best_number {
329            0..=100_000 => 8192,
330            100_001..=200_000 => 4096,
331            200_001..=300_000 => 2048,
332            300_001..=600_000 => 1024,
333            _ => 512,
334        };
335
336        let queued_blocks = self.best_queued_number.saturating_sub(best_number);
337        let exceeds_memory_limits = self.exceeds_memory_limits();
338
339        if queued_blocks > max_queued_blocks || exceeds_memory_limits {
340            self.queue_status = ImportQueueStatus::Saturated;
341
342            if self
343                .last_saturated_queue_log_time
344                .is_none_or(|last_time| last_time.elapsed() > BUSY_QUEUE_LOG_INTERVAL)
345            {
346                if exceeds_memory_limits {
347                    let memory_mb = self.downloaded_blocks_memory / (1024 * 1024);
348                    let max_memory_mb =
349                        self.memory_config.max_downloaded_blocks_memory / (1024 * 1024);
350                    tracing::debug!(
351                        best_number,
352                        memory_usage_mb = memory_mb,
353                        max_memory_mb = max_memory_mb,
354                        blocks_in_memory = self.downloaded_blocks.len(),
355                        max_blocks = self.memory_config.max_blocks_in_memory,
356                        "⏸️ Pausing download: memory pressure detected",
357                    );
358                } else {
359                    tracing::debug!(
360                        best_number,
361                        best_queued_number = self.best_queued_number,
362                        "⏸️ Pausing download: too many blocks ({queued_blocks}) in the queue",
363                    );
364                }
365                self.last_saturated_queue_log_time.replace(Instant::now());
366            }
367        } else {
368            self.queue_status = ImportQueueStatus::Ready;
369        }
370
371        self.queue_status
372    }

374    /// Prepares the next block data request, ensuring the request size aligns with the current
375    /// blockchain height to avoid overly large downloads and improve latency.
376    ///
377    /// This function selects a batch of blocks from `self.missing_blocks` based on the
378    /// `self.best_queued_number`. As the chain grows, the maximum request size decreases to
379    /// reduce the burden on peers. If the number of blocks exceeds the `max_request_size`,
380    /// the function truncates the list to the maximum allowed, storing any remaining blocks
381    /// back in `self.missing_blocks` for future requests.
382    ///
383    /// The batch size is now adaptive based on memory pressure and configuration.
384    ///
385    /// # Returns
386    ///
387    /// A `SyncAction::Request` containing a `SyncRequest::GetData` with the list of blocks to
388    /// request from the peer.
389    pub(super) fn schedule_next_download_batch(&mut self) -> SyncAction {
390        let base_max_request_size = match self.best_queued_number {
391            0..=99_999 => 1024,
392            100_000..=199_999 => 512,
393            200_000..=299_999 => 128,
394            300_000..=399_999 => 64,
395            400_000..=499_999 => 32,
396            500_000..=599_999 => 16,
397            600_000..=699_999 => 8,
398            700_000..=799_999 => 4,
399            _ => 2,
400        };
401
402        // Apply adaptive batch sizing based on memory pressure
403        let max_request_size = if self.memory_config.enable_adaptive_batch_sizing {
404            let memory_usage_ratio = self.downloaded_blocks_memory as f64
405                / self.memory_config.max_downloaded_blocks_memory as f64;
406
407            let blocks_ratio = self.downloaded_blocks.len() as f64
408                / self.memory_config.max_blocks_in_memory as f64;
409
410            let pressure_ratio = memory_usage_ratio.max(blocks_ratio);
411
412            if pressure_ratio > 0.8 {
413                // High memory pressure: reduce batch size significantly
414                (base_max_request_size / 4).max(1)
415            } else if pressure_ratio > 0.6 {
416                // Medium memory pressure: reduce batch size moderately
417                (base_max_request_size / 2).max(1)
418            } else {
419                // Low memory pressure: use base batch size
420                base_max_request_size
421            }
422        } else {
423            base_max_request_size
424        };
425
426        let mut blocks_to_download = std::mem::take(&mut self.missing_blocks);
427
428        let new_missing_blocks = if blocks_to_download.len() > max_request_size {
429            blocks_to_download.split_off(max_request_size)
430        } else {
431            vec![]
432        };
433
434        self.missing_blocks = new_missing_blocks;
435
436        // Track batch size for pipelining
437        let batch_size = blocks_to_download.len();
438        self.last_batch_size = batch_size;
439
440        // Extend requested_blocks instead of replacing (for pipelining support)
441        self.requested_blocks
442            .extend(blocks_to_download.iter().copied());
443
444        tracing::debug!(
445            from = ?self.peer_id,
446            pending_blocks_to_download = self.missing_blocks.len(),
447            "📦 Downloading {batch_size} blocks",
448        );
449
450        let block_data_request = blocks_to_download
451            .into_iter()
452            .map(Inventory::Block)
453            .collect::<Vec<_>>();
454
455        SyncAction::get_data(block_data_request, self.peer_id)
456    }

458    /// Restarts the block downloader with a new peer.
459    ///
460    /// This preserves already-downloaded blocks to avoid wasting bandwidth.
461    /// Only clears pending requests that won't be fulfilled by the new peer.
462    pub(super) fn restart(&mut self, new_peer: PeerId) {
463        let preserved_blocks = self.downloaded_blocks.len();
464        let preserved_memory = self.downloaded_blocks_memory;
465
466        self.peer_id = new_peer;
467        self.downloaded_blocks_count = 0;
468        // Clear only pending requests - these won't be fulfilled by the new peer
469        self.requested_blocks.clear();
470        // Preserve downloaded_blocks - they're still valid and ready for import
471        // Preserve blocks_in_queue - they're being processed
472        // Preserve queued_blocks - they track block numbers for downloaded blocks
473        self.orphan_blocks_pool.clear();
474        self.last_progress_time = Instant::now();
475        self.queue_status = ImportQueueStatus::Ready;
476        // Don't reset downloaded_blocks_memory - we preserved the blocks
477
478        if preserved_blocks > 0 {
479            tracing::debug!(
480                preserved_blocks,
481                preserved_memory_kb = preserved_memory / 1024,
482                "Preserved downloaded blocks during peer switch"
483            );
484        }
485    }

488    pub(crate) fn handle_processed_blocks(&mut self, results: ImportManyBlocksResult) {
489        self.last_progress_time = Instant::now();
490        for (import_result, hash) in &results.results {
491            let block_number = self.blocks_in_queue.remove(hash);
492            self.queued_blocks.remove(hash);
493
494            match import_result {
495                Ok(_) => {}
496                Err(BlockImportError::UnknownParent) => panic!("Unknown parent {hash}"),
497                Err(err) => {
498                    // TODO: handle error properly
499                    if let Some(number) = block_number {
500                        panic!("Failed to import block #{number:?},{hash:?}: {err:?}");
501                    } else {
502                        panic!("Failed to import block #{hash:?}: {err:?}");
503                    }
504                }
505            }
506        }
507    }

509    /// Takes downloaded blocks and prepares them for import.
510    pub(crate) fn prepare_blocks_for_import(
511        &mut self,
512        max_block_number: Option<u32>,
513    ) -> (Vec<BlockHash>, Vec<BitcoinBlock>) {
514        let mut blocks = Vec::new();
515        let mut total_memory_freed = 0;
516
517        // Process blocks and track memory being freed
518        for block in self.downloaded_blocks.drain(..) {
519            let block_hash = block.block_hash();
520
521            let block_number = self
522                .queued_blocks
523                .block_number(block_hash)
524                .unwrap_or_else(|| {
525                    panic!("Corrupted state, number for {block_hash} not found in `queued_blocks`")
526                });
527
528            // Get cached memory usage to avoid recalculation
529            let block_memory = self
530                .queued_blocks
531                .block_memory_usage(block_hash)
532                .unwrap_or_else(|| calculate_block_memory_usage(&block));
533            total_memory_freed += block_memory;
534
535            if max_block_number.is_some_and(|target_block| block_number > target_block) {
536                // Block is filtered out - don't add to import queue
537            } else {
538                self.blocks_in_queue.insert(block_hash, block_number);
539                blocks.push((block_number, block));
540            }
541        }
542
543        // Update memory tracking - these blocks are no longer in downloaded_blocks
544        self.downloaded_blocks_memory = self
545            .downloaded_blocks_memory
546            .saturating_sub(total_memory_freed);
547
548        // Ensure the blocks sent to the import queue are ordered.
549        blocks.sort_unstable_by_key(|(number, _)| *number);
550
551        blocks
552            .into_iter()
553            .map(|(number, block)| {
554                (
555                    self.queued_blocks
556                        .block_hash(number)
557                        .expect("Corrupted state, hash for {number} not found in `queued_blocks`"),
558                    block,
559                )
560            })
561            .unzip()
562    }

564    /// Add the block that is ready to be imported.
565    pub(super) fn add_block(
566        &mut self,
567        block_number: u32,
568        block_hash: BlockHash,
569        block: BitcoinBlock,
570        from: PeerId,
571    ) {
572        let mut insert_block = |block_number, block_hash, block: BitcoinBlock| {
573            // Track memory usage
574            let block_memory = calculate_block_memory_usage(&block);
575            self.downloaded_blocks_memory += block_memory;
576
577            self.downloaded_blocks.push(block);
578            self.queued_blocks
579                .insert(block_number, block_hash, block_memory);
580            if block_number > self.best_queued_number {
581                self.best_queued_number = block_number;
582            }
583        };
584
585        insert_block(block_number, block_hash, block);
586
587        let children = self.orphan_blocks_pool.remove_blocks_for_parent(block_hash);
588
589        if !children.is_empty() {
590            tracing::trace!(blocks = ?children, "Connected orphan blocks");
591            for (index, child_block) in children.into_iter().enumerate() {
592                let hash = child_block.block_hash();
593                let number = block_number + index as u32 + 1;
594                insert_block(number, hash, child_block);
595            }
596        }
597
598        self.downloaded_blocks_count += 1;
599        self.peer_store.record_block_download(from);
600    }

619/// Calculates the approximate memory usage of a block.
620fn calculate_block_memory_usage(block: &BitcoinBlock) -> usize {
621    // Calculate size more efficiently by encoding header and transactions separately
622    let header_size = {
623        let mut size_counter = SizeCounter::new();
624        if block.header.consensus_encode(&mut size_counter).is_ok() {
625            size_counter.size()
626        } else {
627            BITCOIN_BLOCK_HEADER_SIZE
628        }
629    };
630
631    let txs_size: usize = block
632        .txdata
633        .iter()
634        .map(|tx| {
635            let mut size_counter = SizeCounter::new();
636            if tx.consensus_encode(&mut size_counter).is_ok() {
637                size_counter.size()
638            } else {
639                AVERAGE_TRANSACTION_SIZE_BYTES
640            }
641        })
642        .sum();
643
644    // Add some overhead for internal data structures
645    header_size + txs_size + BLOCK_MEMORY_OVERHEAD
646}

773    fn adaptive_batch_sizing_under_memory_pressure() {
774        let peer_id: PeerId = "127.0.0.1:8333".parse().unwrap();
775        let memory_config = create_test_memory_config();
776        let peer_store = create_test_peer_store();
777
778        // Set best_queued_number to a high value to get a smaller base batch size
779        let mut downloader =
780            BlockDownloader::new(peer_id, 800_000, peer_store, memory_config.clone());
781
782        // Set up missing blocks for download - base batch size for 800k+ blocks is 2
783        let test_blocks = subcoin_test_service::block_data();
784        let missing_blocks: Vec<BlockHash> = test_blocks
785            .iter()
786            .cycle() // Repeat the blocks to create more entries
787            .take(10) // Take 10 blocks total (more than base batch size of 2)
788            .map(|block| block.block_hash())
789            .collect();
790
791        downloader.set_missing_blocks(missing_blocks.clone());
792
793        // Simulate high memory usage by directly setting memory usage and block count
794        downloader.downloaded_blocks_memory =
795            (memory_config.max_downloaded_blocks_memory as f64 * 0.9) as usize;
796        // Also add blocks to memory to increase pressure (reach our test limit of 5)
797        for _i in 1..=4 {
798            downloader.downloaded_blocks.push(test_blocks[1].clone());
799        }
800
801        // Schedule download batch - should be reduced due to memory pressure
802        let sync_action = downloader.schedule_next_download_batch();
803
804        if let crate::sync::SyncAction::Request(crate::sync::SyncRequest::Data(inv, _)) =
805            sync_action
806        {
807            // Under high memory pressure, batch size should be 1 (base size 2 / 4 = 0.5, max(1) = 1)
808            println!(
809                "Requested {} blocks out of {} missing blocks",
810                inv.len(),
811                missing_blocks.len()
812            );
813            assert_eq!(
814                inv.len(),
815                1,
816                "Under high memory pressure, batch size should be 1"
817            );
818            assert!(
819                inv.len() < missing_blocks.len(),
820                "Batch size should be reduced under memory pressure"
821            );
822        } else {
823            panic!("Expected SyncAction::Request with Data");
824        }
825    }

950    fn queued_blocks_tracks_memory_usage() {
951        let mut queued_blocks = QueuedBlocks::default();
952        let test_blocks = subcoin_test_service::block_data();
953        let test_block = &test_blocks[1];
954        let block_hash = test_block.block_hash();
955
956        // Calculate memory usage
957        let memory_usage = calculate_block_memory_usage(test_block);
958        assert!(memory_usage > 0);
959
960        // Insert block with memory tracking
961        queued_blocks.insert(1, block_hash, memory_usage);
962
963        // Verify memory usage is tracked
964        assert_eq!(
965            queued_blocks.block_memory_usage(block_hash),
966            Some(memory_usage)
967        );
968        assert_eq!(queued_blocks.block_number(block_hash), Some(1));
969        assert_eq!(queued_blocks.block_hash(1), Some(block_hash));
970
971        // Test removal clears memory tracking
972        queued_blocks.remove(&block_hash);
973        assert_eq!(queued_blocks.block_memory_usage(block_hash), None);
974        assert_eq!(queued_blocks.block_number(block_hash), None);
975
976        // Test clear removes all memory tracking
977        queued_blocks.insert(1, block_hash, memory_usage);
978        queued_blocks.insert(2, test_blocks[2].block_hash(), memory_usage + 100);
979        queued_blocks.clear();
980        assert_eq!(queued_blocks.block_memory_usage(block_hash), None);
981        assert_eq!(
982            queued_blocks.block_memory_usage(test_blocks[2].block_hash()),
983            None
984        );
985    }

989        let peer_id: PeerId = "127.0.0.1:8333".parse().unwrap();
990        let memory_config = create_test_memory_config();
991        let peer_store = create_test_peer_store();
992
993        let mut downloader = BlockDownloader::new(peer_id, 0, peer_store, memory_config);
994
995        // Add a block
996        let test_blocks = subcoin_test_service::block_data();
997        let block = test_blocks[1].clone();
998        let block_hash = block.block_hash();
999
1000        downloader.add_block(1, block_hash, block.clone(), peer_id);
1001
1002        let initial_memory = downloader.downloaded_blocks_memory_usage();
1003        assert!(initial_memory > 0);
1004
1005        // Verify the memory is cached in queued_blocks
1006        let cached_memory = downloader.queued_blocks.block_memory_usage(block_hash);
1007        assert!(cached_memory.is_some());
1008        assert_eq!(cached_memory.unwrap(), calculate_block_memory_usage(&block));
1009
1010        // Prepare blocks for import (this should use cached memory values)
1011        let (hashes, blocks) = downloader.prepare_blocks_for_import(None);
1012
1013        assert_eq!(hashes.len(), 1);
1014        assert_eq!(blocks.len(), 1);
1015
1016        // Memory usage should be reduced
1017        let final_memory = downloader.downloaded_blocks_memory_usage();
1018        assert!(final_memory < initial_memory);
1019
1020        // Block should still be tracked in queued_blocks until fully processed
1021        assert!(
1022            downloader
1023                .queued_blocks
1024                .block_memory_usage(block_hash)
1025                .is_some()
1026        );
1027
1028        // Simulate processing completion
1029        use sc_consensus::BlockImportStatus;
1030        use sc_consensus_nakamoto::ImportManyBlocksResult;
1031        let results = ImportManyBlocksResult {
1032            imported: 1,
1033            block_count: 1,
1034            results: vec![(
1035                Ok(BlockImportStatus::ImportedUnknown(
1036                    1,
1037                    Default::default(),
1038                    None,
1039                )),
1040                block_hash,
1041            )],
1042        };
1043        downloader.handle_processed_blocks(results);
1044
1045        // Now the block should no longer be tracked in queued_blocks memory
1046        assert_eq!(
1047            downloader.queued_blocks.block_memory_usage(block_hash),
1048            None
1049        );
1050    }
