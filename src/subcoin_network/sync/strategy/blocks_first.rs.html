17// Maximum number of block inventories in `inv`, as a response to `getblocks` request.
18//
19// Assuming our best is #0, the expected block response of upcoming `getblocks`
20// request is [1, 500] when doing the initial sync.
21const MAX_GET_BLOCKS_RESPONSE: u32 = 500;
22
23/// Download state.
24#[derive(Debug, Clone)]
25enum State {
26    /// Downloading not started yet.
27    Idle,
28    /// Restarting the download process.
29    Restarting,
30    /// Peer misbehavior detected, will disconnect the peer shortly.
31    Disconnecting,
32    /// Actively downloading new block inventories in the specified range.
33    FetchingInventory(Range<u32>),
34    /// Downloading full blocks upon the completion of inventories download.
35    FetchingBlockData(Range<u32>),
36    /// All blocks up to the target block have been successfully downloaded,
37    /// the download process has been completed.
38    Completed,
39}
40
41impl State {
42    fn is_fetching_block_data(&self) -> bool {
43        matches!(self, Self::FetchingBlockData(..))
44    }
45}

47/// Responsible for sending `GetBlocks` requests to retrieve block inventories.
48#[derive(Clone)]
49struct InventoryRequester<Block, Client> {
50    client: Arc<Client>,
51    peer_id: PeerId,
52    target_block_number: u32,
53    last_locator_request_start: u32,
54    _phantom: PhantomData<Block>,
55}
56
57enum InvRequestOutcome {
58    /// Indicates a redundant request to avoid unnecessary inventory fetching.
59    RepeatedRequest,
60    /// New `GetBlocks` request with specified locator hashes and range.
61    NewGetBlocks {
62        payload: LocatorRequest,
63        range: Range<u32>,
64    },
65}
66
67impl<Block, Client> InventoryRequester<Block, Client>
68where
69    Block: BlockT,
70    Client: HeaderBackend<Block> + AuxStore,
71{
72    fn next_inventory_request(&mut self, block_downloader: &BlockDownloader) -> InvRequestOutcome {
73        let from = self
74            .client
75            .best_number()
76            .max(block_downloader.best_queued_number);
77
78        if from > 0 && self.last_locator_request_start == from {
79            return InvRequestOutcome::RepeatedRequest;
80        }
81
82        self.last_locator_request_start = from;
83
84        let BlockLocator {
85            latest_block,
86            locator_hashes,
87        } = self.client.block_locator(Some(from), |height: u32| {
88            block_downloader.queued_blocks.block_hash(height)
89        });
90
91        let end = self
92            .target_block_number
93            .min(latest_block + MAX_GET_BLOCKS_RESPONSE);
94
95        tracing::debug!(
96            latest_block,
97            best_queued_number = block_downloader.best_queued_number,
98            "Requesting new block inventories",
99        );
100
101        let payload = LocatorRequest {
102            locator_hashes,
103            stop_hash: BlockHash::all_zeros(),
104            to: self.peer_id,
105        };
106
107        InvRequestOutcome::NewGetBlocks {
108            payload,
109            range: latest_block + 1..end + 1,
110        }
111    }
112}

114/// Download blocks using the Blocks-First strategy.
115#[derive(Clone)]
116pub struct BlocksFirstStrategy<Block, Client> {
117    state: State,
118    client: Arc<Client>,
119    peer_id: PeerId,
120    /// The final block number we are targeting when the download is complete.
121    target_block_number: u32,
122    block_downloader: BlockDownloader,
123    inv_requester: InventoryRequester<Block, Client>,
124}
125
126impl<Block, Client> BlocksFirstStrategy<Block, Client>
127where
128    Block: BlockT,
129    Client: HeaderBackend<Block> + AuxStore,
130{
131    pub(crate) fn new(
132        client: Arc<Client>,
133        peer_id: PeerId,
134        target_block_number: u32,
135        peer_store: Arc<dyn PeerStore>,
136        memory_config: MemoryConfig,
137    ) -> (Self, SyncAction) {
138        let best_number = client.best_number();
139
140        let inv_requester = InventoryRequester {
141            client: client.clone(),
142            peer_id,
143            target_block_number,
144            last_locator_request_start: 0u32,
145            _phantom: Default::default(),
146        };
147
148        let mut blocks_first_sync = Self {
149            peer_id,
150            client,
151            target_block_number,
152            state: State::Idle,
153            inv_requester,
154            block_downloader: BlockDownloader::new(peer_id, best_number, peer_store, memory_config),
155        };
156
157        let outcome = blocks_first_sync
158            .inv_requester
159            .next_inventory_request(&blocks_first_sync.block_downloader);
160        let sync_action = blocks_first_sync.process_inv_request_outcome(outcome);
161
162        (blocks_first_sync, sync_action)
163    }

165    pub(crate) fn sync_status(&self) -> SyncStatus {
166        if self.block_downloader.queue_status.is_saturated()
167            || (self.state.is_fetching_block_data()
168                && self.block_downloader.missing_blocks.is_empty())
169        {
170            SyncStatus::Importing {
171                target: self.target_block_number,
172                peers: vec![self.peer_id],
173            }
174        } else {
175            SyncStatus::Downloading {
176                target: self.target_block_number,
177                peers: vec![self.peer_id],
178            }
179        }
180    }

182    pub(crate) fn can_swap_sync_peer(&self) -> Option<PeerId> {
183        (self.block_downloader.downloaded_blocks_count == 0).then_some(self.peer_id)
184    }

186    pub(crate) fn swap_sync_peer(&mut self, peer_id: PeerId, target_block_number: u32) {
187        self.peer_id = peer_id;
188        self.target_block_number = target_block_number;
189        self.block_downloader.peer_id = peer_id;

191    }

203    pub(crate) fn on_tick(&mut self) -> SyncAction {
204        if matches!(self.state, State::Restarting) {
205            return self.inventory_request_action();
206        }
207
208        if self.client.best_number() == self.target_block_number {
209            self.state = State::Completed;
210            return SyncAction::SetIdle;
211        }
212
213        if self.block_downloader.queue_status.is_saturated() {
214            // If the queue was overloaded, evalute the current queue status again.
215            let is_ready = self
216                .block_downloader
217                .evaluate_queue_status(self.client.best_number())
218                .is_ready();
219
220            if is_ready {
221                if matches!(self.state, State::FetchingBlockData(..)) {
222                    if self.block_downloader.missing_blocks.is_empty() {
223                        return self.inventory_request_action();
224                    } else if self.block_downloader.requested_blocks.is_empty() {
225                        return self.block_downloader.schedule_next_download_batch();
226                    }
227                } else {
228                    return self.inventory_request_action();
229                }
230            } else {
231                return SyncAction::None;
232            }
233        }
234
235        if let Some(stalled_peer) = self.block_downloader.has_stalled() {
236            return SyncAction::RestartSyncWithStalledPeer(stalled_peer);
237        }
238
239        if matches!(self.state, State::FetchingBlockData(..)) {
240            // If the queue was not overloaded, but we are in the downloading blocks mode,
241            // see whether there are pending blocks in the block_downloader.
242            let is_ready = self
243                .block_downloader
244                .evaluate_queue_status(self.client.best_number())
245                .is_ready();
246
247            if is_ready && self.block_downloader.requested_blocks.is_empty() {
248                if self.block_downloader.missing_blocks.is_empty() {
249                    return self.inventory_request_action();
250                } else {
251                    return self.block_downloader.schedule_next_download_batch();
252                }
253            }
254        }
255
256        SyncAction::None
257    }

259    pub(crate) fn restart(&mut self, new_peer: PeerId, target_block_number: u32) {
260        self.state = State::Restarting;
261        self.peer_id = new_peer;
262        self.target_block_number = target_block_number;
263        self.block_downloader.restart(new_peer);
264        self.inv_requester.peer_id = new_peer;
265        self.inv_requester.last_locator_request_start = 0u32;
266    }

268    // Handle `inv` message.
269    //
270    // NOTE: `inv` can be received unsolicited as an announcement of a new block,
271    // or in reply to `getblocks`.
272    pub(crate) fn on_inv(&mut self, inventories: Vec<Inventory>, from: PeerId) -> SyncAction {
273        if from != self.peer_id {
274            // TODO: handle transaction inventories
275            // tracing::debug!(?from, current_sync_peer = ?self.peer_id, "Recv unexpected {} inventories", inventories.len());
276            // tracing::debug!(?from, "TODO: inventories: {inventories:?}");
277            return SyncAction::None;
278        }
279
280        if inventories
281            .iter()
282            .filter(|inv| matches!(inv, Inventory::Block(_)))
283            .count()
284            > MAX_GET_BLOCKS_RESPONSE as usize
285        {
286            tracing::warn!(
287                ?from,
288                "Received inv with more than {MAX_GET_BLOCKS_RESPONSE} block entries"
289            );
290            self.state = State::Disconnecting;
291            return SyncAction::DisconnectPeer(self.peer_id, Error::InvHasTooManyBlockItems);
292        }
293
294        let range = match &self.state {
295            State::FetchingInventory(range) => range,
296            state => {
297                tracing::debug!(?state, "Ignored inventories {inventories:?}");
298                return SyncAction::None;
299            }
300        };
301
302        let mut missing_blocks = Vec::new();
303
304        for inv in inventories {
305            match inv {
306                Inventory::Block(block_hash) => {
307                    if !self.client.block_exists(block_hash)
308                        && self.block_downloader.is_unknown_block(block_hash)
309                    {
310                        missing_blocks.push(block_hash);
311                    }
312                }
313                Inventory::WitnessBlock(_block_hash) => {}
314                Inventory::Transaction(_txid) => {}
315                Inventory::WitnessTransaction(_txid) => {}
316                _ => {}
317            }
318        }
319
320        if missing_blocks.is_empty() {
321            return self.inventory_request_action();
322        }
323
324        self.state = State::FetchingBlockData(range.clone());
325        self.block_downloader.set_missing_blocks(missing_blocks);
326        self.block_downloader.schedule_next_download_batch()
327    }

329    pub(crate) fn on_block(&mut self, block: BitcoinBlock, from: PeerId) -> SyncAction {
330        if from != self.peer_id {
331            tracing::debug!(?from, current_sync_peer = ?self.peer_id, "Recv unexpected block #{}", block.block_hash());
332            return SyncAction::None;
333        }
334
335        let last_get_blocks_target = match &self.state {
336            State::FetchingBlockData(range) => range.end - 1,
337            state => {
338                if let Ok(height) = block.bip34_block_height() {
339                    tracing::debug!(
340                        ?state,
341                        current_sync_peer = ?self.peer_id,
342                        "Recv unexpected block #{height},{} from {from:?}",
343                        block.block_hash()
344                    );
345                } else {
346                    tracing::debug!(
347                        ?state,
348                        current_sync_peer = ?self.peer_id,
349                        "Recv unexpected block {} from {from:?}",
350                        block.block_hash()
351                    );
352                }
353
354                return SyncAction::None;
355            }
356        };
357
358        let block_hash = block.block_hash();
359
360        if self.block_downloader.block_exists(block_hash) {
361            tracing::debug!(?block_hash, "Received duplicate block");
362            return SyncAction::None;
363        }
364
365        let parent_block_hash = block.header.prev_blockhash;
366
367        let maybe_parent = self
368            .block_downloader
369            .block_number(parent_block_hash)
370            .or_else(|| self.client.block_number(parent_block_hash));
371
372        let receive_requested_block = self.block_downloader.on_block_response(block_hash);
373
374        if let Some(parent_block_number) = maybe_parent {
375            let block_number = parent_block_number + 1;
376
377            tracing::trace!("Add pending block #{block_number},{block_hash}");
378
379            self.block_downloader
380                .add_block(block_number, block_hash, block, from);
381
382            self.schedule_block_download(block_number, block_hash, last_get_blocks_target)
383        } else {
384            if self
385                .block_downloader
386                .orphan_blocks_pool
387                .contains_unknown_block(&block_hash)
388            {
389                tracing::debug!("Received orphan block {block_hash} again");
390                return SyncAction::None;
391            }
392
393            // Orphan blocks
394            if receive_requested_block {
395                self.block_downloader.add_orphan_block(block_hash, block);
396                SyncAction::None
397            } else {
398                tracing::debug!(
399                    best_number = ?self.client.info().best_number,
400                    best_queued_number = self.block_downloader.best_queued_number,
401                    parent_block_hash = ?block.header.prev_blockhash,
402                    "Received unrequested orphan block {block_hash}"
403                );
404
405                self.block_downloader.add_unknown_block(block_hash, block);
406
407                if self
408                    .block_downloader
409                    .evaluate_queue_status(self.client.best_number())
410                    .is_saturated()
411                {
412                    SyncAction::None
413                } else {
414                    self.inventory_request_action()
415                }
416            }
417        }
418    }

420    /// Request more blocks if there are remaining blocks in the current block list, otherwise
421    /// request a new list of missing blocks for downloading.
422    fn schedule_block_download(
423        &mut self,
424        block_number: u32,
425        block_hash: BlockHash,
426        last_get_blocks_target: u32,
427    ) -> SyncAction {
428        match block_number.cmp(&last_get_blocks_target) {
429            CmpOrdering::Less => {
430                // The last `getblocks` request is not yet finished, waiting for more blocks to come.
431                SyncAction::None
432            }
433            CmpOrdering::Greater => {
434                if block_number >= self.target_block_number {
435                    // Peer may send us higher blocks than our previous known peer_best when the chain grows.
436                    tracing::debug!(
437                        target_block_number = self.target_block_number,
438                        "Received block #{block_number},{block_hash} higher than the target #{}",
439                        self.target_block_number
440                    );
441                    self.state = State::Completed;
442                    SyncAction::SetIdle
443                } else {
444                    SyncAction::None
445                }
446            }
447            CmpOrdering::Equal => {
448                // No more new blocks request as there are enough ongoing blocks in the queue.
449                if self
450                    .block_downloader
451                    .evaluate_queue_status(self.client.best_number())
452                    .is_saturated()
453                {
454                    return SyncAction::None;
455                }
456
457                if self.block_downloader.missing_blocks.is_empty() {
458                    self.inventory_request_action()
459                } else {
460                    self.block_downloader.schedule_next_download_batch()
461                }
462            }
463        }
464    }

466    #[inline]
467    fn inventory_request_action(&mut self) -> SyncAction {
468        let inv_request_outcome = self
469            .inv_requester
470            .next_inventory_request(&self.block_downloader);
471        self.process_inv_request_outcome(inv_request_outcome)
472    }
473
474    fn process_inv_request_outcome(
475        &mut self,
476        inv_request_outcome: InvRequestOutcome,
477    ) -> SyncAction {
478        match inv_request_outcome {
479            InvRequestOutcome::RepeatedRequest => SyncAction::None,
480            InvRequestOutcome::NewGetBlocks { payload, range } => {
481                self.state = State::FetchingInventory(range);
482                SyncAction::get_inventory(payload)
483            }
484        }
485    }
