22/// Represents the range of blocks to be downloaded during the headers-first sync.
23///
24/// # Note
25///
26/// The `getdata` request for blocks must be ordered before sending them to the peer,
27/// this does not guarantee the block responses from peers are ordered, but it's helpful
28/// for the peers with good connections (i.e., when syncing from a local node).
29#[derive(Debug, Clone)]
30enum BlockDownload {
31    /// Request all blocks within the specified range in a single `getdata` message.
32    ///
33    /// This option is only used when the sync node is a local node.
34    AllBlocks {
35        start: IndexedBlock,
36        end: IndexedBlock,
37    },
38    /// Request blocks in smaller, manageable batches.
39    Batches {
40        /// Is the blocks download paused?
41        paused: bool,
42    },
43}
44
45/// Represents the current state of the download process.
46#[derive(Debug, Clone)]
47enum State {
48    /// Downloading not started yet.
49    Idle,
50    /// Restarting the download by requesting headers.
51    RestartingHeaders,
52    /// Restarting the download by continuing with block download from the specified range.
53    RestartingBlocks {
54        start: IndexedBlock,
55        end: IndexedBlock,
56    },
57    /// Peer misbehavior detected, will disconnect the peer shortly.
58    Disconnecting,
59    /// Actively downloading new headers in the specified range (start, end].
60    ///
61    /// Block at the height `start` already exists in our system, `start` being
62    /// exclusive is to quickly verify the parent block of first header in the response.
63    FetchingHeaders {
64        start: IndexedBlock,
65        end: IndexedBlock,
66    },
67    /// Actively downloading blocks corresponding to previously downloaded headers (start, end].
68    FetchingBlockData(BlockDownload),
69    /// All blocks up to the target block have been successfully
70    /// downloaded, the download process has been completed.
71    Completed,
72}
73
74impl Display for State {
75    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
76        match self {
77            Self::Idle => write!(f, "Idle"),
78            Self::RestartingHeaders => write!(f, "RestartingHeaders"),
79            Self::RestartingBlocks { start, end } => {
80                write!(f, "RestartingBlocks {{ start: {start}, end: {end} }}")
81            }
82            Self::Disconnecting => write!(f, "Disconnecting"),
83            Self::FetchingHeaders { start, end } => {
84                write!(f, "FetchingHeaders {{ start: {start}, end: {end} }}")
85            }
86            Self::FetchingBlockData(_range) => write!(f, "FetchingBlockData"),
87            Self::Completed => write!(f, "Completed"),
88        }
89    }
90}

92enum HeaderRequestOutcome {
93    /// No checkpoints remain for requesting more headers.
94    ExhaustedCheckpoint,
95    /// A duplicate locator; skip additional requests.
96    RepeatedRequest,
97    /// Request new headers within the specified range.
98    NewGetHeaders {
99        payload: LocatorRequest,
100        range: (IndexedBlock, IndexedBlock),
101    },
102}

104struct HeaderRequester<Block, Client> {
105    client: Arc<Client>,
106    peer_id: PeerId,
107    /// Ordered map of headers, where the key is the block hash and the value is the block number.
108    ///
109    /// Keep the headers ordered so that fetching the blocks orderly later is possible.
110    headers: IndexMap<BlockHash, u32>,
111    /// Optional range of blocks indicating the completed header download up to the next checkpoint (start, end].
112    completed_range: Option<(IndexedBlock, IndexedBlock)>,
113    last_locator_request_start: u32,
114    _phantom: PhantomData<Block>,
115}
116
117impl<Block, Client> HeaderRequester<Block, Client>
118where
119    Block: BlockT,
120    Client: HeaderBackend<Block> + AuxStore,
121{
122    /// Computes the list of missing blocks after the predefined set of headers are all downloaded.
123    fn compute_missing_blocks(&mut self, best_number: u32) -> Vec<BlockHash> {
124        assert!(
125            self.completed_range.is_some(),
126            "`compute_missing_blocks()` should be only invoked after the completion of block list download.",
127        );
128
129        let mut missing_blocks = self
130            .headers
131            .iter()
132            .filter_map(|(block_hash, block_number)| {
133                let block_hash = *block_hash;
134
135                if *block_number > best_number {
136                    return Some(block_hash);
137                }
138
139                if self.client.block_exists(block_hash) {
140                    None
141                } else {
142                    Some(block_hash)
143                }
144            })
145            .collect::<Vec<_>>();
146
147        // Sort to avoid too many orphan blocks.
148        missing_blocks.sort_by_key(|block_hash| {
149            self.headers
150                .get(block_hash)
151                .expect("Header must be available; qed ")
152        });
153
154        missing_blocks
155    }
156
157    /// Prepares a request for the next set of headers based on the specified
158    /// block number and the next checkpoint.
159    ///
160    /// Ensures no duplicate locators are sent consecutively, and stores the last
161    /// requested locator start position.
162    fn next_header_request_at(
163        &mut self,
164        block_number: u32,
165        block_hash: BlockHash,
166    ) -> HeaderRequestOutcome {
167        let Some(checkpoint) = crate::checkpoint::next_checkpoint(block_number + 1) else {
168            return HeaderRequestOutcome::ExhaustedCheckpoint;
169        };
170
171        let start = IndexedBlock {
172            number: block_number,
173            hash: block_hash,
174        };
175
176        let end = checkpoint;
177
178        // Ignore the back-to-back duplicate locators.
179        if block_number > 0 && block_number == self.last_locator_request_start {
180            return HeaderRequestOutcome::RepeatedRequest;
181        }
182
183        self.last_locator_request_start = block_number;
184
185        self.headers = IndexMap::with_capacity((end.number - start.number) as usize);
186        self.completed_range = None;
187
188        let locator_hashes = self
189            .client
190            .block_locator(Some(block_number), |_height: u32| None)
191            .locator_hashes;
192
193        let payload = LocatorRequest {
194            locator_hashes,
195            stop_hash: checkpoint.hash,
196            to: self.peer_id,
197        };
198
199        HeaderRequestOutcome::NewGetHeaders {
200            payload,
201            range: (start, end),
202        }
203    }
204}

206/// Implements the Headers-First download strategy.
207pub struct HeadersFirstStrategy<Block, Client> {
208    state: State,
209    client: Arc<Client>,
210    peer_id: PeerId,
211    header_verifier: HeaderVerifier<Block, Client>,
212    block_downloader: BlockDownloader,
213    header_requester: HeaderRequester<Block, Client>,
214    target_block_number: u32,
215}
216
217impl<Block, Client> HeadersFirstStrategy<Block, Client>
218where
219    Block: BlockT,
220    Client: HeaderBackend<Block> + AuxStore,
221{
222    pub(crate) fn new(
223        client: Arc<Client>,
224        header_verifier: HeaderVerifier<Block, Client>,
225        peer_id: PeerId,
226        target_block_number: u32,
227        peer_store: Arc<dyn PeerStore>,
228    ) -> (Self, SyncAction) {
229        let best_number = client.best_number();
230
231        let header_requester = HeaderRequester {
232            client: client.clone(),
233            peer_id,
234            last_locator_request_start: 0u32,
235            headers: IndexMap::new(),
236            completed_range: None,
237            _phantom: Default::default(),
238        };
239
240        let mut headers_first_sync = Self {
241            client,
242            header_verifier,
243            peer_id,
244            state: State::Idle,
245            block_downloader: BlockDownloader::new(peer_id, best_number, peer_store),
246            header_requester,
247            target_block_number,
248        };
249
250        let sync_action = headers_first_sync.header_request_action();
251
252        (headers_first_sync, sync_action)
253    }

255    pub(crate) fn sync_status(&self) -> SyncStatus {
256        if self.block_downloader.queue_status.is_overloaded() {
257            SyncStatus::Importing {
258                target: self.target_block_number,
259                peers: vec![self.peer_id],
260            }
261        } else {
262            SyncStatus::Downloading {
263                target: self.target_block_number,
264                peers: vec![self.peer_id],
265            }

291    pub(crate) fn on_tick(&mut self) -> SyncAction {
292        if matches!(self.state, State::RestartingHeaders) {
293            return self.header_request_action();
294        }
295
296        if let State::RestartingBlocks { start, end } = self.state {
297            return self.start_block_download_on_header_download_completion(start, end);
298        }
299
300        if self.client.best_number() == self.target_block_number {
301            self.state = State::Completed;
302            return SyncAction::SetIdle;
303        }
304
305        if self.block_downloader.queue_status.is_overloaded() {
306            let is_ready = self
307                .block_downloader
308                .evaluate_queue_status(self.client.best_number())
309                .is_ready();
310
311            if is_ready {
312                // Resume the block download, otherwise schedule the next header request.
313                match &mut self.state {
314                    State::FetchingBlockData(BlockDownload::Batches { paused }) if *paused => {
315                        *paused = false;
316
317                        if !self.block_downloader.missing_blocks.is_empty() {
318                            tracing::debug!(
319                                missing_blocks = self.block_downloader.missing_blocks.len(),
320                                "Resumed downloading blocks",
321                            );
322                            return self.block_downloader.schedule_next_download_batch();
323                        } else {
324                            return self.header_request_action();
325                        }
326                    }
327                    _ => return self.header_request_action(),
328                }
329            } else {
330                return SyncAction::None;
331            }
332        }
333
334        if let Some(stalled_peer) = self.block_downloader.has_stalled() {
335            return SyncAction::RestartSyncWithStalledPeer(stalled_peer);
336        }
337
338        SyncAction::None
339    }

341    pub(crate) fn restart(&mut self, new_peer: PeerId, target_block_number: u32) {
342        if let Some((start, end)) = self.header_requester.completed_range {
343            self.state = State::RestartingBlocks { start, end };
344        } else {
345            self.header_requester.headers.clear();
346            self.state = State::RestartingHeaders;
347        }
348        self.peer_id = new_peer;
349        self.target_block_number = target_block_number;
350        self.block_downloader.restart(new_peer);
351        self.header_requester.peer_id = new_peer;
352        self.header_requester.last_locator_request_start = 0u32;
353    }

355    fn header_request_action(&mut self) -> SyncAction {
356        let best_number = self.client.best_number();
357        let best_hash = self
358            .client
359            .block_hash(best_number)
360            .expect("Best hash must exist; qed");
361        let header_request_outcome = self
362            .header_requester
363            .next_header_request_at(best_number, best_hash);
364        self.process_header_request_outcome(header_request_outcome)
365    }
366
367    fn process_header_request_outcome(
368        &mut self,
369        header_request_outcome: HeaderRequestOutcome,
370    ) -> SyncAction {
371        match header_request_outcome {
372            HeaderRequestOutcome::RepeatedRequest => SyncAction::None,
373            HeaderRequestOutcome::ExhaustedCheckpoint => {
374                tracing::debug!("No more checkpoints, switching to blocks-first sync");
375                self.state = State::Completed;
376                SyncAction::SwitchToBlocksFirstSync
377            }
378            HeaderRequestOutcome::NewGetHeaders {
379                payload,
380                range: (start, end),
381            } => {
382                tracing::debug!("Downloading headers ({start}, {end}]");
383                self.state = State::FetchingHeaders { start, end };
384                SyncAction::get_headers(payload)
385            }
386        }
387    }

389    // Handle `headers` message.
390    //
391    // `headers` are expected to contain at most 2000 entries, in ascending order. [b1, b2, b3, ..., b2000].
392    pub(crate) fn on_headers(&mut self, headers: Vec<BitcoinHeader>, from: PeerId) -> SyncAction {
393        if headers.len() > MAX_HEADERS_SIZE {
394            return SyncAction::DisconnectPeer(from, Error::TooManyHeaders);
395        }
396
397        let Some(first_header) = headers.first() else {
398            // TODO: https://github.com/bitcoin/bitcoin/blob/6f9db1ebcab4064065ccd787161bf2b87e03cc1f/src/net_processing.cpp#L3014
399            tracing::debug!("Received empty response of getheaders");
400            return SyncAction::None;
401        };
402
403        let (start, end) = match &self.state {
404            State::FetchingHeaders { start, end } => (*start, *end),
405            state => {
406                tracing::debug!(%state, "Ignoring headers unexpected");
407                return SyncAction::None;
408            }
409        };
410
411        let mut prev_hash = first_header.prev_blockhash;
412
413        let mut prev_number = if prev_hash == start.hash {
414            start.number
415        } else if let Some(block_number) = self.header_requester.headers.get(&prev_hash).copied() {
416            block_number
417        } else if let Some(block_number) = self.client.block_number(prev_hash) {
418            block_number
419        } else {
420            tracing::info!(
421                ?first_header,
422                best_number = ?self.client.info().best_number,
423                "Cannot find the parent of the first header in headers, disconnecting"
424            );
425            self.state = State::Disconnecting;
426            return SyncAction::DisconnectPeer(self.peer_id, Error::MissingFirstHeaderParent);
427        };
428
429        for header in headers {
430            if header.prev_blockhash != prev_hash {
431                self.state = State::Disconnecting;
432                return SyncAction::DisconnectPeer(self.peer_id, Error::HeadersNotInAscendingOrder);
433            }
434
435            if !self.header_verifier.has_valid_proof_of_work(&header) {
436                self.state = State::Disconnecting;
437                return SyncAction::DisconnectPeer(
438                    from,
439                    Error::BadProofOfWork(header.block_hash()),
440                );
441            }
442
443            let block_hash = header.block_hash();
444            let block_number = prev_number + 1;
445
446            // We can't convert the Bitcoin header to a Substrate header right now as creating a
447            // Substrate header requires the full block data that is still missing.
448            self.header_requester
449                .headers
450                .insert(block_hash, block_number);
451
452            prev_hash = block_hash;
453            prev_number = block_number;
454        }
455
456        let final_block_number = prev_number;
457        let target_block_number = end.number;
458        let target_block_hash = end.hash;
459
460        if final_block_number == target_block_number {
461            self.header_requester.completed_range.replace((start, end));
462            self.start_block_download_on_header_download_completion(start, end)
463        } else {
464            tracing::debug!("ðŸ“„ Downloaded headers ({final_block_number}/{target_block_number})");
465
466            SyncAction::get_headers(LocatorRequest {
467                locator_hashes: vec![prev_hash],
468                stop_hash: target_block_hash,
469                to: self.peer_id,
470            })
471        }
472    }

474    // Fetch the block data of headers we have just downloaded.
475    fn start_block_download_on_header_download_completion(
476        &mut self,
477        start: IndexedBlock,
478        end: IndexedBlock,
479    ) -> SyncAction {
480        // TODO: sync blocks from multiple peers in parallel.
481
482        let best_number = self.client.best_number();
483
484        let missing_blocks = self.header_requester.compute_missing_blocks(best_number);
485
486        if missing_blocks.is_empty() {
487            tracing::debug!("No missing blocks, starting new headers request");
488            return self.header_request_action();
489        }
490
491        // If the sync peer is running from local, the bandwidth is not a bottleneck,
492        // simply request all blocks at once.
493        let download_all_blocks_in_one_request = is_local_address(&self.peer_id);
494
495        if download_all_blocks_in_one_request {
496            tracing::debug!(
497                best_number,
498                best_queued_number = self.block_downloader.best_queued_number,
499                downloaded_headers_count = self.header_requester.headers.len(),
500                "Headers downloaded, starting to download {} blocks from {start} to {end}",
501                missing_blocks.len()
502            );
503
504            self.state = State::FetchingBlockData(BlockDownload::AllBlocks { start, end });
505
506            let inv = missing_blocks
507                .into_iter()
508                .map(Inventory::Block)
509                .collect::<Vec<_>>();
510
511            SyncAction::get_data(inv, self.peer_id)
512        } else {
513            self.state = State::FetchingBlockData(BlockDownload::Batches { paused: false });
514            self.block_downloader.set_missing_blocks(missing_blocks);
515            self.block_downloader.schedule_next_download_batch()
516        }
517    }

519    pub(crate) fn on_block(&mut self, block: BitcoinBlock, from: PeerId) -> SyncAction {
520        let block_hash = block.block_hash();
521
522        match &self.state {
523            State::FetchingBlockData(_block_download) => {}
524            state => {
525                // TODO: we may receive the blocks from a peer that has been considered as stalled,
526                // should we try to cache and use such blocks since the bandwidth has been consumed
527                // already?
528                tracing::warn!(
529                    ?state,
530                    current_sync_peer = ?self.peer_id,
531                    "Not in the block download mode, dropping block {block_hash} from {from:?}",
532                );
533                return SyncAction::None;
534            }
535        };
536
537        let receive_requested_block = self.block_downloader.on_block_response(block_hash);
538
539        let parent_block_hash = block.header.prev_blockhash;
540
541        let maybe_parent = self
542            .block_downloader
543            .block_number(parent_block_hash)
544            .or_else(|| self.client.block_number(parent_block_hash));
545
546        if let Some(parent_block_number) = maybe_parent {
547            let block_number = parent_block_number + 1;
548
549            tracing::trace!("Add pending block #{block_number},{block_hash}");
550
551            self.block_downloader
552                .add_block(block_number, block_hash, block, from);
553
554            self.schedule_block_download(block_number, block_hash)
555        } else {
556            if receive_requested_block {
557                self.block_downloader.add_orphan_block(block_hash, block);
558            } else {
559                tracing::debug!("Discard unrequested orphan block {block_hash}");
560            }
561
562            SyncAction::None
563        }
564    }
565
566    fn schedule_block_download(&mut self, block_number: u32, block_hash: BlockHash) -> SyncAction {
567        let block_download = match &mut self.state {
568            State::FetchingBlockData(block_download) => block_download,
569            _state => unreachable!("Must be FetchingBlockData as checked; qed"),
570        };
571
572        let should_request_more_headers = match block_download {
573            BlockDownload::AllBlocks { start, end } => {
574                if end.hash == block_hash {
575                    tracing::debug!("Downloaded blocks in ({start}, {end}]");
576                    true
577                } else {
578                    false
579                }
580            }
581            BlockDownload::Batches { paused } => {
582                if self.block_downloader.requested_blocks.is_empty() {
583                    if !self.block_downloader.missing_blocks.is_empty() {
584                        if self
585                            .block_downloader
586                            .evaluate_queue_status(self.client.best_number())
587                            .is_overloaded()
588                        {
589                            *paused = true;
590                            return SyncAction::None;
591                        }
592
593                        // Request next batch of blocks.
594                        return self.block_downloader.schedule_next_download_batch();
595                    }
596
597                    tracing::debug!("Downloaded checkpoint block #{block_number},{block_hash}");
598
599                    true
600                } else {
601                    false
602                }
603            }
604        };
605
606        if should_request_more_headers {
607            if self
608                .block_downloader
609                .evaluate_queue_status(self.client.best_number())
610                .is_overloaded()
611            {
612                return SyncAction::None;
613            }
614
615            let header_request_outcome = self
616                .header_requester
617                .next_header_request_at(block_number, block_hash);
618            self.process_header_request_outcome(header_request_outcome)
619        } else {
620            SyncAction::None
621        }
622    }

625fn is_local_address(addr: &PeerId) -> bool {
626    match addr.ip() {
627        IpAddr::V4(ipv4) => ipv4.is_loopback() || ipv4.is_private() || ipv4.is_unspecified(),
628        IpAddr::V6(_ipv6) => false,
629    }
