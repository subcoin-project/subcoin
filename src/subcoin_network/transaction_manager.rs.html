1use crate::PeerId;
2use crate::network_api::IncomingTransaction;
3use bitcoin::{Transaction, Txid};
4use indexmap::IndexMap;
5use indexmap::map::Entry;
6use std::collections::HashSet;
7use std::time::{Duration, SystemTime};
8
9const TRANSACTION_TIMEOUT_DURATION_SECS: u64 = 10 * 60;
10
11#[derive(Debug)]
12struct TransactionInfo {
13    /// The actual transaction to be sent to the network.
14    transaction: Transaction,
15    /// Set of peers to which we advertised this transaction.
16    ///
17    /// Note that having a peer in this set doesn't guarantee the the peer actually
18    /// received the transaction.
19    advertised: HashSet<PeerId>,
20    /// How long the transaction should be stored.
21    ttl: SystemTime,
22}
23
24impl TransactionInfo {
25    fn new(transaction: Transaction) -> Self {
26        Self {
27            transaction,
28            advertised: HashSet::new(),
29            ttl: SystemTime::now() + Duration::from_secs(TRANSACTION_TIMEOUT_DURATION_SECS),
30        }
31    }
32}
33
34/// This struct manages the transactions received from the network.
35#[derive(Debug)]
36pub(crate) struct TransactionManager {
37    /// List of transactions tracked by this manager, in the FIFO order.
38    transactions: IndexMap<Txid, TransactionInfo>,
39}
40
41impl TransactionManager {
42    /// Maximum number of transactions the manager holds.
43    const MAX_TRANSACTIONS: usize = 256;
44
45    pub fn new() -> Self {
46        Self {
47            transactions: IndexMap::new(),
48        }
49    }
50
51    /// Broadcast known transaction IDs to the connected peers.
52    pub fn on_tick<'a>(
53        &mut self,
54        connected_peers: impl Iterator<Item = &'a PeerId>,
55    ) -> Vec<(PeerId, Vec<Txid>)> {
56        // Remove timeout transactions.
57        let now = SystemTime::now();
58        self.transactions.retain(|txid, info| {
59            if info.ttl < now {
60                tracing::debug!("Removing timeout transaction {txid}");
61                false
62            } else {
63                true
64            }
65        });
66
67        connected_peers
68            .filter_map(|address| {
69                let mut to_advertise = vec![];
70
71                for (txid, info) in self.transactions.iter_mut() {
72                    if !info.advertised.contains(address) {
73                        to_advertise.push(*txid);
74                        info.advertised.insert(*address);
75                    }
76                }
77
78                if to_advertise.is_empty() {
79                    None
80                } else {
81                    Some((*address, to_advertise))
82                }
83            })
84            .collect()
85    }
86
87    pub fn get_transaction(&self, txid: &Txid) -> Option<Transaction> {
88        self.transactions
89            .get(txid)
90            .map(|tx_info| tx_info.transaction.clone())
91    }
92
93    pub fn add_transaction(
94        &mut self,
95        incoming_transaction: IncomingTransaction,
96    ) -> Result<Txid, String> {
97        let IncomingTransaction { txid, transaction } = incoming_transaction;
98
99        if self.transactions.len() == Self::MAX_TRANSACTIONS {
100            self.transactions.shift_remove_index(0);
101        }
102
103        match self.transactions.entry(txid) {
104            Entry::Occupied(_) => Err(format!("Already have transaction {txid}")),
105            Entry::Vacant(entry) => {
106                entry.insert(TransactionInfo::new(transaction));
107                Ok(txid)
108            }
109        }
110    }
111}
