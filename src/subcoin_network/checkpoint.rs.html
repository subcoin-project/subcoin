169pub(crate) fn next_checkpoint(block_number: u32) -> Option<IndexedBlock> {
170    match CHECKPOINTS.binary_search_by_key(&block_number, |checkpoint| checkpoint.number) {
171        Ok(_) => None,
172        Err(index) => CHECKPOINTS.get(index).cloned(),
173    }
174}
175
176pub(crate) fn last_checkpoint_height() -> u32 {
177    CHECKPOINTS
178        .last()
179        .expect("Checkpoints are not empty; qed")
180        .number
181}
