11/// Timeout for transaction requests (if we don't receive after requesting).
12const TX_REQUEST_TIMEOUT: Duration = Duration::from_secs(60);
13
14/// Maximum number of transactions to request from a single peer at once.
15const MAX_TX_REQUEST_BATCH: usize = 100;
16
17/// Trait for querying peer transaction relay capabilities.
18///
19/// This trait abstracts the peer manager operations needed for tx relay,
20/// enabling testing with mock implementations.
21pub trait PeerTxRelayInfo {
22    /// Check if peer wants to relay transactions.
23    fn peer_wants_tx_relay(&self, peer: PeerId) -> bool;
24    /// Get the minimum fee rate (sat/kvB) this peer wants to receive (BIP133).
25    fn get_fee_filter(&self, peer: PeerId) -> Option<u64>;
26}
27
28/// Information about a requested transaction.
29#[derive(Debug, Clone)]
30struct RequestInfo {
31    /// Peer we requested this transaction from.
32    from_peer: PeerId,
33    /// When we made the request.
34    requested_at: Instant,
35}
36
37/// Actions to be taken by the network processor in response to transaction relay events.
38#[derive(Debug)]
39pub enum TxAction {
40    /// No action needed.
41    None,
42    /// Send GetData message to request transactions.
43    SendGetData(Vec<Inventory>, PeerId),
44    /// Announce transaction IDs to peers.
45    /// Vec of (peer_id, txids_to_announce).
46    AnnounceToPeers(Vec<(PeerId, Vec<Txid>)>),
47    /// Send full transactions to peers.
48    /// Vec of (txid, peer_id).
49    ServeTxs(Vec<(Txid, PeerId)>),
50    /// Penalize a peer for sending an invalid transaction.
51    PenalizePeer(PeerId),
52    /// Disconnect a peer due to protocol violation.
53    #[allow(dead_code)]
54    DisconnectPeer(PeerId, Error),
55}

57/// Transaction relay coordinator.
58///
59/// This component orchestrates transaction relay and owns the transaction pool.
60/// It provides a single point of access for all transaction operations, avoiding
61/// state duplication with PeerManager or MemPool.
62///
63/// Design principles:
64/// - Owns the tx_pool for encapsulation
65/// - Queries PeerManager for peer capabilities (fee filters, relay permissions)
66/// - Tracks only relay-specific state (announcements, pending requests)
67/// - Returns TxAction for NetworkProcessor to execute
68pub(crate) struct TxRelay<Block, Client, Pool> {
69    /// The transaction pool for validation and storage.
70    tx_pool: Arc<Pool>,
71
72    /// Track which transactions we've announced to each peer.
73    peer_announcements: HashMap<PeerId, HashSet<Txid>>,
74
75    /// Track transactions we've requested but not received yet.
76    requested_txs: HashMap<Txid, RequestInfo>,
77
78    _phantom: PhantomData<(Block, Client)>,
79}
80
81// Methods that don't require Client trait bounds
82impl<Block, Client, Pool> TxRelay<Block, Client, Pool>
83where
84    Block: BlockT,
85    Pool: TxPool,
86{
87    /// Creates a new transaction relay coordinator with the given pool.
88    pub fn new(tx_pool: Arc<Pool>) -> Self {
89        Self {
90            tx_pool,
91            peer_announcements: HashMap::new(),
92            requested_txs: HashMap::new(),
93            _phantom: PhantomData,
94        }
95    }
96
97    /// Handles incoming inventory messages containing transaction announcements.
98    ///
99    /// Decides which transactions to request based on:
100    /// - Whether we already have the transaction in mempool
101    /// - Whether we've already requested it
102    ///
103    /// **Note**: Caller should verify peer has relay permission before calling.
104    pub fn on_inv(&mut self, inv: Vec<Inventory>, from: PeerId) -> TxAction {
105        let mut to_request = Vec::new();
106
107        for item in inv {
108            if let Inventory::Transaction(txid) = item {
109                // Skip if we already have it in mempool
110                if self.tx_pool.contains(&txid) {
111                    continue;
112                }
113
114                // Skip if we've already requested it
115                if self.requested_txs.contains_key(&txid) {
116                    continue;
117                }
118
119                // Skip if we've recently seen it
120                // TODO: Add bloom filter or cache for recently seen txids
121
122                to_request.push(Inventory::Transaction(txid));
123
124                // Track the request
125                self.requested_txs.insert(
126                    txid,
127                    RequestInfo {
128                        from_peer: from,
129                        requested_at: Instant::now(),
130                    },
131                );
132
133                // Limit batch size
134                if to_request.len() >= MAX_TX_REQUEST_BATCH {
135                    break;
136                }
137            }
138        }
139
140        if to_request.is_empty() {
141            TxAction::None
142        } else {
143            tracing::debug!("Requesting {} transactions from {from}", to_request.len());
144            TxAction::SendGetData(to_request, from)
145        }
146    }

148    /// Handles periodic broadcast of pending transactions.
149    ///
150    /// Queries mempool for transactions pending broadcast and decides which peers
151    /// to announce to based on fee filters and relay permissions.
152    pub fn on_tick<P>(&mut self, peer_info: &P, connected_peers: &[PeerId]) -> TxAction
153    where
154        P: PeerTxRelayInfo,
155    {
156        self.cleanup_timeouts();
157
158        // Query pool for transactions pending broadcast
159        let pending_broadcast = self.tx_pool.pending_broadcast();
160
161        if pending_broadcast.is_empty() {
162            return TxAction::None;
163        }
164
165        // Build per-peer batches: peer -> [txids to announce]
166        let mut peer_txids: HashMap<PeerId, Vec<Txid>> = HashMap::new();
167
168        for (txid, fee_rate) in pending_broadcast {
169            for &peer_id in connected_peers {
170                // Skip if peer doesn't want tx relay
171                if !peer_info.peer_wants_tx_relay(peer_id) {
172                    continue;
173                }
174
175                // Skip if we've already announced to this peer
176                if let Some(announced) = self.peer_announcements.get(&peer_id) {
177                    if announced.contains(&txid) {
178                        continue;
179                    }
180                }
181
182                // Check fee filter (BIP133)
183                if let Some(peer_min_fee) = peer_info.get_fee_filter(peer_id) {
184                    if fee_rate < peer_min_fee {
185                        tracing::trace!(
186                            "Not relaying tx {txid} to peer {peer_id}: fee rate {fee_rate} < peer's filter {peer_min_fee}"
187                        );
188                        continue;
189                    }
190                }
191
192                // Add to this peer's batch
193                peer_txids.entry(peer_id).or_default().push(txid);
194
195                // Track announcement
196                self.peer_announcements
197                    .entry(peer_id)
198                    .or_default()
199                    .insert(txid);
200            }
201        }
202
203        if peer_txids.is_empty() {
204            TxAction::None
205        } else {
206            // Return batched announcements for all peers
207            TxAction::AnnounceToPeers(peer_txids.into_iter().collect())
208        }
209    }

211    /// Handles GetData requests for transactions.
212    ///
213    /// Returns actions to serve the requested transactions if we have them.
214    /// Only returns txids for transactions that exist in our mempool.
215    pub fn on_get_data(&self, inv: Vec<Inventory>, from: PeerId) -> TxAction {
216        let mut txids = Vec::new();
217
218        for item in inv {
219            if let Inventory::Transaction(txid) = item {
220                // Only serve transactions we actually have
221                if self.tx_pool.contains(&txid) {
222                    tracing::debug!("Peer {from} requested tx {txid}");
223                    txids.push((txid, from));
224                } else {
225                    tracing::debug!("Peer {from} requested tx {txid} but we don't have it");
226                }
227            }
228        }
229
230        if txids.is_empty() {
231            TxAction::None
232        } else {
233            TxAction::ServeTxs(txids)
234        }
235    }

237    /// Called when a transaction validation completes.
238    ///
239    /// Handles both accepted and rejected transactions, returning appropriate
240    /// actions for peer penalty or cleanup.
241    pub fn on_validated_tx(
242        &mut self,
243        result: TxValidationResult,
244        received_from: PeerId,
245    ) -> TxAction {
246        match result {
247            TxValidationResult::Accepted { txid, .. } => {
248                self.requested_txs.remove(&txid);
249                // The transaction will be announced via on_tick when mempool returns
250                // it from pending_broadcast()
251                TxAction::None
252            }
253            TxValidationResult::Rejected { txid, reason } => {
254                self.requested_txs.remove(&txid);
255                // Penalize peer if this was a hard rejection (protocol violation)
256                if reason.should_penalize_peer() {
257                    TxAction::PenalizePeer(received_from)
258                } else {
259                    // Soft rejection - don't penalize
260                    TxAction::None
261                }
262            }
263        }
264    }
265
266    /// Called when a peer disconnects.
267    ///
268    /// Cleans up any per-peer state.
269    pub fn on_peer_disconnected(&mut self, peer_id: PeerId) {
270        self.peer_announcements.remove(&peer_id);
271
272        // Remove pending requests from this peer
273        self.requested_txs
274            .retain(|_, info| info.from_peer != peer_id);
275    }
276
277    /// Removes timed-out transaction requests.
278    fn cleanup_timeouts(&mut self) {
279        let now = Instant::now();
280        self.requested_txs.retain(|txid, info| {
281            let timed_out = now.duration_since(info.requested_at) > TX_REQUEST_TIMEOUT;
282            if timed_out {
283                tracing::debug!("Transaction request for {txid} timed out");
284            }
285            !timed_out
286        });
287    }
288
289    /// Validates a transaction using the internal pool.
290    ///
291    /// This is a forwarding method for NetworkProcessor's hot path.
292    pub fn validate_transaction(&self, tx: Transaction) -> TxValidationResult {
293        self.tx_pool.validate_transaction(tx)
294    }
295
296    /// Checks if a transaction is in the pool.
297    pub fn contains(&self, txid: Txid) -> bool {
298        self.tx_pool.contains(&txid)
299    }
300
301    /// Gets a transaction from the pool if present.
302    pub fn get_transaction(&self, txid: Txid) -> Option<Arc<Transaction>> {
303        self.tx_pool.get(&txid)
304    }
305
306    /// Marks transactions as broadcast to the network.
307    pub fn mark_broadcast(&self, txids: &[Txid]) {
308        self.tx_pool.mark_broadcast(txids);
309    }
