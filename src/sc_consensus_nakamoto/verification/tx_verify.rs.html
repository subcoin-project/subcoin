7// MinCoinbaseScriptLen is the minimum length a coinbase script can be.
8const MIN_COINBASE_SCRIPT_LEN: usize = 2;
9
10// MaxCoinbaseScriptLen is the maximum length a coinbase script can be.
11const MAX_COINBASE_SCRIPT_LEN: usize = 100;
12
13/// Transaction verification error.
14#[derive(Debug, thiserror::Error)]
15pub enum Error {
16    #[error("Transaction has no inputs")]
17    EmptyInput,
18    #[error("Transaction has no outputs")]
19    EmptyOutput,
20    #[error("Transaction is too large")]
21    TransactionOversize,
22    #[error("Transaction contains duplicate inputs at index {0}")]
23    DuplicateTxInput(usize),
24    #[error("Output value (0) is too large")]
25    OutputValueTooLarge(Amount),
26    #[error("Total output value (0) is too large")]
27    TotalOutputValueTooLarge(Amount),
28    #[error(
29        "Coinbase transaction script length of {0} is out of range \
30        (min: {MIN_COINBASE_SCRIPT_LEN}, max: {MAX_COINBASE_SCRIPT_LEN})"
31    )]
32    BadCoinbaseLength(usize),
33    #[error("Transaction input refers to a previous output that is null")]
34    PreviousOutputNull,
35}

37/// Checks whether the transaction is final at the given height and block time.
38pub fn is_final_tx(tx: &Transaction, height: u32, block_time: u32) -> bool {
39    if tx.lock_time == LockTime::ZERO {
40        return true;
41    }
42
43    let lock_time = if tx.lock_time.to_consensus_u32() < LOCK_TIME_THRESHOLD {
44        height
45    } else {
46        block_time
47    };
48
49    if tx.lock_time.to_consensus_u32() < lock_time {
50        return true;
51    }
52
53    // Even if tx.nLockTime isn't satisfied by nBlockHeight/nBlockTime, a
54    // transaction is still considered final if all inputs' nSequence ==
55    // SEQUENCE_FINAL (0xffffffff), in which case nLockTime is ignored.
56    //
57    // Because of this behavior OP_CHECKLOCKTIMEVERIFY/CheckLockTime() will
58    // also check that the spending input's nSequence != SEQUENCE_FINAL,
59    // ensuring that an unsatisfied nLockTime value will actually cause
60    // IsFinalTx() to return false here:
61    tx.input.iter().all(|txin| txin.sequence.is_final())
62}

64// <https://github.com/bitcoin/bitcoin/blob/6f9db1ebcab4064065ccd787161bf2b87e03cc1f/src/consensus/tx_check.cpp#L11>
65pub fn check_transaction_sanity(tx: &Transaction) -> Result<(), Error> {
66    if tx.input.is_empty() {
67        return Err(Error::EmptyInput);
68    }
69
70    if tx.output.is_empty() {
71        return Err(Error::EmptyOutput);
72    }
73
74    if Weight::from_wu((tx.base_size() * WITNESS_SCALE_FACTOR) as u64) > MAX_BLOCK_WEIGHT {
75        return Err(Error::TransactionOversize);
76    }
77
78    let mut value_out = Amount::ZERO;
79    tx.output.iter().try_for_each(|txout| {
80        if txout.value > Amount::MAX_MONEY {
81            return Err(Error::OutputValueTooLarge(txout.value));
82        }
83
84        value_out += txout.value;
85
86        if value_out > Amount::MAX_MONEY {
87            return Err(Error::TotalOutputValueTooLarge(value_out));
88        }
89
90        Ok(())
91    })?;
92
93    // Check for duplicate inputs.
94    let mut seen_inputs = HashSet::with_capacity(tx.input.len());
95    for (index, txin) in tx.input.iter().enumerate() {
96        if !seen_inputs.insert(txin.previous_output) {
97            return Err(Error::DuplicateTxInput(index));
98        }
99    }
100
101    // Coinbase script length must be between min and max length.
102    if tx.is_coinbase() {
103        let script_sig_len = tx.input[0].script_sig.len();
104
105        if !(MIN_COINBASE_SCRIPT_LEN..=MAX_COINBASE_SCRIPT_LEN).contains(&script_sig_len) {
106            return Err(Error::BadCoinbaseLength(script_sig_len));
107        }
108    } else {
109        // Previous transaction outputs referenced by the inputs to this
110        // transaction must not be null.
111        if tx.input.iter().any(|txin| txin.previous_output.is_null()) {
112            return Err(Error::PreviousOutputNull);
113        }
114    }
115
116    Ok(())
117}

119/// Counts the sigops for this transaction using legacy counting.
120pub fn get_legacy_sig_op_count(tx: &Transaction) -> usize {
121    tx.input
122        .iter()
123        .map(|txin| txin.script_sig.count_sigops_legacy())
124        .sum::<usize>()
125        + tx.output
126            .iter()
127            .map(|txout| txout.script_pubkey.count_sigops_legacy())
128            .sum::<usize>()
129}
