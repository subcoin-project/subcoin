6/// Returns the script validation flags for the specified block.
7///
8/// <https://github.com/bitcoin/bitcoin/blob/6f9db1ebcab4064065ccd787161bf2b87e03cc1f/src/validation.cpp#L2360>
9pub fn get_block_script_flags(
10    height: u32,
11    block_hash: BlockHash,
12    chain_params: &ChainParams,
13) -> c_uint {
14    if let Some(flag) = chain_params
15        .script_flag_exceptions
16        .get(&block_hash)
17        .copied()
18    {
19        return flag;
20    }
21
22    let mut flags = bitcoinconsensus::VERIFY_P2SH | bitcoinconsensus::VERIFY_WITNESS;
23
24    // Enforce the DERSIG (BIP66) rule
25    if height >= chain_params.params.bip66_height {
26        flags |= bitcoinconsensus::VERIFY_DERSIG;
27    }
28
29    // Enforce CHECKLOCKTIMEVERIFY (BIP65)
30    if height >= chain_params.params.bip65_height {
31        flags |= bitcoinconsensus::VERIFY_CHECKLOCKTIMEVERIFY;
32    }
33
34    // Enforce CHECKSEQUENCEVERIFY (BIP112)
35    if height >= chain_params.csv_height {
36        flags |= bitcoinconsensus::VERIFY_CHECKSEQUENCEVERIFY;
37    }
38
39    // Enforce BIP147 NULLDUMMY (activated simultaneously with segwit)
40    if height >= chain_params.segwit_height {
41        flags |= bitcoinconsensus::VERIFY_NULLDUMMY;
42    }
43
44    flags
45}

47/// Verifies the input for given input_index in the transaction is valid.
48pub fn verify_input_script(
49    spent_output: &TxOut,
50    spending_transaction: &[u8],
51    input_index: usize,
52    flags: u32,
53) -> Result<(), Error> {
54    let script_verify_result = bitcoinconsensus::verify_with_flags(
55        spent_output.script_pubkey.as_bytes(),
56        spent_output.value.to_sat(),
57        spending_transaction,
58        input_index,
59        flags,
60    );
61
62    match script_verify_result {
63        Ok(()) | Err(bitcoinconsensus::Error::ERR_SCRIPT) => Ok(()),
64        Err(script_error) => Err(script_error.into()),
65    }
66}
