28/// Workspace for validating a single transaction.
29///
30/// Holds intermediate state during validation to avoid recomputation.
31pub struct ValidationWorkspace {
32    /// Transaction being validated.
33    pub tx: Arc<Transaction>,
34
35    /// Base fee (sum of input values - sum of output values).
36    pub base_fee: bitcoin::Amount,
37
38    /// Modified fee (includes priority adjustments).
39    pub modified_fee: bitcoin::Amount,
40
41    /// Transaction weight.
42    pub tx_weight: Weight,
43
44    /// Virtual size in bytes.
45    pub vsize: i64,
46
47    /// Signature operation cost.
48    pub sigop_cost: i64,
49
50    /// Lock points for BIP68/BIP112.
51    pub lock_points: LockPoints,
52
53    /// Whether this transaction spends a coinbase output.
54    pub spends_coinbase: bool,
55
56    /// In-mempool ancestors (entry IDs).
57    #[allow(dead_code)]
58    pub ancestors: HashSet<EntryId>,
59
60    /// Conflicting mempool transactions (txids).
61    pub conflicts: HashSet<bitcoin::Txid>,
62
63    /// Full conflict set for RBF (if any).

92/// Pre-validation checks before script execution.
93///
94/// Corresponds to Bitcoin Core's `PreChecks()` function.
95/// Validates:
96/// - Transaction sanity (CheckTransaction)
97/// - Not coinbase
98/// - Standardness (if required)
99/// - Finality (nLockTime; BIP68 deferred)
100/// - Not already in mempool
101/// - Input availability
102/// - Fee requirements
103pub fn pre_checks<Block, Client>(
104    ws: &mut ValidationWorkspace,
105    inner: &MemPoolInner,
106    coins_cache: &mut CoinsViewCache<Block, Client>,
107    options: &MemPoolOptions,
108    current_height: u32,
109    current_mtp: i64,
110    _best_block: Block::Hash,
111) -> Result<(), MempoolError>
112where
113    Block: BlockT,
114    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + Send + Sync,
115    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
116{
117    let tx = &ws.tx;
118
119    // 1. CheckTransaction (sanity checks)
120    check_transaction_sanity(tx)?;
121
122    // 2. Reject coinbase
123    if tx.is_coinbase() {
124        return Err(MempoolError::Coinbase);
125    }
126
127    // 3. Check standardness
128    if options.require_standard {
129        let dust_relay_feerate = FeeRate::from_sat_per_kvb(options.dust_relay_feerate);
130        is_standard_tx(
131            tx,
132            options.max_datacarrier_bytes,
133            options.permit_bare_multisig,
134            dust_relay_feerate,
135        )
136        .map_err(|e| MempoolError::NotStandard(format!("{e:?}")))?;
137    }
138
139    // 4. Check if already in mempool (by wtxid)
140    if inner.contains_wtxid(&tx.compute_wtxid()) {
141        return Err(MempoolError::AlreadyInMempool);
142    }
143
144    // 5. Check if same txid exists (different witness)
145    if inner.contains_txid(&tx.compute_txid()) {
146        return Err(MempoolError::AlreadyInMempool);
147    }
148
149    // 6. Check for conflicts with mempool transactions (RBF handling)
150    let mut has_conflicts = false;
151    for input in &tx.input {
152        if let Some(conflicting_txid) = inner.get_conflict_tx(&input.previous_output) {
153            has_conflicts = true;
154            ws.conflicts.insert(conflicting_txid);
155        }
156    }
157
158    // If there are conflicts, check RBF policy
159    if has_conflicts {
160        if !options.enable_rbf {
161            return Err(MempoolError::TxConflict(
162                "conflicts with mempool transaction but RBF is disabled".to_string(),
163            ));
164        }
165
166        // Validate RBF rules (BIP125)
167        let conflict_set = check_rbf_policy(ws, inner, coins_cache, options)?;
168        ws.conflict_set = Some(conflict_set);
169    }
170
171    // 7. Batch-prefetch all input coins
172    let outpoints: Vec<_> = tx.input.iter().map(|txin| txin.previous_output).collect();
173    coins_cache.ensure_coins(&outpoints)?;
174
175    let mut prev_outputs: HashMap<_, _> = HashMap::with_capacity(outpoints.len());
176
177    // 8. Check all inputs are available
178    let mut missing = Vec::new();
179    for outpoint in &outpoints {
180        if !coins_cache.have_coin(outpoint) {
181            missing.push(outpoint.txid);
182        }
183    }
184    if !missing.is_empty() {
185        missing.sort();
186        missing.dedup();
187        return Err(MempoolError::MissingInputs { parents: missing });
188    }
189
190    // 9. Calculate fees and check for negative fee
191    let mut input_value = bitcoin::Amount::ZERO;
192    let mut spends_coinbase = false;
193
194    for outpoint in &outpoints {
195        let coin = coins_cache
196            .get_coin(outpoint)?
197            .ok_or_else(|| MempoolError::MissingInputs {
198                parents: vec![outpoint.txid],
199            })?;
200
201        input_value = input_value
202            .checked_add(coin.output.value)
203            .ok_or(MempoolError::FeeOverflow)?;
204
205        prev_outputs.insert(*outpoint, coin.output.clone());
206
207        if coin.is_coinbase {
208            spends_coinbase = true;
209
210            // Check coinbase maturity (100 blocks)
211            let coin_age = current_height.saturating_sub(coin.height);
212            if coin_age < 100 {
213                return Err(MempoolError::NonFinal);
214            }
215        }
216    }
217
218    let output_value: bitcoin::Amount = tx.output.iter().map(|txout| txout.value).sum();
219
220    if input_value < output_value {
221        return Err(MempoolError::NegativeFee);
222    }
223
224    let base_fee = input_value
225        .checked_sub(output_value)
226        .ok_or(MempoolError::FeeOverflow)?;
227
228    // 10. Check minimum relay fee
229    let min_relay_fee_rate = options.min_relay_fee_rate();
230    let min_fee = min_relay_fee_rate.get_fee(ws.vsize);
231    if base_fee < min_fee {
232        let actual_kvb = FeeRate::from_amount_and_vsize(base_fee, ws.vsize)
233            .map_err(|e| MempoolError::InvalidFeeRate(e.to_string()))?
234            .as_sat_per_kvb();
235        let min_kvb = min_relay_fee_rate.as_sat_per_kvb();
236
237        return Err(MempoolError::FeeTooLow {
238            min_kvb,
239            actual_kvb,
240        });
241    }
242
243    // 11. Check finality (nLockTime only; BIP68 sequence locks deferred)
244    let current_time_secs = std::time::SystemTime::now()
245        .duration_since(std::time::UNIX_EPOCH)
246        .expect("Time went backwards")
247        .as_secs() as u32;
248
249    if !is_final_tx(tx, current_height, current_time_secs) {
250        return Err(MempoolError::NonFinal);
251    }
252
253    // 12. Calculate sigop cost
254    let sigop_cost = ws
255        .tx
256        .total_sigop_cost(|outpoint| prev_outputs.get(outpoint).cloned());
257    let sigop_cost = i64::try_from(sigop_cost).expect("sigop cost must fit into i64");
258
259    if sigop_cost > MAX_STANDARD_TX_SIGOPS_COST {
260        return Err(MempoolError::TooManySigops(sigop_cost));
261    }
262
263    // 13. Calculate BIP68 sequence lock points
264    let lock_points = calculate_lock_points_at_tip(tx, coins_cache, current_height, current_mtp)?;
265
266    // 14. Store computed values in workspace
267    ws.base_fee = base_fee;
268    ws.modified_fee = base_fee; // TODO: Apply priority deltas
269    ws.spends_coinbase = spends_coinbase;
270    ws.sigop_cost = sigop_cost;
271    ws.lock_points = lock_points;
272
273    Ok(())
274}

276/// Check ancestor/descendant limits before adding to mempool.
277///
278/// Ensures that adding this transaction won't violate:
279/// - MAX_ANCESTORS (25)
280/// - MAX_ANCESTOR_SIZE (101 KB)
281/// - MAX_DESCENDANTS (25)
282/// - MAX_DESCENDANT_SIZE (101 KB)
283pub fn check_package_limits(
284    ws: &ValidationWorkspace,
285    inner: &MemPoolInner,
286    options: &MemPoolOptions,
287) -> Result<(), MempoolError> {
288    // Find all in-mempool parents
289    let mut ancestors = HashSet::new();
290
291    for input in &ws.tx.input {
292        if let Some(parent_id) = inner.arena.get_by_txid(&input.previous_output.txid) {
293            // This input spends a mempool transaction
294            inner.calculate_ancestors(parent_id, &mut ancestors);
295        }
296    }
297
298    // Check ancestor limits
299    let ancestor_count = ancestors.len() + 1; // +1 for this tx
300    if ancestor_count > options.max_ancestors() {
301        return Err(MempoolError::TooManyAncestors(ancestor_count));
302    }
303
304    // Calculate total ancestor size
305    let mut ancestor_size = ws.vsize;
306    for &ancestor_id in &ancestors {
307        if let Some(entry) = inner.arena.get(ancestor_id) {
308            ancestor_size += entry.vsize();
309        }
310    }
311
312    if ancestor_size > options.max_ancestor_size() {
313        return Err(MempoolError::AncestorSizeTooLarge(ancestor_size));
314    }
315
316    // Check descendant limits for each ancestor
317    for &ancestor_id in &ancestors {
318        let entry = inner
319            .arena
320            .get(ancestor_id)
321            .expect("Ancestor entry must exist");
322
323        // Adding this tx will add 1 descendant and ws.vsize to all ancestors
324        let new_desc_count = entry.count_with_descendants + 1;
325        let new_desc_size = entry.size_with_descendants + ws.vsize;
326
327        if new_desc_count > options.max_descendants() {
328            return Err(MempoolError::TooManyDescendants(new_desc_count as usize));
329        }
330
331        if new_desc_size > options.max_descendant_size() {
332            return Err(MempoolError::DescendantSizeTooLarge(new_desc_size));
333        }
334    }
335
336    Ok(())
337}

339/// Finalize transaction addition to mempool.
340///
341/// Creates TxMemPoolEntry and adds it to arena, updating ancestor/descendant state.
342/// If conflict_set is present, removes conflicting transactions first (RBF).
343#[allow(clippy::too_many_arguments)]
344pub fn finalize_tx<Block, Client>(
345    ws: ValidationWorkspace,
346    inner: &mut MemPoolInner,
347    coins_cache: &mut CoinsViewCache<Block, Client>,
348    current_height: u32,
349    current_time: i64,
350    entry_block_mtp: i64,
351    entry_block_hash: bitcoin::BlockHash,
352    sequence: u64,
353) -> Result<EntryId, MempoolError>
354where
355    Block: BlockT,
356    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + Send + Sync,
357    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
358{
359    // Handle RBF: Remove conflicting transactions if present
360    if let Some(conflict_set) = &ws.conflict_set {
361        // Remove from mempool (with update_descendants = true for RBF)
362        inner.remove_staged(
363            &conflict_set.all_conflicts,
364            true, // update_descendants = true for RBF
365            crate::types::RemovalReason::Replaced,
366        );
367
368        // Clean up coins cache overlay for removed transactions
369        for removed_tx in &conflict_set.removed_transactions {
370            coins_cache.remove_mempool_tx(removed_tx);
371        }
372    }
373
374    // Find in-mempool parents
375    let mut parents = HashSet::new();
376    let mut ancestors = HashSet::new();
377
378    for input in &ws.tx.input {
379        if let Some(parent_id) = inner.arena.get_by_txid(&input.previous_output.txid) {
380            parents.insert(parent_id);
381            inner.calculate_ancestors(parent_id, &mut ancestors);
382        }
383    }
384
385    // Calculate initial ancestor state (includes this tx)
386    let mut count_with_ancestors = 1;
387    let mut size_with_ancestors = ws.vsize;
388    let mut fees_with_ancestors = ws.modified_fee;
389    let mut sigops_with_ancestors = ws.sigop_cost;
390
391    for &ancestor_id in &ancestors {
392        let entry = inner.arena.get(ancestor_id).expect("Ancestor must exist");
393        count_with_ancestors += 1;
394        size_with_ancestors += entry.vsize();
395        fees_with_ancestors = fees_with_ancestors
396            .checked_add(entry.modified_fee)
397            .ok_or(MempoolError::FeeOverflow)?;
398        sigops_with_ancestors += entry.sigop_cost;
399    }
400
401    // Create entry with initial state
402    let entry = TxMemPoolEntry {
403        tx: ws.tx.clone(),
404        fee: ws.base_fee,
405        modified_fee: ws.modified_fee,
406        tx_weight: ws.tx_weight,
407        time: current_time,
408        entry_height: current_height,
409        entry_block_mtp,
410        entry_block_hash,
411        entry_sequence: sequence,
412        spends_coinbase: ws.spends_coinbase,
413        sigop_cost: ws.sigop_cost,
414        lock_points: ws.lock_points,
415        count_with_ancestors,
416        size_with_ancestors,
417        fees_with_ancestors,
418        sigops_with_ancestors,
419        count_with_descendants: 1,
420        size_with_descendants: ws.vsize,
421        fees_with_descendants: ws.modified_fee,
422        parents,
423        children: HashSet::new(),
424        cached_ancestor_key: crate::arena::AncestorScoreKey {
425            neg_feerate_frac: (0, 1),
426            txid: ws.tx.compute_txid(),
427        }, // Will be recomputed by arena
428        cached_descendant_key: crate::arena::DescendantScoreKey {
429            neg_feerate_frac: (0, 1),
430            time: current_time,
431        }, // Will be recomputed by arena
432        idx_randomized: None,
433    };
434
435    // Insert into arena (computes and caches index keys)
436    let entry_id = inner.arena.insert(entry);
437
438    // Update parent->child links
439    for &parent_id in &ancestors {
440        if let Some(parent_entry) = inner.arena.get_mut(parent_id) {
441            parent_entry.children.insert(entry_id);
442        }
443    }
444
445    // Update ancestor state for all ancestors
446    let size_delta = ws.vsize;
447    let fee_delta = bitcoin::SignedAmount::from_sat(ws.modified_fee.to_sat() as i64);
448    let sigops_delta = ws.sigop_cost;
449
450    for &ancestor_id in &ancestors {
451        inner.arena.update_ancestor_state(
452            ancestor_id,
453            size_delta,
454            fee_delta,
455            1, // count_delta
456            sigops_delta,
457        );
458    }
459
460    // Update descendant state for all ancestors
461    for &ancestor_id in &ancestors {
462        inner.arena.update_descendant_state(
463            ancestor_id,
464            size_delta,
465            fee_delta,
466            1, // count_delta
467        );
468    }
469
470    // Add to map_next_tx (mark outputs as spent)
471    let txid = ws.tx.compute_txid();
472    for input in &ws.tx.input {
473        inner.map_next_tx.insert(input.previous_output, txid);
474    }
475
476    // Add to mempool overlay in coins cache
477    coins_cache.add_mempool_coins(&ws.tx);
478
479    // Update statistics
480    inner.total_tx_size += ws.tx_weight.to_wu();
481    inner.total_fee = inner
482        .total_fee
483        .checked_add(ws.base_fee)
484        .ok_or(MempoolError::FeeOverflow)?;
485
486    // Mark as unbroadcast
487    inner.unbroadcast.insert(txid);
488
489    Ok(entry_id)
490}

522/// Verify transaction scripts under the provided verification flags.
523pub fn check_inputs<Block, Client>(
524    ws: &ValidationWorkspace,
525    coins_cache: &mut CoinsViewCache<Block, Client>,
526    flags: VerifyFlags,
527) -> Result<(), MempoolError>
528where
529    Block: BlockT,
530    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + Send + Sync,
531    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
532{
533    let tx = &ws.tx;
534
535    for (input_index, txin) in tx.input.iter().enumerate() {
536        let coin = coins_cache
537            .get_coin(&txin.previous_output)?
538            .ok_or_else(|| MempoolError::MissingInputs {
539                parents: vec![txin.previous_output.txid],
540            })?;
541
542        let mut checker =
543            TransactionSignatureChecker::new(tx, input_index, coin.output.value.to_sat());
544
545        verify_script(
546            &txin.script_sig,
547            &coin.output.script_pubkey,
548            &txin.witness,
549            &flags,
550            &mut checker,
551        )
552        .map_err(|e| MempoolError::ScriptValidationFailed(format!("input {input_index}: {e}")))?;
553    }
554
555    Ok(())
556}

558/// Check if a replacement transaction satisfies BIP125 rules.
559///
560/// BIP125 Rules:
561/// 1. All original transactions signal replaceability
562/// 2. Replacement doesn't introduce new unconfirmed inputs
563/// 3. Replacement pays higher absolute fee
564/// 4. Replacement pays for its own bandwidth
565/// 5. Replacement pays for replaced bandwidth
566/// 6. No more than max_replacement_txs original transactions replaced
567pub fn check_rbf_policy<Block, Client>(
568    ws: &ValidationWorkspace,
569    inner: &MemPoolInner,
570    _coins_cache: &CoinsViewCache<Block, Client>,
571    options: &MemPoolOptions,
572) -> Result<ConflictSet, MempoolError>
573where
574    Block: BlockT,
575    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + Send + Sync,
576    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
577{
578    let tx = &ws.tx;
579
580    // Step 1: Find ALL direct conflicts (not just first one!)
581    let mut direct_conflicts = HashSet::new();
582
583    for input in &tx.input {
584        if let Some(conflicting_txid) = inner.get_conflict_tx(&input.previous_output) {
585            if let Some(entry_id) = inner.arena.get_by_txid(&conflicting_txid) {
586                direct_conflicts.insert(entry_id);
587            }
588        }
589    }
590
591    if direct_conflicts.is_empty() {
592        return Err(MempoolError::NoConflictToReplace);
593    }
594
595    // Step 2: Expand to include all descendants
596    let mut all_conflicts = HashSet::new();
597    for &conflict_id in &direct_conflicts {
598        inner.calculate_descendants(conflict_id, &mut all_conflicts);
599    }
600
601    // Rule 6: Check replacement count limit EARLY
602    if all_conflicts.len() > options.max_replacement_txs {
603        return Err(MempoolError::TooManyReplacements(all_conflicts.len()));
604    }
605
606    // Rule 1: All direct conflicts must signal RBF
607    for &conflict_id in &direct_conflicts {
608        let conflict_entry = inner
609            .arena
610            .get(conflict_id)
611            .ok_or(MempoolError::MissingConflict)?;
612
613        if !conflict_entry.signals_rbf() {
614            return Err(MempoolError::TxNotReplaceable);
615        }
616    }
617
618    // Step 3: Calculate total fees and size of ALL replaced txs
619    let mut replaced_fees = bitcoin::Amount::ZERO;
620    let mut replaced_size: i64 = 0;
621    let mut removed_transactions = Vec::with_capacity(all_conflicts.len());
622
623    for &conflict_id in &all_conflicts {
624        let entry = inner
625            .arena
626            .get(conflict_id)
627            .ok_or(MempoolError::MissingConflict)?;
628
629        replaced_fees = replaced_fees
630            .checked_add(entry.modified_fee)
631            .ok_or(MempoolError::FeeOverflow)?;
632        replaced_size += entry.vsize();
633
634        // CRITICAL: Capture Arc<Transaction> for coins cache cleanup
635        removed_transactions.push(entry.tx.clone());
636    }
637
638    // Rule 2: Replacement doesn't introduce new unconfirmed inputs
639    for input in &tx.input {
640        if let Some(parent_id) = inner.arena.get_by_txid(&input.previous_output.txid) {
641            if !all_conflicts.contains(&parent_id) {
642                // Spending mempool tx that's not being replaced!
643                return Err(MempoolError::NewUnconfirmedInput);
644            }
645        }
646    }
647
648    // Rule 3: Replacement pays higher absolute fee
649    if ws.modified_fee <= replaced_fees {
650        return Err(MempoolError::InsufficientFee(format!(
651            "replacement fee {} <= replaced fees {replaced_fees}",
652            ws.modified_fee
653        )));
654    }
655
656    // Rule 4: Pays for own bandwidth (additional fee >= min_relay * new_size)
657    let min_relay_rate = options.min_relay_fee_rate();
658    let own_bandwidth_fee = min_relay_rate.get_fee(ws.vsize);
659
660    let additional_fee_sat = ws.modified_fee.to_sat() as i64 - replaced_fees.to_sat() as i64;
661    if additional_fee_sat < own_bandwidth_fee.to_sat() as i64 {
662        return Err(MempoolError::InsufficientFee(format!(
663            "doesn't pay for own bandwidth: additional {} < required {own_bandwidth_fee}",
664            bitcoin::Amount::from_sat(additional_fee_sat as u64)
665        )));
666    }
667
668    // Rule 5: Pays for replaced bandwidth (additional >= min_relay * replaced_size)
669    let replaced_bandwidth_fee = min_relay_rate.get_fee(replaced_size);
670
671    if additional_fee_sat < replaced_bandwidth_fee.to_sat() as i64 {
672        return Err(MempoolError::InsufficientFee(format!(
673            "doesn't pay for replaced bandwidth: additional {} < required {replaced_bandwidth_fee}",
674            bitcoin::Amount::from_sat(additional_fee_sat as u64)
675        )));
676    }
677
678    Ok(ConflictSet {
679        direct_conflicts,
680        all_conflicts,
681        removed_transactions,
682        replaced_fees,
683        replaced_size,
684    })
685}

687/// Check whether the transaction is final with respect to the current height and time.
688fn is_final_tx(tx: &Transaction, height: u32, block_time: u32) -> bool {
689    if tx.lock_time == LockTime::ZERO {
690        return true;
691    }
692
693    let lock_time_limit = if tx.lock_time.to_consensus_u32() < LOCK_TIME_THRESHOLD {
694        height
695    } else {
696        block_time
697    };
698
699    if tx.lock_time.to_consensus_u32() < lock_time_limit {
700        return true;
701    }
702
703    tx.input.iter().all(|txin| txin.sequence.is_final())
704}

706/// Sort package transactions in topological order (parents before children).
707///
708/// Returns transactions sorted so all parents appear before their children.
709/// Returns error if package contains cycles.
710pub fn topological_sort_package(
711    transactions: Vec<Arc<Transaction>>,
712) -> Result<Vec<Arc<Transaction>>, MempoolError> {
713    use std::collections::{HashMap, VecDeque};
714
715    // Build txid -> tx mapping
716    let tx_map: HashMap<Txid, Arc<Transaction>> = transactions
717        .iter()
718        .map(|tx| (tx.compute_txid(), tx.clone()))
719        .collect();
720
721    // Build dependency graph: txid -> set of in-package children
722    let mut children: HashMap<Txid, Vec<Txid>> = HashMap::new();
723    let mut in_degree: HashMap<Txid, usize> = HashMap::new();
724
725    // Initialize in-degree for all txs
726    for tx in &transactions {
727        in_degree.insert(tx.compute_txid(), 0);
728    }
729
730    // Build edges
731    for tx in &transactions {
732        let txid = tx.compute_txid();
733        for input in &tx.input {
734            let parent_txid = input.previous_output.txid;
735            // Only track in-package dependencies
736            if tx_map.contains_key(&parent_txid) {
737                children.entry(parent_txid).or_default().push(txid);
738                *in_degree.get_mut(&txid).expect("txid must exist") += 1;
739            }
740        }
741    }
742
743    // Kahn's algorithm for topological sort
744    let mut queue: VecDeque<Txid> = in_degree
745        .iter()
746        .filter(|&(_, &degree)| degree == 0)
747        .map(|(&txid, _)| txid)
748        .collect();
749
750    let mut sorted = Vec::new();
751
752    while let Some(txid) = queue.pop_front() {
753        sorted.push(tx_map.get(&txid).expect("txid must exist").clone());
754
755        if let Some(child_list) = children.get(&txid) {
756            for &child_txid in child_list {
757                let degree = in_degree.get_mut(&child_txid).expect("child must exist");
758                *degree -= 1;
759                if *degree == 0 {
760                    queue.push_back(child_txid);
761                }
762            }
763        }
764    }
765
766    // Check for cycles
767    if sorted.len() != transactions.len() {
768        return Err(MempoolError::PackageCyclicDependencies);
769    }
770
771    Ok(sorted)
772}

774/// Calculate total package fees and feerate.
775///
776/// Builds local map of in-package outputs to handle dependencies.
777/// Returns (total_fee, total_vsize, package_feerate).
778pub fn calculate_package_feerate<Block, Client>(
779    sorted_txs: &[Arc<Transaction>],
780    inner: &MemPoolInner,
781    coins_cache: &mut CoinsViewCache<Block, Client>,
782) -> Result<(bitcoin::Amount, i64, FeeRate), MempoolError>
783where
784    Block: BlockT,
785    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + Send + Sync,
786    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
787{
788    use bitcoin::OutPoint;
789
790    // Build map of in-package outputs: outpoint -> TxOut
791    // This allows children to see parent outputs before they're in mempool
792    let mut package_outputs: HashMap<OutPoint, bitcoin::TxOut> = HashMap::new();
793    for tx in sorted_txs {
794        let txid = tx.compute_txid();
795        for (vout, output) in tx.output.iter().enumerate() {
796            package_outputs.insert(
797                OutPoint {
798                    txid,
799                    vout: vout as u32,
800                },
801                output.clone(),
802            );
803        }
804    }
805
806    let mut total_fee = bitcoin::Amount::ZERO;
807    let mut total_vsize = 0i64;
808
809    for tx in sorted_txs {
810        // Calculate vsize from weight
811        let vsize = tx.weight().to_vbytes_ceil() as i64;
812        total_vsize = total_vsize
813            .checked_add(vsize)
814            .ok_or(MempoolError::FeeOverflow)?;
815
816        // Calculate input value
817        let mut input_value = bitcoin::Amount::ZERO;
818        for input in &tx.input {
819            let outpoint = &input.previous_output;
820
821            // Check in this order:
822            // 1. In-package outputs (earlier in topological order)
823            // 2. Mempool outputs
824            // 3. UTXO from coins_cache
825            let output = if let Some(pkg_output) = package_outputs.get(outpoint) {
826                pkg_output.clone()
827            } else if let Some(parent_id) = inner.arena.get_by_txid(&outpoint.txid) {
828                // Spends mempool transaction
829                let parent =
830                    inner
831                        .arena
832                        .get(parent_id)
833                        .ok_or_else(|| MempoolError::MissingInputs {
834                            parents: vec![outpoint.txid],
835                        })?;
836                parent
837                    .tx
838                    .output
839                    .get(outpoint.vout as usize)
840                    .ok_or_else(|| MempoolError::MissingInputs {
841                        parents: vec![outpoint.txid],
842                    })?
843                    .clone()
844            } else {
845                // Spends UTXO
846                let coin =
847                    coins_cache
848                        .get_coin(outpoint)?
849                        .ok_or_else(|| MempoolError::MissingInputs {
850                            parents: vec![outpoint.txid],
851                        })?;
852                coin.output
853            };
854
855            input_value = input_value
856                .checked_add(output.value)
857                .ok_or(MempoolError::FeeOverflow)?;
858        }
859
860        // Calculate output value
861        let output_value: bitcoin::Amount = tx.output.iter().map(|out| out.value).sum();
862
863        // Add to total fee
864        let tx_fee = input_value
865            .checked_sub(output_value)
866            .ok_or(MempoolError::NegativeFee)?;
867        total_fee = total_fee
868            .checked_add(tx_fee)
869            .ok_or(MempoolError::FeeOverflow)?;
870    }
871
872    // Guard against division by zero
873    if total_vsize == 0 {
874        return Err(MempoolError::PackageSizeTooLarge(0));
875    }
876
877    // Calculate package feerate: (total_fee_sats * 1000) / total_vsize
878    let feerate_sat_kvb = total_fee
879        .to_sat()
880        .checked_mul(1000)
881        .ok_or(MempoolError::FeeOverflow)?
882        .checked_div(total_vsize as u64)
883        .unwrap_or(0);
884
885    let package_feerate = FeeRate::from_sat_per_kvb(feerate_sat_kvb);
886
887    Ok((total_fee, total_vsize, package_feerate))
888}

957/// Validate and accept a package of transactions (two-phase commit).
958///
959/// Phase 1: Pre-validate all transactions without modifying mempool
960/// Phase 2: Finalize all transactions if all validations passed
961#[allow(clippy::too_many_arguments)]
962pub fn validate_package<Block, Client>(
963    package: &crate::types::Package,
964    inner: &mut MemPoolInner,
965    coins_cache: &mut CoinsViewCache<Block, Client>,
966    options: &MemPoolOptions,
967    current_height: u32,
968    best_block: Block::Hash,
969    current_time: i64,
970    sequence_start: u64,
971) -> Result<crate::types::PackageValidationResult, MempoolError>
972where
973    Block: BlockT,
974    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + sc_client_api::AuxStore + Send + Sync,
975    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
976{
977    // Step 1: Check package limits
978    if package.transactions.len() > options.max_package_count {
979        return Err(MempoolError::PackageTooLarge(
980            package.transactions.len(),
981            options.max_package_count,
982        ));
983    }
984
985    // Calculate total vsize
986    let total_vsize: u64 = package
987        .transactions
988        .iter()
989        .map(|tx| tx.weight().to_vbytes_ceil())
990        .sum();
991
992    if total_vsize > options.max_package_vsize {
993        return Err(MempoolError::PackageSizeTooLarge(total_vsize));
994    }
995
996    // Step 2: Topological sort (parents before children)
997    let sorted_txs = topological_sort_package(package.transactions.clone())?;
998
999    // Step 3: Calculate package feerate (with in-package outputs map)
1000    let (_total_fee, _total_vsize_i64, package_feerate) =
1001        calculate_package_feerate(&sorted_txs, inner, coins_cache)?;
1002
1003    // Step 4: Check package meets minimum feerate
1004    if package_feerate < options.min_relay_fee_rate() {
1005        return Err(MempoolError::PackageFeeTooLow(format!(
1006            "package feerate {package_feerate:?} < min {:?}",
1007            options.min_relay_fee_rate()
1008        )));
1009    }
1010
1011    // PHASE 1: Pre-validate all transactions (NO mempool modification)
1012    let mut validated_workspaces = Vec::new();
1013    // TODO (Phase 8): Get actual MTP from runtime API instead of using current_time
1014    let current_mtp = current_time;
1015
1016    for tx in &sorted_txs {
1017        match pre_validate_package_tx(
1018            tx.clone(),
1019            inner,
1020            coins_cache,
1021            options,
1022            current_height,
1023            current_mtp,
1024            best_block,
1025            package_feerate,
1026        ) {
1027            Ok(ws) => {
1028                // Add to coins cache overlay for next tx in package
1029                // This makes in-package parent outputs visible to children
1030                coins_cache.add_mempool_coins(&ws.tx);
1031                validated_workspaces.push(ws);
1032            }
1033            Err(e) => {
1034                // Validation failed - rollback coins cache overlay
1035                for ws in &validated_workspaces {
1036                    coins_cache.remove_mempool_tx(&ws.tx);
1037                }
1038                return Err(MempoolError::PackageTxValidationFailed(
1039                    tx.compute_txid(),
1040                    format!("{e}"),
1041                ));
1042            }
1043        }
1044    }
1045
1046    // PHASE 2: All validations passed - finalize all transactions
1047    let mut accepted = Vec::new();
1048    let mut sequence = sequence_start;
1049
1050    // Get Bitcoin block hash for entry_block_hash
1051    use subcoin_primitives::BackendExt;
1052    let entry_block_hash = coins_cache
1053        .client()
1054        .clone()
1055        .bitcoin_block_hash_for(best_block)
1056        .unwrap_or_else(bitcoin::BlockHash::all_zeros);
1057
1058    for ws in validated_workspaces {
1059        let txid = ws.tx.compute_txid();
1060        finalize_tx(
1061            ws,
1062            inner,
1063            coins_cache,
1064            current_height,
1065            current_time,
1066            current_mtp,
1067            entry_block_hash,
1068            sequence,
1069        )?;
1070        accepted.push(txid);
1071        sequence += 1;
1072    }
1073
1074    Ok(crate::types::PackageValidationResult {
1075        accepted,
1076        package_feerate,
1077    })
1078}

1080/// BIP68 sequence lock constants
1081const SEQUENCE_LOCKTIME_DISABLE_FLAG: u32 = 1u32 << 31;
1082const SEQUENCE_LOCKTIME_TYPE_FLAG: u32 = 1u32 << 22;
1083const SEQUENCE_LOCKTIME_MASK: u32 = 0x0000ffff;
1084const SEQUENCE_LOCKTIME_GRANULARITY: i64 = 512; // 512 seconds
1085
1086/// Calculate lock points for a transaction at the current chain tip (BIP68/BIP112).
1087///
1088/// This function computes the earliest block height and median time past (MTP) at which
1089/// a transaction becomes valid, based on its inputs' sequence values according to BIP68.
1090///
1091/// ## BIP68 Relative Lock-Time Encoding
1092///
1093/// Each input's nSequence value encodes:
1094/// - **Bit 31 (DISABLE_FLAG)**: If set, sequence locks are disabled for this input
1095/// - **Bit 22 (TYPE_FLAG)**: If clear = height-based, if set = time-based
1096/// - **Bits 0-15 (VALUE)**: The relative lock-time value
1097///
1098/// Height-based: Value is number of blocks since input's confirmation
1099/// Time-based: Value * 512 seconds since input's MTP
1100///
1101/// Coinbase inputs MUST have DISABLE_FLAG set (BIP68 rule).
1102///
1103/// ## Lock Points Calculation
1104///
1105/// For each input (unless DISABLE_FLAG is set):
1106/// 1. Lookup the input coin's block metadata (height, MTP)
1107/// 2. If height-based: `lock_height = coin_height + sequence_value + 1`
1108/// 3. If time-based: `lock_time = coin_mtp + (sequence_value * 512)`
1109/// 4. Track max lock height/time across all inputs
1110/// 5. Track highest input block hash (for reorg detection)
1111///
1112/// ## Returns
1113///
1114/// `LockPoints` with:
1115/// - `height`: Minimum block height for validity (-1 if no height locks)
1116/// - `time`: Minimum MTP for validity (0 if no time locks)
1117/// - `max_input_block`: Highest block containing any input (for reorg detection)
1118///
1119/// ## Errors
1120///
1121/// Returns `MempoolError::NonBIP68Final` if the transaction is not yet final according to
1122/// current chain tip.
1123pub fn calculate_lock_points_at_tip<Block, Client>(
1124    tx: &Transaction,
1125    coins_cache: &mut CoinsViewCache<Block, Client>,
1126    current_height: u32,
1127    current_mtp: i64,
1128) -> Result<LockPoints, MempoolError>
1129where
1130    Block: BlockT,
1131    Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + Send + Sync,
1132    Client::Api: subcoin_runtime_primitives::SubcoinApi<Block>,
1133{
1134    let mut lock_height: i32 = -1;
1135    let mut lock_time: i64 = 0;
1136    let max_input_block: Option<bitcoin::BlockHash> = None;
1137    let mut max_input_height: u32 = 0;
1138
1139    for input in &tx.input {
1140        // Get coin for this input
1141        let coin = coins_cache
1142            .get_coin(&input.previous_output)?
1143            .ok_or_else(|| MempoolError::MissingInputs {
1144                parents: vec![input.previous_output.txid],
1145            })?;
1146
1147        let sequence = input.sequence.0;
1148
1149        // Check BIP68: coinbase inputs MUST have DISABLE_FLAG set
1150        if coin.is_coinbase && (sequence & SEQUENCE_LOCKTIME_DISABLE_FLAG) == 0 {
1151            return Err(MempoolError::NonBIP68Final);
1152        }
1153
1154        // Skip if sequence locks disabled for this input
1155        if (sequence & SEQUENCE_LOCKTIME_DISABLE_FLAG) != 0 {
1156            continue;
1157        }
1158
1159        // Track highest input block for reorg detection
1160        if coin.height > max_input_height {
1161            max_input_height = coin.height;
1162            // We'll need to get the block hash from the runtime
1163            // For now, we'll set this in a second pass or leave as None
1164            // TODO: Get block hash from runtime API
1165        }
1166
1167        let masked_sequence = sequence & SEQUENCE_LOCKTIME_MASK;
1168
1169        if (sequence & SEQUENCE_LOCKTIME_TYPE_FLAG) == 0 {
1170            // Height-based relative lock
1171            // Formula: lock_height = coin_height + masked_sequence + 1
1172            let coin_lock_height = coin
1173                .height
1174                .checked_add(masked_sequence)
1175                .and_then(|h| h.checked_add(1))
1176                .ok_or(MempoolError::NonBIP68Final)?;
1177
1178            lock_height = lock_height.max(coin_lock_height as i32);
1179        } else {
1180            // Time-based relative lock (512-second granularity)
1181            // Formula: lock_time = coin_mtp + (masked_sequence * 512)
1182            let offset = (masked_sequence as i64)
1183                .checked_mul(SEQUENCE_LOCKTIME_GRANULARITY)
1184                .ok_or(MempoolError::NonBIP68Final)?;
1185
1186            let coin_lock_time = coin
1187                .median_time_past
1188                .checked_add(offset)
1189                .ok_or(MempoolError::NonBIP68Final)?;
1190
1191            lock_time = lock_time.max(coin_lock_time);
1192        }
1193    }
1194
1195    // Check if transaction is final at current tip
1196    if lock_height > 0 && (current_height as i32) < lock_height {
1197        return Err(MempoolError::NonBIP68Final);
1198    }
1199
1200    if lock_time > 0 && current_mtp < lock_time {
1201        return Err(MempoolError::NonBIP68Final);
1202    }
1203
1204    Ok(LockPoints {
1205        height: lock_height,
1206        time: lock_time,
1207        max_input_block,
1208    })
1209}
