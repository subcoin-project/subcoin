1use jsonrpsee::RpcModule;
2use sc_service::SpawnTaskHandle;
3use sc_utils::mpsc::TracingUnboundedSender;
4use std::sync::Arc;
5use subcoin_indexer::IndexerQuery;
6use subcoin_network::NetworkApi;
7use subcoin_primitives::TransactionIndex;
8use subcoin_runtime::interface::OpaqueBlock;
9use subcoin_service::FullClient;
10use substrate_frame_rpc_system::{System as FrameSystem, SystemApiServer as _};

12/// Configuration for the RPC module.
13pub struct RpcConfig {
14    /// Bitcoin network (mainnet, testnet, etc.).
15    pub bitcoin_network: bitcoin::Network,
16    /// Whether to enable Bitcoin Core compatible RPC methods.
17    pub enable_bitcoind_rpc: bool,
18}
19
20/// Instantiate all full RPC extensions.
21#[allow(clippy::too_many_arguments)]
22pub fn gen_rpc_module(
23    system_info: sc_rpc::system::SystemInfo,
24    client: Arc<FullClient>,
25    spawn_handle: SpawnTaskHandle,
26    system_rpc_tx: TracingUnboundedSender<sc_rpc::system::Request<OpaqueBlock>>,
27    network_api: Arc<dyn NetworkApi>,
28    transaction_indexer: Arc<dyn TransactionIndex + Send + Sync>,
29    indexer_query: Option<Arc<IndexerQuery>>,
30    rpc_config: RpcConfig,
31) -> Result<RpcModule<()>, sc_service::Error> {
32    use sc_rpc::chain::ChainApiServer;
33    use sc_rpc::state::{ChildStateApiServer, StateApiServer};
34    use sc_rpc::system::SystemApiServer;
35
36    let mut module = RpcModule::new(());
37
38    let dummy_pool = Arc::new(subcoin_service::transaction_pool::new_dummy_pool());
39
40    let task_executor = Arc::new(spawn_handle);
41
42    let system = sc_rpc::system::System::new(system_info, system_rpc_tx).into_rpc();
43    let chain = sc_rpc::chain::new_full(client.clone(), task_executor.clone()).into_rpc();
44    let (state, child_state) = {
45        let (state, child_state) = sc_rpc::state::new_full(client.clone(), task_executor, None);
46        (state.into_rpc(), child_state.into_rpc())
47    };
48    let _frame_system = FrameSystem::new(client.clone(), dummy_pool).into_rpc();
49
50    let into_service_error =
51        |e: jsonrpsee::core::error::RegisterMethodError| sc_service::Error::Application(e.into());
52
53    // Substrate RPCs.
54    module.merge(system).map_err(into_service_error)?;
55    module.merge(chain).map_err(into_service_error)?;
56    module.merge(state).map_err(into_service_error)?;
57    module.merge(child_state).map_err(into_service_error)?;
58    // module.merge(frame_system).map_err(into_service_error)?;
59
60    // Subcoin RPCs.
61    subcoin_rpc::SubcoinRpc::<_, _, subcoin_service::TransactionAdapter>::new(
62        client.clone(),
63        network_api.clone(),
64        transaction_indexer.clone(),
65        indexer_query,
66        rpc_config.bitcoin_network,
67    )
68    .merge_into(&mut module)
69    .map_err(into_service_error)?;
70
71    // Bitcoin Core compatible RPCs (for electrs compatibility).
72    if rpc_config.enable_bitcoind_rpc {
73        subcoin_rpc_bitcoind::BitcoindRpc::<_, _, subcoin_service::TransactionAdapter>::new(
74            client,
75            network_api,
76            transaction_indexer,
77            rpc_config.bitcoin_network,
78        )
79        .merge_into(&mut module)
80        .map_err(into_service_error)?;
81
82        tracing::info!("Bitcoin Core compatible RPC methods enabled");
83    }
84
85    Ok(module)
86}
