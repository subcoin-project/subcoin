6/// Chain.
7///
8/// Currently only Bitcoin is supported, more chains may be supported in the future.
9// TODO: This clippy warning will be fixed once more chains are supported.
10#[allow(clippy::enum_variant_names)]
11#[derive(Clone, Copy, Default, Debug, clap::ValueEnum)]
12pub enum Chain {
13    /// Bitcoin mainnet.
14    #[default]
15    BitcoinMainnet,
16    /// Bitcoin testnet
17    BitcoinTestnet,
18    /// Bitcoin signet.
19    BitcoinSignet,
20}
21
22impl Chain {
23    /// Returns the value of `id` in `SubstrateCli::load_spec(id)`.
24    pub fn chain_spec_id(&self) -> &'static str {
25        // Convert to kebab-case for consistency in CLI.
26        match self {
27            Self::BitcoinMainnet => "bitcoin-mainnet",
28            Self::BitcoinTestnet => "bitcoin-testnet",
29            Self::BitcoinSignet => "bitcoin-signet",
30        }
31    }
32}
33
34/// Provides the default values for optional parameters.
35pub(crate) struct Defaults;
36
37impl Defaults {
38    pub(crate) fn min_sync_peer_threshold(is_dev: bool) -> usize {
39        if is_dev { 0 } else { 3 }
40    }
41
42    pub(crate) fn sync_strategy(is_dev: bool) -> SyncStrategy {
43        if is_dev {
44            SyncStrategy::BlocksFirst
45        } else {
46            SyncStrategy::HeadersFirst
47        }
48    }
49
50    pub(crate) fn script_engine(is_dev: bool) -> ScriptEngine {
51        if is_dev {
52            ScriptEngine::Subcoin
53        } else {
54            ScriptEngine::Core
55        }
56    }
57}

59/// Subcoin networking params.
60#[derive(Debug, Clone, Parser)]
61pub struct NetworkParams {
62    /// Specify the remote peer address to connect.
63    #[clap(long, value_name = "SEEDNODE")]
64    pub seednodes: Vec<String>,
65
66    /// Connect to the nodes specified by `--seednode` only.
67    ///
68    /// Do not attempt to connect to the builtin seednodes.
69    #[clap(long)]
70    pub seednode_only: bool,
71
72    /// Specify the local address and listen on it.
73    #[clap(long, default_value = "127.0.0.1:8333")]
74    pub listen: PeerId,
75
76    /// Whether to connect to the nodes using IPv6 address.
77    #[clap(long)]
78    pub ipv4_only: bool,
79
80    /// Specify the maximum number of inbound subcoin networking peers.
81    #[clap(long, default_value = "100")]
82    pub max_inbound_peers: usize,
83
84    /// Specify the maximum number of outbound subcoin networking peers.
85    #[clap(long, default_value = "20")]
86    pub max_outbound_peers: usize,
87
88    /// Persistent peer latency threshold in milliseconds (ms).
89    ///
90    /// Only peers with a latency lower than this threshold can possibly be saved to disk.
91    ///
92    /// Default value is 500 ms
93    #[clap(long, default_value_t = 500)]
94    pub persistent_peer_latency_threshold: u128,
95
96    /// Minimum peer threshold required to start block sync.
97    ///
98    /// The chain sync won't be started until the number of sync peers reaches this threshold.
99    /// Providing `0` disables the peer threshold limit.
100    ///
101    /// If not provided, defaults to 0 in development mode and 3 otherwise.
102    #[arg(long)]
103    pub min_sync_peer_threshold: Option<usize>,
104}

106#[derive(Debug, Clone, Parser)]
107pub struct CommonParams {
108    /// Specify the development chain.
109    ///
110    /// This flag sets `--min-sync-peer-threshold=0`, `--sync-strategy=blocks-first` and `--script-engine=subcoin`
111    /// flags, unless explicitly overridden.
112    #[arg(long)]
113    pub dev: bool,
114
115    /// Specify the chain.
116    #[arg(long, value_name = "CHAIN", default_value = "bitcoin-mainnet")]
117    pub chain: Chain,
118
119    /// Specify the block verification level.
120    #[clap(long, default_value = "full")]
121    pub block_verification: BlockVerification,
122
123    /// Specifies the backend for Bitcoin script verification.
124    ///
125    /// If not provided, defaults to `ScriptEngine::Subcoin` in development mode
126    /// and `ScriptEngine::Core` otherwise.
127    #[clap(long)]
128    pub script_engine: Option<ScriptEngine>,
129
130    /// Specify custom base path.
131    #[arg(long, short = 'd', value_name = "PATH")]
132    pub base_path: Option<PathBuf>,
133
134    /// Sets a custom logging filter (syntax: `<target>=<level>`).
135    ///
136    /// Log levels (least to most verbose) are `error`, `warn`, `info`, `debug`, and `trace`.
137    ///
138    /// By default, all targets log `info`. The global log level can be set with `-l<level>`.
139    ///
140    /// Multiple `<target>=<level>` entries can be specified and separated by a comma.
141    ///
142    /// *Example*: `--log error,sync=debug,grandpa=warn`.
143    /// Sets Global log level to `error`, sets `sync` target to debug and grandpa target to `warn`.
144    #[arg(short = 'l', long, value_name = "LOG_PATTERN", num_args = 1..)]
145    pub log: Vec<String>,
146
147    /// Sets a custom profiling filter.
148    ///
149    /// Syntax is the same as for logging (`--log`).
150    #[arg(long, value_name = "TARGETS")]
151    pub tracing_targets: Option<String>,
152}
153
154impl CommonParams {
155    /// Converts the current [`CommonParams`] to a [`sc_cli::SharedParams`].
156    pub fn as_shared_params(&self) -> sc_cli::SharedParams {
157        // TODO: expose more flags?
158        sc_cli::SharedParams {
159            chain: Some(self.chain.chain_spec_id().to_string()),
160            dev: false, // TODO: align with is_dev?
161            base_path: self.base_path.clone(),
162            log: self.log.clone(),
163            detailed_log_output: false,
164            disable_log_color: false,
165            enable_log_reloading: false,
166            tracing_targets: self.tracing_targets.clone(),
167            tracing_receiver: sc_cli::TracingReceiver::Log,
168        }
169    }
170
171    /// Determines the Bitcoin network type based on the current chain setting.
172    pub fn bitcoin_network(&self) -> bitcoin::Network {
173        match self.chain {
174            Chain::BitcoinMainnet => bitcoin::Network::Bitcoin,
175            Chain::BitcoinTestnet => bitcoin::Network::Testnet,
176            Chain::BitcoinSignet => bitcoin::Network::Signet,
177        }
178    }
179
180    pub fn import_config(&self, is_dev: bool) -> ImportConfig {
181        ImportConfig {
182            network: self.bitcoin_network(),
183            block_verification: self.block_verification,
184            execute_block: true,
185            script_engine: self
186                .script_engine
187                .unwrap_or_else(|| Defaults::script_engine(is_dev)),
188        }
189    }
190}
