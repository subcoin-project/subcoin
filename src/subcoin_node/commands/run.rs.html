16/// Options for configuring the Subcoin network behavior.
17#[derive(Copy, Default, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, clap::ValueEnum)]
18pub enum SubcoinNetworkOption {
19    /// Fully enables the Subcoin network, including block sync.
20    #[default]
21    Full,
22    /// Do no run Subcoin network at all.
23    None,
24    /// Enables the Subcoin network with block sync disabled.
25    ///
26    /// This option allows block sync to occur exclusively over the Substrate network,
27    /// while retaining partial Bitcoin P2P functionality (e.g., broadcasting Bitcoin transactions).
28    NoBlockSync,
29}
30
31impl SubcoinNetworkOption {
32    fn no_block_sync(&self) -> bool {
33        matches!(self, Self::None | Self::NoBlockSync)
34    }
35}

37/// The `run` command used to start a Subcoin node.

40    /// Specify the Bitcoin major sync strategy.
41    ///
42    /// If not provided, defaults to `SyncStrategy::BlocksFirst` in development mode and
43    /// `SyncStrategy::HeadersFirst` otherwise.
44    #[clap(long)]
45    pub sync_strategy: Option<SyncStrategy>,
46
47    /// Specify the target block height for the initial block download.
48    ///
49    /// The syncing process will stop once the best block height reaches the specified target.
50    /// This flag is only used when syncing a snapshot node, which will steadily serve the state
51    /// at specified block height for other node to do a state sync and UTXO snapshot download.
52    #[clap(long)]
53    pub sync_target: Option<u32>,
54
55    /// Do not run the finalizer which will finalize the blocks on confirmation depth.
56    #[clap(long)]
57    pub no_finalizer: bool,
58
59    /// Enable transaction indexing service.
60    #[clap(long)]
61    pub tx_index: bool,
62
63    /// Specify the Subcoin network behavior.
64    #[clap(long, default_value = "full")]
65    pub subcoin_network: SubcoinNetworkOption,
66
67    /// Disable automatic hardware benchmarks.
68    ///
69    /// By default these benchmarks are automatically ran at startup and measure
70    /// the CPU speed, the memory bandwidth and the disk speed.
71    ///
72    /// The results are then printed out in the logs, and also sent as part of
73    /// telemetry, if telemetry is enabled.
74    #[arg(long)]
75    pub no_hardware_benchmarks: bool,

106impl Run {
107    fn subcoin_network_config(&self, network: bitcoin::Network) -> subcoin_network::Config {
108        let is_dev = self.common_params.dev;
109
110        subcoin_network::Config {
111            network,
112            listen_on: self.subcoin_network_params.listen,
113            seednodes: self.subcoin_network_params.seednodes.clone(),
114            seednode_only: self.subcoin_network_params.seednode_only,
115            ipv4_only: self.subcoin_network_params.ipv4_only,
116            sync_target: self.sync_target,
117            max_outbound_peers: self.subcoin_network_params.max_outbound_peers,
118            max_inbound_peers: self.subcoin_network_params.max_inbound_peers,
119            min_sync_peer_threshold: self
120                .subcoin_network_params
121                .min_sync_peer_threshold
122                .unwrap_or_else(|| Defaults::min_sync_peer_threshold(is_dev)),
123            persistent_peer_latency_threshold: self
124                .subcoin_network_params
125                .persistent_peer_latency_threshold,
126            sync_strategy: self
127                .sync_strategy
128                .unwrap_or_else(|| Defaults::sync_strategy(is_dev)),
129            block_sync: self.block_sync_option(),
130            base_path: base_path_or_default(
131                self.common_params.base_path.clone().map(Into::into),
132                &crate::substrate_cli::SubstrateCli::executable_name(),
133            )
134            .path()
135            .to_path_buf(),
136            memory_config: Default::default(),
137        }
138    }
139
140    fn block_sync_option(&self) -> BlockSyncOption {
141        let fast_sync_enabled = matches!(
142            self.substrate_network_params.sync,
143            SyncMode::Fast | SyncMode::FastUnsafe
144        );
145
146        // The block sync from bitcoin P2P network will be temporarily disabled
147        // if the fast sync is enabled.
148        match (self.subcoin_network.no_block_sync(), fast_sync_enabled) {
149            (true, _) => BlockSyncOption::Off,
150            (false, true) => BlockSyncOption::PausedUntilFastSync,
151            (false, false) => BlockSyncOption::AlwaysOn,
152        }
153    }
154}

168    /// Start subcoin node.
169    pub async fn start(
170        self,
171        mut config: Configuration,
172        storage_monitor: sc_storage_monitor::StorageMonitorParams,
173    ) -> sc_cli::Result<TaskManager> {
174        let Self {
175            shared_params: _,
176            run,
177        } = self;
178
179        if matches!(run.substrate_network_params.sync, SyncMode::Warp) {
180            return Err(sc_cli::Error::Input(
181                "--sync=warp unsupported, please use --sync=fast or --sync=fast-unsafe".to_string(),
182            ));
183        }
184
185        let bitcoin_network = run.common_params.bitcoin_network();
186        let import_config = run.common_params.import_config(run.common_params.dev);
187        let no_finalizer = run.no_finalizer;
188
189        let subcoin_service::NodeComponents {
190            client,
191            backend,
192            mut task_manager,
193            telemetry,
194            ..
195        } = subcoin_service::new_node(subcoin_service::SubcoinConfiguration {
196            network: bitcoin_network,
197            config: &config,
198            no_hardware_benchmarks: run.no_hardware_benchmarks,
199            storage_monitor,
200        })?;
201
202        let chain_info = client.usage_info().chain;
203
204        tracing::info!("ðŸ“¦ Highest known block at #{}", chain_info.best_number);
205
206        let spawn_handle = task_manager.spawn_handle();
207
208        let bitcoin_block_import =
209            BitcoinBlockImporter::<_, _, _, _, subcoin_service::TransactionAdapter>::new(
210                client.clone(),
211                client.clone(),
212                import_config,
213                Arc::new(subcoin_service::CoinStorageKey),
214                config.prometheus_registry(),
215            );
216
217        let import_queue = sc_consensus_nakamoto::bitcoin_import_queue(
218            &task_manager.spawn_essential_handle(),
219            bitcoin_block_import,
220        );
221
222        let (system_rpc_tx, substrate_sync_service) = match config.network.network_backend {
223            sc_network::config::NetworkBackendType::Libp2p => {
224                subcoin_service::start_substrate_network::<
225                    sc_network::NetworkWorker<
226                        subcoin_runtime::interface::OpaqueBlock,
227                        <subcoin_runtime::interface::OpaqueBlock as sp_runtime::traits::Block>::Hash,
228                    >,
229                >(
230                    &mut config,
231                    client.clone(),
232                    backend,
233                    &mut task_manager,
234                    bitcoin_network,
235                    telemetry,
236                )?
237            }
238            sc_network::config::NetworkBackendType::Litep2p => {
239                subcoin_service::start_substrate_network::<sc_network::Litep2pNetworkBackend>(
240                    &mut config,
241                    client.clone(),
242                    backend,
243                    &mut task_manager,
244                    bitcoin_network,
245                    telemetry,
246                )?
247            }
248        };
249
250        let subcoin_network_config = run.subcoin_network_config(bitcoin_network);
251
252        let network_api: Arc<dyn NetworkApi> =
253            if matches!(run.subcoin_network, SubcoinNetworkOption::None) {
254                task_manager.keep_alive(import_queue);
255                Arc::new(subcoin_network::NoNetwork)
256            } else {
257                let network_handle = subcoin_network::build_network(
258                    client.clone(),
259                    subcoin_network_config,
260                    import_queue,
261                    &task_manager,
262                    config.prometheus_registry().cloned(),
263                    Some(substrate_sync_service.clone()),
264                )
265                .await
266                .map_err(|err| sc_cli::Error::Application(Box::new(err)))?;
267
268                Arc::new(network_handle)
269            };
270
271        let transaction_indexer: Arc<dyn TransactionIndex + Send + Sync> = if run.tx_index {
272            let transaction_indexer = subcoin_indexer::TransactionIndexer::<
273                _,
274                _,
275                _,
276                subcoin_service::TransactionAdapter,
277            >::new(
278                bitcoin_network,
279                client.clone(),
280                task_manager.spawn_handle(),
281            )
282            .map_err(|err| sp_blockchain::Error::Application(Box::new(err)))?;
283            spawn_handle.spawn("tx-index", None, transaction_indexer.run());
284            Arc::new(subcoin_indexer::TransactionIndexProvider::new(
285                client.clone(),
286            ))
287        } else {
288            Arc::new(subcoin_primitives::NoTransactionIndex)
289        };
290
291        // Start JSON-RPC server.
292        let gen_rpc_module = || {
293            let system_info = sc_rpc::system::SystemInfo {
294                chain_name: config.chain_spec.name().into(),
295                impl_name: config.impl_name.clone(),
296                impl_version: config.impl_version.clone(),
297                properties: config.chain_spec.properties(),
298                chain_type: config.chain_spec.chain_type(),
299            };
300            let system_rpc_tx = system_rpc_tx.clone();
301
302            crate::rpc::gen_rpc_module(
303                system_info,
304                client.clone(),
305                task_manager.spawn_handle(),
306                system_rpc_tx,
307                network_api.clone(),
308                transaction_indexer.clone(),
309            )
310        };
311
312        let rpc = sc_service::start_rpc_servers(
313            &config.rpc,
314            config.prometheus_registry(),
315            &config.tokio_handle,
316            gen_rpc_module,
317            None,
318        )?;
319        task_manager.keep_alive((config.base_path.clone(), rpc));
320
321        if !no_finalizer {
322            spawn_handle.spawn("finalizer", None, {
323                subcoin_service::SubcoinFinalizer::new(
324                    client.clone(),
325                    spawn_handle.clone(),
326                    CONFIRMATION_DEPTH,
327                    network_api.clone(),
328                    Some(substrate_sync_service),
329                )
330                .run()
331            });
332        }
333
334        // Spawn subcoin informant task.
335        spawn_handle.spawn(
336            "subcoin-informant",
337            None,
338            subcoin_informant::build(client.clone(), network_api),
339        );
340
341        Ok(task_manager)
342    }
343}
