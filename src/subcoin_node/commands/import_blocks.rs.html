23/// Custom value parser to handle both file paths and raw block data in hex format.
24fn parse_raw_block(input: &str) -> Result<String, String> {
25    // Try to treat the input as a file path
26    let path = PathBuf::from(input);
27
28    if path.exists() && path.is_file() {
29        // If the file exists, read it as raw block data in hex.
30        match std::fs::read_to_string(&path) {
31            Ok(data) => Ok(data.trim().to_string()),
32            Err(err) => Err(format!(
33                "Failed to read block data from {}: {err}",
34                path.display()
35            )),
36        }
37    } else {
38        // Otherwise, treat the input as a hex string
39        Ok(input.to_string())
40    }
41}

43/// Import Bitcoin blocks into the node.
44#[derive(clap::Parser, Debug, Clone)]
45pub struct ImportBlocks {
46    /// Path to the bitcoind database.
47    ///
48    /// This corresponds to the value of the `-data-dir` argument in the bitcoind program.
49    ///
50    /// The blocks will be fetched from remote using the blockstream API if this argument
51    /// is not specified. Note that using the remote source is only for testing purpose.
52    #[clap(long, value_parser)]
53    pub data_dir: Option<PathBuf>,
54
55    /// Number of blocks to import.
56    ///
57    /// The process will stop after importing the specified number of blocks.
58    #[clap(long)]
59    pub block_count: Option<usize>,
60
61    /// Block number of last block to import.
62    ///
63    /// The default value is to the highest block in the database.
64    #[clap(long)]
65    pub end_block: Option<usize>,
66
67    /// Imports a single block into the node.
68    ///
69    /// The value can be the raw block data in hex or a path containing the block data.
70    #[clap(long, value_parser = parse_raw_block, conflicts_with = "data_dir")]
71    pub raw_block: Option<String>,
72
73    /// Whether to execute the transactions within the blocks.
74    #[clap(long, default_value_t = true)]
75    pub execute_transactions: bool,
76
77    #[allow(missing_docs)]
78    #[clap(flatten)]
79    pub common_params: CommonParams,
80
81    #[allow(missing_docs)]
82    #[clap(flatten)]
83    pub prometheus_params: PrometheusParams,
84
85    #[allow(missing_docs)]
86    #[clap(flatten)]
87    pub import_params: ImportParams,
88}

99impl ImportBlocksCmd {
100    /// Constructs a new instance of [`ImportBlocksCmd`].
101    pub fn new(cmd: &ImportBlocks) -> Self {
102        let shared_params = cmd.common_params.as_shared_params();
103        let import_params = cmd.import_params.clone();
104        Self {
105            shared_params,
106            import_params,
107            block_count: cmd.block_count,
108            to: cmd.end_block,
109            raw_block: cmd.raw_block.clone(),
110        }
111    }
112
113    /// Run the import-blocks command
114    pub async fn run(
115        &self,
116        client: Arc<FullClient>,
117        maybe_data_dir: Option<PathBuf>,
118        import_config: ImportConfig,
119        spawn_handle: SpawnTaskHandle,
120        maybe_prometheus_config: Option<PrometheusConfig>,
121    ) -> sc_cli::Result<()> {
122        // Import single block.
123        if let Some(block_data) = &self.raw_block {
124            let raw_block =
125                hex::decode(block_data).map_err(|err| format!("Invalid hex data: {err}"))?;
126            let block = bitcoin::Block::consensus_decode(&mut raw_block.as_slice())
127                .expect("Bad block data");
128
129            let mut bitcoin_block_import =
130                BitcoinBlockImporter::<_, _, _, _, subcoin_service::TransactionAdapter>::new(
131                    client.clone(),
132                    client.clone(),
133                    import_config,
134                    Arc::new(subcoin_service::CoinStorageKey),
135                    maybe_prometheus_config
136                        .as_ref()
137                        .map(|config| config.registry.clone())
138                        .as_ref(),
139                );
140
141            let block_hash = block.block_hash();
142
143            bitcoin_block_import
144                .import_block(block, BlockOrigin::Own)
145                .await
146                .map_err(sp_blockchain::Error::Consensus)?;
147
148            tracing::info!("Imported block#{block_hash} successfully");
149
150            return Ok(());
151        }
152
153        self.import_blocks(
154            client,
155            maybe_data_dir,
156            import_config,
157            spawn_handle,
158            maybe_prometheus_config,
159        )
160        .await
161    }
162
163    /// Imports a range of blocks.
164    async fn import_blocks(
165        &self,
166        client: Arc<FullClient>,
167        maybe_data_dir: Option<PathBuf>,
168        import_config: ImportConfig,
169        spawn_handle: SpawnTaskHandle,
170        maybe_prometheus_config: Option<PrometheusConfig>,
171    ) -> sc_cli::Result<()> {
172        let block_provider = BitcoinBlockProvider::new(maybe_data_dir)?;
173
174        let max = block_provider.block_count().await;
175        let to = self.to.unwrap_or(max).min(max);
176
177        let from = (client.info().best_number + 1) as usize;
178
179        tracing::info!("Start to import blocks from #{from} to #{to}",);
180
181        const INTERVAL: Duration = Duration::from_secs(1);
182
183        // The last time `display` or `new` has been called.
184        let mut last_update = Instant::now();
185
186        // Head of chain block number from the last time `display` has been called.
187        // `None` if `display` has never been called.
188        let mut last_number: Option<NumberFor<OpaqueBlock>> = None;
189
190        let mut total_imported = 0;
191
192        let mut bitcoin_block_import =
193            BitcoinBlockImporter::<_, _, _, _, subcoin_service::TransactionAdapter>::new(
194                client.clone(),
195                client.clone(),
196                import_config,
197                Arc::new(subcoin_service::CoinStorageKey),
198                maybe_prometheus_config
199                    .as_ref()
200                    .map(|config| config.registry.clone())
201                    .as_ref(),
202            );
203
204        if let Some(PrometheusConfig { port, registry }) = maybe_prometheus_config {
205            spawn_handle.spawn(
206                "prometheus-endpoint",
207                None,
208                substrate_prometheus_endpoint::init_prometheus(port, registry).map(drop),
209            );
210        }
211
212        for index in from..=to {
213            let block = block_provider.block_at(index).await?;
214            bitcoin_block_import
215                .import_block(block, BlockOrigin::Own)
216                .await
217                .map_err(sp_blockchain::Error::Consensus)?;
218
219            let now = Instant::now();
220            let interval_elapsed = now > last_update + INTERVAL;
221
222            if total_imported % 1000 == 0 || interval_elapsed {
223                if total_imported > 0 {
224                    let info = client.info();
225
226                    let best_number = info.best_number;
227                    let substrate_block_hash = info.best_hash;
228
229                    let bitcoin_block_hash =
230                        BackendExt::<OpaqueBlock>::bitcoin_block_hash_for(&client, info.best_hash)
231                            .unwrap_or_else(|| {
232                                panic!(
233                                    "Bitcoin block hash for substrate#{best_number},{substrate_block_hash} is missing",
234                                )
235                            });
236
237                    let speed = calculate_import_speed::<OpaqueBlock>(
238                        best_number,
239                        last_number,
240                        last_update,
241                    );
242
243                    tracing::info!(
244                        "Imported {total_imported} blocks,{speed}, best#{best_number},{bitcoin_block_hash} ({substrate_block_hash})",
245                    );
246
247                    last_number.replace(best_number);
248                    last_update = now;
249
250                    // Yield here allows to make the process actually interruptible by ctrl_c.
251                    Yield::new().await;
252                } else if interval_elapsed {
253                    tracing::info!("Imported {total_imported} blocks");
254                    last_update = now;
255                }
256            }
257
258            total_imported += 1;
259
260            if let Some(block_count) = self.block_count {
261                if total_imported == block_count {
262                    break;
263                }
264            }
265        }
266
267        tracing::info!("Imported {total_imported} blocks successfully");
268
269        Ok(())
270    }
271}

273/// Calculates `(best_number - last_number) / (now - last_update)` and returns a `String`
274/// representing the speed of import.
275fn calculate_import_speed<B: BlockT>(
276    best_number: NumberFor<B>,
277    last_number: Option<NumberFor<B>>,
278    last_update: Instant,
279) -> String {
280    // Number of milliseconds elapsed since last time.
281    let elapsed_ms = {
282        let elapsed = last_update.elapsed();
283        let since_last_millis = elapsed.as_secs() * 1000;
284        let since_last_subsec_millis = elapsed.subsec_millis() as u64;
285        since_last_millis + since_last_subsec_millis
286    };
287
288    // Number of blocks that have been imported since last time.
289    let diff = match last_number {
290        None => return String::new(),
291        Some(n) => best_number.saturating_sub(n),
292    };
293
294    if let Ok(diff) = TryInto::<u128>::try_into(diff) {
295        // If the number of blocks can be converted to a regular integer, then it's easy: just
296        // do the math and turn it into a `f64`.
297        let speed = diff
298            .saturating_mul(10_000)
299            .checked_div(u128::from(elapsed_ms))
300            .map_or(0.0, |s| s as f64)
301            / 10.0;
302        format!(" {speed:4.1} bps")
303    } else {
304        // If the number of blocks can't be converted to a regular integer, then we need a more
305        // algebraic approach and we stay within the realm of integers.
306        let one_thousand = NumberFor::<B>::from(1_000u32);
307        let elapsed =
308            NumberFor::<B>::from(<u32 as TryFrom<_>>::try_from(elapsed_ms).unwrap_or(u32::MAX));
309
310        let speed = diff
311            .saturating_mul(one_thousand)
312            .checked_div(&elapsed)
313            .unwrap_or_else(Zero::zero);
314        format!(" {speed} bps")
315    }
316}

332enum BitcoinBlockProvider {
333    /// Local bitcoind database.
334    Local(BitcoinDB),
335    /// Remote source.
336    #[cfg(feature = "remote-import")]
337    Remote(BlockstreamClient),
338}
339
340impl BitcoinBlockProvider {
341    fn new(maybe_data_dir: Option<PathBuf>) -> sc_cli::Result<Self> {
342        match maybe_data_dir {
343            Some(data_dir) => {
344                tracing::info!("Using local bitcoind database: {}", data_dir.display());
345                let db = BitcoinDB::new(data_dir.as_ref(), true)
346                    .map_err(|err| sc_cli::Error::Application(Box::new(err)))?;
347                Ok(Self::Local(db))
348            }
349            #[cfg(feature = "remote-import")]
350            None => {
351                tracing::info!("Using remote block provider.");
352                Ok(Self::Remote(BlockstreamClient::new()))
353            }
354            #[cfg(not(feature = "remote-import"))]
355            None => Err(
356                "Remote block import source can be enabled with `--features remote-import`".into(),
357            ),
358        }
359    }
360
361    async fn block_at(&self, height: usize) -> sc_cli::Result<bitcoin::Block> {
362        match self {
363            Self::Local(db) => {
364                let raw_block = db.get_raw_block(height).map_err(|err| {
365                    std::io::Error::other(format!(
366                        "Failed to get bitcoin block at #{height}: {err}"
367                    ))
368                })?;
369
370                Ok(bitcoin::Block::consensus_decode(&mut raw_block.as_slice())
371                    .expect("Bad block in the database"))
372            }
373            #[cfg(feature = "remote-import")]
374            Self::Remote(rpc_client) => rpc_client
375                .get_block_by_height(height as u32)
376                .await
377                .map_err(|err| sc_cli::Error::Application(Box::new(err))),
378        }
379    }
380
381    async fn block_count(&self) -> usize {
382        match self {
383            Self::Local(db) => db.get_block_count(),
384            #[cfg(feature = "remote-import")]
385            Self::Remote(rpc_client) => rpc_client
386                .get_tip_height()
387                .await
388                .expect("Failed to fetch tip height")
389                as usize,
390        }
391    }
