1use bitcoin::blockdata::script::Script;
2use bitcoin::hashes::{Hash, sha256d};
3
4/// Utilities
5#[derive(Debug, clap::Subcommand)]
6pub enum Tools {
7    /// Revert a sha256 hash.
8    ///
9    /// Unlike the sha256 hash are displayed in the reversed format in the Bitcoin tools,
10    /// H256 on polkadot.js.org are not. This is command is useful when you copy a hash
11    /// from Bitcoin and want to paste it in polkadot.js.org.
12    #[command(name = "revert-h256")]
13    RevertH256 {
14        #[arg(index = 1)]
15        h256: String,
16    },
17
18    /// Parse the raw scriptPubkey of UTXO in the runtime.
19    ///
20    /// You can inspect the chain state from polkadot.js.org, copy the `scriptPubkey`
21    /// in the storage and use this command to convert it to the human-readable
22    /// Bitcoin Script format.
23    #[command(name = "parse-script-pubkey-in-storage")]
24    ParseScriptPubkeyInStorage {
25        #[arg(index = 1)]
26        input: String,
27    },
28}
29
30fn revert_sha256d(h256d: &str) -> sc_cli::Result<String> {
31    let hash: sha256d::Hash = h256d
32        .strip_prefix("0x")
33        .unwrap_or(h256d)
34        .parse()
35        .map_err(|err| sc_cli::Error::Input(format!("Invalid h256: {err}")))?;
36
37    Ok(hash
38        .as_byte_array()
39        .iter()
40        .map(|byte| format!("{byte:02x}"))
41        .collect::<Vec<_>>()
42        .join(""))
43}
44
45impl Tools {
46    pub fn run(self) -> sc_cli::Result<()> {
47        match self {
48            Self::RevertH256 { h256 } => {
49                println!("Reverted: {}", revert_sha256d(&h256)?);
50            }
51            Self::ParseScriptPubkeyInStorage { input } => {
52                let str_without_0x = input.strip_prefix("0x").unwrap_or(&input);
53
54                let hex_bytes = hex::decode(str_without_0x)
55                    .map_err(|err| sc_cli::Error::Input(format!("Invalid input: {err}")))?;
56
57                let script = Script::from_bytes(&hex_bytes);
58                println!("{script:?}");
59            }
60        }
61        Ok(())
62    }
63}
