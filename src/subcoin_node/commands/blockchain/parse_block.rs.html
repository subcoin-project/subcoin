1use crate::commands::blockchain::{ClientParams, MergedParams};
2use sc_client_api::{BlockBackend, HeaderBackend};
3use std::sync::Arc;
4use subcoin_service::FullClient;
5
6#[derive(Debug, clap::Parser)]
7pub struct ParseBlock {
8    /// Specify the number of block to dump.
9    ///
10    /// Defaults to the best block.
11    #[clap(long)]
12    height: Option<u32>,
13
14    #[allow(missing_docs)]
15    #[clap(flatten)]
16    client_params: ClientParams,
17}
18
19pub struct ParseBlockCmd {
20    height: Option<u32>,
21    pub(super) params: MergedParams,
22}
23
24impl ParseBlockCmd {
25    pub fn execute(self, client: Arc<FullClient>) -> sc_cli::Result<()> {
26        let block_number = self.height.unwrap_or_else(|| client.info().best_number);
27
28        let block_hash = client.hash(block_number)?.ok_or_else(|| {
29            sp_blockchain::Error::Backend(format!("Hash for {block_number} not found"))
30        })?;
31
32        let substrate_block = client.block(block_hash)?.ok_or_else(|| {
33            sp_blockchain::Error::Backend(format!(
34                "Block for #{block_number},{block_hash} not found"
35            ))
36        })?;
37
38        let btc_block = subcoin_primitives::convert_to_bitcoin_block::<
39            subcoin_runtime::interface::OpaqueBlock,
40            subcoin_service::TransactionAdapter,
41        >(substrate_block.block)
42        .expect("Failed to convert to Bitcoin Block");
43
44        println!(
45            "block_size: {}",
46            bitcoin::consensus::serialize(&btc_block).len()
47        );
48
49        let mut num_op_return = 0;
50
51        for (i, tx) in btc_block.txdata.into_iter().enumerate() {
52            for (j, output) in tx.output.into_iter().enumerate() {
53                let is_op_return = output.script_pubkey.is_op_return();
54                println!("{i}:{j}: {is_op_return:?}");
55
56                if is_op_return {
57                    num_op_return += 1;
58                }
59            }
60        }
61
62        println!("There are {num_op_return} OP_RETURN in block #{block_number},{block_hash}");
63
64        Ok(())
65    }
66}
67
68impl From<ParseBlock> for ParseBlockCmd {
69    fn from(parse_block: ParseBlock) -> Self {
70        let ParseBlock {
71            height,
72            client_params,
73        } = parse_block;
74        Self {
75            height,
76            params: client_params.into_merged_params(),
77        }
78    }
