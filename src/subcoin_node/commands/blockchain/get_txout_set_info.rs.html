80async fn gettxoutsetinfo(
81    client: &Arc<FullClient>,
82    height: Option<u32>,
83    verbose: bool,
84    progress_bar: bool,
85) -> sc_cli::Result<TxOutSetInfo> {
86    let (block_number, bitcoin_block_hash, utxo_iter) = fetch_utxo_set_at(client, height)?;
87
88    let substrate_block_hash = client.hash(block_number)?.ok_or_else(|| {
89        sp_blockchain::Error::Backend(format!("Hash for #{block_number} not found"))
90    })?;
91
92    let total_coins = client
93        .runtime_api()
94        .coins_count(substrate_block_hash)
95        .map_err(|err| sc_cli::Error::from(sp_blockchain::Error::RuntimeApiError(err)))?;
96
97    const INTERVAL: Duration = Duration::from_secs(5);
98
99    let txouts = Arc::new(AtomicUsize::new(0));
100    let mut bogosize = 0;
101    let mut total_amount = 0;
102
103    let mut script_pubkey_size = 0;
104
105    let mut last_update = Instant::now();
106
107    let mut muhash = subcoin_crypto::muhash::MuHash3072::new();
108
109    if progress_bar {
110        crate::utils::show_progress_in_background(
111            txouts.clone(),
112            total_coins,
113            format!("Loading UTXO set at block #{block_number}..."),
114        );
115    }
116
117    for (txid, vout, coin) in utxo_iter {
118        let data = subcoin_utxo_snapshot::tx_out_ser(bitcoin::OutPoint { txid, vout }, &coin)
119            .map_err(|err| sc_cli::Error::Application(Box::new(err)))?;
120
121        muhash.insert(&data);
122
123        let old = txouts.fetch_add(1, Ordering::SeqCst);
124
125        total_amount += coin.amount;
126        // https://github.com/bitcoin/bitcoin/blob/33af14e31b9fa436029a2bb8c2b11de8feb32f86/src/kernel/coinstats.cpp#L40
127        bogosize += 50 + coin.script_pubkey.len();
128
129        script_pubkey_size += coin.script_pubkey.len();
130
131        if verbose && last_update.elapsed() > INTERVAL {
132            println!(
133                "Progress: Unspent Transaction Outputs: {}, \
134                ScriptPubkey Size: {script_pubkey_size} bytes, Coin ScriptPubkey Length: {} bytes",
135                old + 1,
136                coin.script_pubkey.len()
137            );
138            last_update = Instant::now();
139        }
140
141        // Yield here allows to make the process interruptible by ctrl_c.
142        Yield::new().await;
143    }
144
145    // Hash the combined hash of all UTXOs
146    let muhash = muhash.txoutset_muhash();
147
148    let tx_out_set_info = TxOutSetInfo {
149        height: block_number,
150        bestblock: bitcoin_block_hash,
151        txouts: txouts.load(Ordering::SeqCst),
152        bogosize,
153        muhash,
154        total_amount,
155    };
156
157    Ok(tx_out_set_info)
158}

160// Custom serializer for total_amount to display 8 decimal places
161fn serialize_as_btc<S>(amount: &u64, serializer: S) -> Result<S::Ok, S::Error>
162where
163    S: serde::Serializer,
164{
165    // Convert satoshis (u64) to BTC (f64)
166    let btc_value = *amount as f64 / 100_000_000.0;
167    // Format the value as a string with 8 decimal places
168    serializer.serialize_str(&format!("{btc_value:.8}"))
169}
170
171#[derive(serde::Serialize)]
172struct TxOutSetInfo {
173    height: u32,
174    bestblock: bitcoin::BlockHash,
175    txouts: usize,
176    bogosize: usize,
177    muhash: String,
178    #[serde(serialize_with = "serialize_as_btc")]
179    total_amount: u64,
180}
