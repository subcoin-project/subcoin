49/// header weight (80 * 4) + tx_data_len(4)
50const BITCOIN_BASE_BLOCK_WEIGHT: u64 = 80 * 4 + 4;
51
52/// Maximum block weight.
53/// https://github.com/bitcoin/bips/blob/master/bip-0141.mediawiki#Block_size
54const BITCOIN_MAX_WEIGHT: u64 = 4_000_000;
55
56#[runtime_version]
57pub const VERSION: RuntimeVersion = RuntimeVersion {
58    spec_name: alloc::borrow::Cow::Borrowed("subcoin"),
59    impl_name: alloc::borrow::Cow::Borrowed("subcoin"),
60    authoring_version: 0,
61    spec_version: 0,
62    impl_version: 0,
63    apis: RUNTIME_API_VERSIONS,
64    transaction_version: 0,
65    system_version: 1,
66};
67
68/// The version information used to identify this runtime when compiled natively.
69#[cfg(feature = "std")]
70pub fn native_version() -> NativeVersion {
71    NativeVersion {
72        runtime_version: VERSION,
73        can_author_with: Default::default(),
74    }
75}

104/// Subcoin block weights.
105pub struct BlockWeights;
106
107impl Get<frame_system::limits::BlockWeights> for BlockWeights {
108    fn get() -> frame_system::limits::BlockWeights {
109        frame_system::limits::BlockWeights {
110            base_block: Weight::from_parts(BITCOIN_BASE_BLOCK_WEIGHT, 0u64),
111            max_block: Weight::from_parts(BITCOIN_MAX_WEIGHT, u64::MAX),
112            per_class: PerDispatchClass::new(|class| {
113                let initial = if class == DispatchClass::Mandatory {
114                    None
115                } else {
116                    Some(Weight::zero())
117                };
118                WeightsPerClass {
119                    base_extrinsic: Weight::zero(),
120                    max_extrinsic: None,
121                    max_total: initial,
122                    reserved: initial,
123                }
124            }),
125        }
126    }
127}

250/// A set of opinionated types aliases commonly used in runtimes.
251///
252/// This is one set of opinionated types. They are compatible with one another, but are not
253/// guaranteed to work if you start tweaking a portion.
254///
255/// Some note-worthy opinions in this prelude:
256///
257/// - `u32` block number.
258/// - [`sp_runtime::MultiAddress`] and [`sp_runtime::MultiSignature`] are used as the account id
259///   and signature types. This implies that this prelude can possibly used with an
260///   "account-index" system (eg `pallet-indices`). And, in any case, it should be paired with
261///   `AccountIdLookup` in [`frame_system::Config::Lookup`].
262mod types_common {
263    use frame_system::Config as SysConfig;
264    use sp_runtime::{OpaqueExtrinsic, generic, traits};
265
266    /// A signature type compatible capably of handling multiple crypto-schemes.
267    pub type Signature = sp_runtime::MultiSignature;
268
269    /// The corresponding account-id type of [`Signature`].
270    pub type AccountId =
271        <<Signature as traits::Verify>::Signer as traits::IdentifyAccount>::AccountId;
272
273    /// The block-number type, which should be fed into [`frame_system::Config`].
274    pub type BlockNumber = u32;
275
276    /// TODO: Ideally we want the hashing type to be equal to SysConfig::Hashing?
277    type HeaderInner = generic::Header<BlockNumber, traits::BlakeTwo256>;
278
279    // NOTE: `AccountIndex` is provided for future compatibility, if you want to introduce
280    // something like `pallet-indices`.
281    type ExtrinsicInner<T, Extra, AccountIndex = ()> = generic::UncheckedExtrinsic<
282        sp_runtime::MultiAddress<AccountId, AccountIndex>,
283        <T as SysConfig>::RuntimeCall,
284        Signature,
285        Extra,
286    >;
287
288    /// The block type, which should be fed into [`frame_system::Config`].
289    ///
290    /// Should be parameterized with `T: frame_system::Config` and a tuple of `SignedExtension`.
291    /// When in doubt, use [`SystemSignedExtensionsOf`].
292    // Note that this cannot be dependent on `T` for block-number because it would lead to a
293    // circular dependency (self-referential generics).
294    pub type BlockOf<T, Extra = ()> = generic::Block<HeaderInner, ExtrinsicInner<T, Extra>>;
295
296    /// The opaque block type. This is the same `BlockOf`, but it has
297    /// [`sp_runtime::OpaqueExtrinsic`] as its final extrinsic type.
298    ///
299    /// This should be provided to the client side as the extrinsic type.
300    pub type OpaqueBlock = generic::Block<HeaderInner, OpaqueExtrinsic>;
301}
